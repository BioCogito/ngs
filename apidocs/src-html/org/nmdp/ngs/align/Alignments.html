062    // all ranges here are 0-based [closed, open)
063
064    /**
065     * Confirm that the specified range is [closed, open).
066     *
067     * @param range range to check, must not be null
068     */
069    static void checkClosedOpen(final Range<Long> range) {
070        checkNotNull(range);
071        checkArgument(BoundType.CLOSED == range.lowerBoundType(), "range must be [closed, open), lower bound type was open");
072        checkArgument(BoundType.OPEN == range.upperBoundType(), "range must be [closed, open), upper bound type was closed");
073    }
074
075    /**
076     * Return the length of the specified range.
077     *
078     * @param range range, must not be null and must be [closed, open)
079     * @return the length of the specified range
080     */
081    public static long length(final Range<Long> range) {
082        checkClosedOpen(range);
083        return Math.max(0L, range.upperEndpoint() - range.lowerEndpoint());
084    }
085
086    /**
087     * Return the lengths of the specified ranges.
088     *
089     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
090     * @return the lengths of the specified ranges
091     */
092    public static List<Long> lengths(final Iterable<Range<Long>> ranges) {
093        checkNotNull(ranges);
094        List<Long> lengths = new ArrayList<Long>();
095        for (Range<Long> range : ranges) {
096            lengths.add(length(range));
097        }
098        return lengths;
099    }
100
101    /**
102     * Return the sum of lengths of the specified ranges, after merging overlapping ranges.
103     *
104     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
105     * @return the sum of lengths of the specified ranges, after merging overlapping ranges
106     */
107    public static long length(final Iterable<Range<Long>> ranges) {
108        checkNotNull(ranges);
109        RangeSet<Long> rangeSet = TreeRangeSet.create();
110        for (Range<Long> range : ranges) {
111            rangeSet.add(range);
112        }
113        long length = 0L;
114        for (Range<Long> range : rangeSet.asRanges()) {
115            length += length(range);
116        }
117        return length;
118    }
119
120    /**
121     * Return the maximum length in the specified ranges, or <code>-1</code> if ranges is empty.
122     *
123     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
124     * @return the maximum length in the specified ranges, or <code>-1</code> if ranges is empty
125     */
126    public static long maximumLength(final Iterable<Range<Long>> ranges) {
127        checkNotNull(ranges);
128        if (Iterables.isEmpty(ranges)) {
129            return -1L;
130        }
131        return Ordering.natural().max(lengths(ranges)).longValue();
132    }
133
134    /**
135     * Return the count of the specified ranges.
136     *
137     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
138     * @return the count of the specified ranges
139     */
140    public static int count(final Iterable<Range<Long>> ranges) {
141        int count = 0;
142        for (Range<Long> range : ranges) {
143            checkClosedOpen(range);
144            count++;
145        }
146        return count;
147    }
148
149    /**
150     * Return the start/lower endpoints in the specified ranges.
151     *
152     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
153     * @return the start/lower endpoints in the specified ranges
154     */
155    public static List<Long> starts(final Iterable<Range<Long>> ranges) {
156        List<Long> starts = new ArrayList<Long>();
157        for (Range<Long> range : ranges) {
158            checkClosedOpen(range);
159            starts.add(range.lowerEndpoint());
160        }
161        return starts;
162    }
163
164    /**
165     * Return the end/upper endpoints in the specified ranges.
166     *
167     * @param ranges ranges, must not be null, must not contain any null ranges, and all ranges must be [closed, open)
168     * @return the end/upper endpoints in the specified ranges
169     */
170    public static List<Long> ends(final Iterable<Range<Long>> ranges) {
171        List<Long> ends = new ArrayList<Long>();
172        for (Range<Long> range : ranges) {
173            checkClosedOpen(range);
174            ends.add(range.upperEndpoint());
175        }
176        return ends;
177    }

179    /**
180     * Return true if the specified symbol is a gap symbol.
181     *
182     * @param symbol symbol
183     * @return true if the specified symbol is a gap symbol
184     */
185    static boolean isGapSymbol(final Symbol symbol) {
186        return AlphabetManager.getGapSymbol().equals(symbol)
187            || DNATools.getDNA().getGapSymbol().equals(symbol);
188    }
189
190    /**
191     * Return true if the specified symbol represents an alignment match.
192     *
193     * @param symbol symbol
194     * @return true if the specified symbol represents an alignment match
195     */
196    static boolean isMatchSymbol(final Symbol symbol) {
197        if (!(symbol instanceof BasisSymbol)) {
198            return false;
199        }
200        BasisSymbol basisSymbol = (BasisSymbol) symbol;
201        Set<Symbol> uniqueSymbols = new HashSet<Symbol>();
202        for (Object o : basisSymbol.getSymbols()) {
203            Symbol s = (Symbol) o;
204            if (isGapSymbol(s)) {
205                return false;
206            }
207            uniqueSymbols.add((Symbol) o);
208        }
209        return (uniqueSymbols.size() == 1);
210    }
211
212    /**
213     * Return true if the specified symbol represents an alignment mismatch.
214     *
215     * @param symbol symbol
216     * @return true if the specified symbol represents an alignment mismatch
217     */
218    static boolean isMismatchSymbol(final Symbol symbol) {
219        if (!(symbol instanceof BasisSymbol)) {
220            return false;
221        }
222        BasisSymbol basisSymbol = (BasisSymbol) symbol;
223        Set<Symbol> uniqueSymbols = new HashSet<Symbol>();
224        for (Object o : basisSymbol.getSymbols()) {
225            Symbol s = (Symbol) o;
226            if (isGapSymbol(s)) {
227                return false;
228            }
229            uniqueSymbols.add((Symbol) o);
230        }
231        return (uniqueSymbols.size() > 1);
232    }
233
234    /**
235     * Return the gaps in the specified gapped symbol list as 0-based [closed, open) ranges.
236     *
237     * @param gappedSymbols gapped symbol list, must not be null
238     * @return the gaps in the specified gapped symbol list as 0-based [closed, open) ranges
239     */
240    public static List<Range<Long>> gaps(final GappedSymbolList gappedSymbols) {
241        checkNotNull(gappedSymbols);
242        List<Range<Long>> gaps = new ArrayList<Range<Long>>();
243        int gapStart = -1;
244        for (int i = 1, length = gappedSymbols.length() + 1; i < length; i++) {
245            if (isGapSymbol(gappedSymbols.symbolAt(i))) {
246                if (gapStart < 0) {
247                    gapStart = i;
248                }
249            }
250            else {
251                if (gapStart > 0) {
252                    // biojava coordinates are 1-based
253                    gaps.add(Range.closedOpen(Long.valueOf(gapStart - 1L), Long.valueOf(i - 1L)));
254                    gapStart = -1;
255                }
256            }
257        }
258        if (gapStart > 0) {
259            gaps.add(Range.closedOpen(Long.valueOf(gapStart - 1L), Long.valueOf(gappedSymbols.length())));
260        }
261        return gaps;
262    }

264    /**
265     * Return the alignment matches in the specified alignment pair as 0-based [closed, open) ranges.
266     *
267     * @param alignmentPair alignment pair, must not be null
268     * @return the alignment matches in the alignment pair as 0-based [closed, open) ranges
269     */
270    public static List<Range<Long>> matches(final AlignmentPair alignmentPair) {
271        checkNotNull(alignmentPair);
272        List<Range<Long>> matches = new ArrayList<Range<Long>>();
273        int matchStart = -1;
274        for (int i = 1, length = alignmentPair.length() + 1; i < length; i++) {
275            if (isMatchSymbol(alignmentPair.symbolAt(i))) {
276                if (matchStart < 0) {
277                    matchStart = i;
278                }
279            }
280            else {
281                if (matchStart > 0) {
282                    // biojava coordinates are 1-based
283                    matches.add(Range.closedOpen(Long.valueOf(matchStart - 1L), Long.valueOf(i - 1L)));
284                    matchStart = -1;
285                }
286            }
287        }
288        if (matchStart > 0) {
289            matches.add(Range.closedOpen(Long.valueOf(matchStart - 1L), Long.valueOf(alignmentPair.length())));
290        }
291        return matches;
292    }

320        }
321        return mismatches;
322    }
