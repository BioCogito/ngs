023package org.nmdp.ngs.align;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.io.IOException;
028
029import java.util.LinkedList;
030import java.util.List;
031
032import com.google.common.io.CharStreams;
033import com.google.common.io.LineProcessor;
034
035/**
036 * BED format reader.
037 */
038public final class BedReader {
039
040    /**
041     * Private no-arg constructor.
042     */
043    private BedReader() {
044        // empty
045    }
046
047
048    /**
049     * Read zero or more BED records from the specified readable.
050     *
051     * @param readable to read from, must not be null
052     * @return zero or more BED records read from the specified readable
053     * @throws IOException if an I/O error occurs
054     */
055    public static Iterable<BedRecord> read(final Readable readable) throws IOException {
056        checkNotNull(readable);
057        Collect collect = new Collect();
058        stream(readable, collect);
059        return collect.records();
060    }
061
062    /**
063     * Stream zero or more BED records from the specified readable.
064     *
065     * @param readable readable to stream from, must not be null
066     * @param listener event based listener callback, must not be null
067     * @throws IOException if an I/O error occurs
068     */
069    public static void stream(final Readable readable, final BedListener listener) throws IOException {
070        checkNotNull(readable);
071        checkNotNull(listener);
072
073        BedLineProcessor lineProcessor = new BedLineProcessor(listener);
074        CharStreams.readLines(readable, lineProcessor);
075    }
076
077    /**
078     * BED line processor.
079     */
080    private static final class BedLineProcessor implements LineProcessor<Object> {
081        /** Line number. */
082        private long lineNumber = 0;
083
084        /** BED listener. */
085        private final BedListener listener;
086
087
088        /**
089         * Create a new BED line processor with the specified BED listener.
090         *
091         * @param listener BED listener, must not be null
092         */
093        private BedLineProcessor(final BedListener listener) {
094            checkNotNull(listener);
095            this.listener = listener;
096        }
097
098
099        @Override
100        public Object getResult() {
101            return null;
102        }
103
104        @Override
105        public boolean processLine(final String line) throws IOException
106        {
107            try {
108                lineNumber++;
109                return listener.record(BedRecord.valueOf(line));
110            }
111            catch (IllegalArgumentException | NullPointerException e) {
112                throw new IOException("could not read BED record at line " + lineNumber + ", caught " + e.getMessage(), e);
113            }
114        }
115    }
116
117
118    /**
119     * Collect.
120     */
121    private static class Collect implements BedListener {
122        /** List of collected BED records. */
123        private final List<BedRecord> records = new LinkedList<BedRecord>();
124
125
126        @Override
127        public boolean record(final BedRecord record) {
128            records.add(record);
129            return true;
130        }
131
132        /**
133         * Return zero or more collected BED records.
134         *
135         * @return zero or more collected BED records
136         */
137        Iterable<BedRecord> records() {
138            return records;
139        }
140    }
141}
