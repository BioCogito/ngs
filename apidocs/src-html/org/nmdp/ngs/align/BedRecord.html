042/**
043 * BED record.
044 *
045 * <p>Supports the same BED formats as <a href="http://bedtools.readthedocs.org/en/latest/content/general-usage.html">bedtools2</a>.
046 * <ul>
047 *   <li>BED3: A BED file where each feature is described by chrom, start, and end.<br/>
048 *       For example: <code>chr1          11873   14409</code></li>
049 *   <li>BED4: A BED file where each feature is described by chrom, start, end, and name.<br/>
050 *       For example: <code>chr1  11873  14409  uc001aaa.3</code></li>
051 *   <li>BED5: A BED file where each feature is described by chrom, start, end, name, and score.<br/>
052 *       For example: <code>chr1 11873 14409 uc001aaa.3 0</code></li>
053 *   <li>BED6: A BED file where each feature is described by chrom, start, end, name, score, and strand.<br/>
054 *       For example: <code>chr1 11873 14409 uc001aaa.3 0 +</code></li>
055 *   <li>BED12: A BED file where each feature is described by all twelve columns listed above.<br/>
056 *       For example: <code>chr1 11873 14409 uc001aaa.3 0 + 11873 11873 0 3 354,109,1189, 0,739,1347,</code></li>
057 * </ul></p>
058 */
059@Immutable
060public final class BedRecord {
061    private final Format format;
062    private final String chrom;
063    private final long start;
064    private final long end;
065    private final String name;
066    private final String score;
067    private final String strand;
068    private final long thickStart;
069    private final long thickEnd;
070    private final String itemRgb;
071    private final int blockCount;
072    private final long[] blockSizes;
073    private final long[] blockStarts;
074    private final int hashCode;
075
076    /** BED format. */
077    enum Format { BED3, BED4, BED5, BED6, BED12 };
078
079    /** Empty long array. */
080    private static final long[] EMPTY = new long[0];
081
082    /** R,G,B pattern. */
083    private static final Pattern RGB = Pattern.compile("^[0-9]+,[0-9]+,[0-9]+$");
084

086    /**
087     * Create a new BED record.
088     *
089     * @param format format, must not be null
090     * @param chrom chrom, must not be null
091     * @param start start, must be at least zero
092     * @param end end, must be at least zero, and greater than or equal to start
093     * @param name name
094     * @param score score
095     * @param strand strand, if present must be either <code>-</code> or <code>+</code>
096     * @param thickStart thick start, must be at least zero
097     * @param thickEnd thick end, must be at least zero and greater than or equal to thick start
098     * @param itemRgb item RGB, if present, must be in R,G,B format, e.g. <code>255,0,0</code>
099     * @param blockCount block count, must be at least zero
100     * @param blockSizes block sizes, must be the same length as block count
101     * @param blockStarts block starts, must be the same length as block count
102     */
103    private BedRecord(final Format format, final String chrom, final long start, final long end, final String name, final String score, final String strand,
104                     final long thickStart, final long thickEnd, final String itemRgb, final int blockCount, final long[] blockSizes, final long[] blockStarts) {
105
106        checkNotNull(format);
107        checkNotNull(chrom);
108        checkArgument(start >= 0L, "start must be at least zero");
109        checkArgument(end >= 0L, "end must be at least zero");
110        checkArgument(end >= start, "end must be greater than or equal to start");
111        checkArgument(thickStart >= 0L, "thickStart must be at least zero");
112        checkArgument(thickEnd >= 0L, "thickEnd must be at least zero");
113        checkArgument(thickEnd >= thickStart, "thickEnd must be greater than or equal to thickStart");
114        checkArgument(blockCount >= 0, "blockCount must be at least zero");
115        checkNotNull(blockSizes);
116        checkNotNull(blockStarts);
117        checkArgument(blockSizes.length == blockCount, "blockSizes must be the same length as blockCount");
118        checkArgument(blockStarts.length == blockCount, "blockStarts must be the same length as blockCount");
119
120        if (strand != null) {
121            checkArgument("-".equals(strand) || "+".equals(strand), "if present, strand must be either - or +");
122        }
123        /*
124
125          appears bedtools2 doesn't enforce this format restriction
126
127        if (itemRgb != null && !RGB.matcher(itemRgb).matches()) {
128            throw new IllegalArgumentException("if present, itemRgb must be in R,G,B format, e.g. 255,0,0");
129        }
130        */
131
132        this.format = format;
133        this.chrom = chrom;
134        this.start = start;
135        this.end = end;
136        this.name = name;
137        this.score = score;
138        this.strand = strand;
139        this.thickStart = thickStart;
140        this.thickEnd = thickEnd;
141        this.itemRgb = itemRgb;
142        this.blockCount = blockCount;
143        if (blockCount > 0) {
144            this.blockSizes = (long[]) blockSizes.clone();
145            this.blockStarts = (long[]) blockStarts.clone();
146        }
147        else {
148            this.blockSizes = blockSizes;
149            this.blockStarts = blockStarts;
150        }
151
152        hashCode = Objects.hash(this.format, this.chrom, this.start, this.end, this.name, this.score, this.strand, this.thickStart, this.thickEnd, this.itemRgb, this.blockCount, this.blockSizes, this.blockStarts);
153    }

359    @Override
360    public boolean equals(final Object o) {
361        if (o == this) {
362            return true;
363        }
364        if (!(o instanceof BedRecord)) {
365            return false;
366        }
367        BedRecord bedRecord = (BedRecord) o;
368
369        return Objects.equals(format, bedRecord.format)
370            && Objects.equals(chrom, bedRecord.chrom)
371            && Objects.equals(start, bedRecord.start)
372            && Objects.equals(end, bedRecord.end)
373            && Objects.equals(name, bedRecord.name)
374            && Objects.equals(score, bedRecord.score)
375            && Objects.equals(strand, bedRecord.strand)
376            && Objects.equals(thickStart, bedRecord.thickStart)
377            && Objects.equals(thickEnd, bedRecord.thickEnd)
378            && Objects.equals(itemRgb, bedRecord.itemRgb)
379            && Objects.equals(blockCount, bedRecord.blockCount)
380            && Objects.equals(blockSizes, bedRecord.blockSizes)
381            && Objects.equals(blockStarts, bedRecord.blockStarts);
382    }

384    @Override
385    public String toString() {
386        StringBuilder sb = new StringBuilder();
387        sb.append(chrom);
388        sb.append("\t");
389        sb.append(start);
390        sb.append("\t");
391        sb.append(end);
392        switch (format) {
393            case BED3:
394                break;
395            case BED4:
396                sb.append("\t");
397                sb.append(name);
398                break;
399            case BED5:
400                sb.append("\t");
401                sb.append(name);
402                sb.append("\t");
403                sb.append(score);
404                break;
405            case BED6:
406                sb.append("\t");
407                sb.append(name);
408                sb.append("\t");
409                sb.append(score);
410                sb.append("\t");
411                sb.append(strand);
412                break;
413            case BED12:
414                sb.append("\t");
415                sb.append(name);
416                sb.append("\t");
417                sb.append(score);
418                sb.append("\t");
419                sb.append(strand);
420                sb.append("\t");
421                sb.append(thickStart);
422                sb.append("\t");
423                sb.append(thickEnd);
424                sb.append("\t");
425                sb.append(itemRgb);
426                sb.append("\t");
427                sb.append(blockCount);
428                sb.append("\t");
429                sb.append(Joiner.on(",").join(Longs.asList(blockSizes)));
430                sb.append("\t");
431                sb.append(Joiner.on(",").join(Longs.asList(blockStarts)));
432                break;
433            default:
434                break;
435        }
436        return sb.toString();
437    }

439    private static long[] parseLongArray(final String value) {
440        List<String> tokens = Splitter.on(",").trimResults().omitEmptyStrings().splitToList(value);
441        long[] longs = new long[tokens.size()];
442        for (int i = 0, size = tokens.size(); i < size; i++) {
443            longs[i] = Long.parseLong(tokens.get(i));
444        }
445        return longs;
446    }

448    /**
449     * Return a new BED record parsed from the specified value.
450     *
451     * @param value value to parse
452     * @return a new BED record parsed from the specified value
453     * @throws IllegalArgumentException if the value is not valid BED[3,4,5,6,12] format
454     * @throws NullPointerException if a required field is missing
455     * @throws NumberFormatException if a long valued field cannot be parsed as a long 
456     */
457    public static BedRecord valueOf(final String value) {
458        checkNotNull(value);
459        List<String> tokens = Splitter.on("\t").trimResults().splitToList(value);
460        if (tokens.size() < 3) {
461            throw new IllegalArgumentException("value must have at least three fields (chrom, start, end)");
462        }
463        String chrom = tokens.get(0);
464        long start = Long.parseLong(tokens.get(1));
465        long end = Long.parseLong(tokens.get(2));
466        if (tokens.size() == 3) {
467            return new BedRecord(chrom, start, end);
468        }
469        else {
470            String name = tokens.get(3);
471            if (tokens.size() == 4) {
472                return new BedRecord(chrom, start, end, name);
473            }
474            else {
475                String score = tokens.get(4);
476                if (tokens.size() == 5) {
477                    return new BedRecord(chrom, start, end, name, score);
478                }
479                else {
480                    String strand = tokens.get(5);
481                    if (tokens.size() == 6) {
482                        return new BedRecord(chrom, start, end, name, score, strand);
483                    }
484                    if (tokens.size() != 12) {
485                        throw new IllegalArgumentException("value is not in BED3, BED4, BED5, BED6 or BED12 format");
486                    }
487                    long thickStart = Long.parseLong(tokens.get(6));
488                    long thickEnd = Long.parseLong(tokens.get(7));
489                    String itemRgb = tokens.get(8);
490                    int blockCount = Integer.parseInt(tokens.get(9));
491                    long[] blockSizes = parseLongArray(tokens.get(10));
492                    long[] blockStarts = parseLongArray(tokens.get(11));
493
494                    return new BedRecord(chrom, start, end, name, score, strand, thickStart, thickEnd, itemRgb, blockCount, blockSizes, blockStarts);
495                }
496            }
497        }
498    }
