023package org.nmdp.ngs.align;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.io.BufferedReader;
028import java.io.File;
029import java.io.FileReader;
030import java.io.IOException;
031
032import java.util.List;
033
034import com.google.common.base.Splitter;
035
036import com.google.common.collect.ImmutableList;
037import com.google.common.collect.Lists;
038
039/**
040 * Utility to run genewise via the command line.
041 */
042public final class Genewise {
043    /** Split on spaces. */
044    private static final Splitter SPLITTER = Splitter.on(' ').trimResults().omitEmptyStrings();
045
046
047    /**
048     * Private no-arg constructor.
049     */
050    private Genewise() {
051        // empty
052    }
053
054
055    /**
056     * Return the exons predicted from the alignment of the specified amino acid HMM file in HMMER2 format against
057     * the specified genomic DNA sequence file in FASTA format.
058     *
059     * @param aminoAcidHmm2File amino acid HMM file in HMMER2 format, must not be null
060     * @param genomicDnaFastaFile genomic DNA sequence file in FASTA format, must not be null
061     * @return zero or more exons predicted from the alignment of the specified amino acid HMM file in HMMER2 format against
062     *    the specified genomic DNA sequence file in FASTA format
063     * @throws IOException if an I/O error occurs
064     */
065    public static Iterable<GenewiseExon> genewiseExons(final File aminoAcidHmm2File, final File genomicDnaFastaFile) throws IOException {
066        checkNotNull(aminoAcidHmm2File);
067        checkNotNull(genomicDnaFastaFile);
068
069        File genewiseResult = File.createTempFile("genewise", ".txt");
070
071        ProcessBuilder genewise = new ProcessBuilder("genewise",
072                                                     "-hmmer", "-tfor", "-genes", "-nosplice_gtag",
073                                                     aminoAcidHmm2File.getPath(), genomicDnaFastaFile.getPath());
074
075        genewise.redirectErrorStream(true);
076        genewise.redirectOutput(ProcessBuilder.Redirect.to(genewiseResult));
077
078        Process genewiseProcess = genewise.start();
079        try {
080            genewiseProcess.waitFor();
081        }
082        catch (InterruptedException e) {
083            // ignore
084        }
085
086        int lineNumber = 0;
087        BufferedReader reader = null;
088        List<GenewiseExon> exons = Lists.newLinkedList();
089        try {
090            reader = new BufferedReader(new FileReader(genewiseResult));
091            while (reader.ready()) {
092                String line = reader.readLine();
093                if (line == null) {
094                    break;
095                }
096                if (line.startsWith("  Exon")) {
097                    List<String> tokens = SPLITTER.splitToList(line);
098                    if (tokens.size() < 5) {
099                        throw new IOException("invalid genewise genes format at line number " + lineNumber + ", line " + line);
100                    }
101                    try {
102                        long start = Long.parseLong(tokens.get(1));
103                        long end = Long.parseLong(tokens.get(2));
104                        if (start > end) {
105                            throw new IOException("invalid genewise exon at line number " + lineNumber + ", start > end");
106                        }
107                        int phase = Integer.parseInt(tokens.get(4));
108                        exons.add(new GenewiseExon(start, end, phase));
109                    }
110                    catch (NumberFormatException e) {
111                        throw new IOException("invalid genewise exon at line number " + lineNumber + ", caught " + e.getMessage());
112                    }
113                }
114                lineNumber++;
115            }
116        }
117        finally {
118            try {
119                reader.close();
120            }
121            catch (Exception e) {
122                // empty
123            }
124            try {
125                genewiseResult.delete();
126            }
127            catch (Exception e) {
128                // empty
129            }
130        }
131        return ImmutableList.copyOf(exons);
132    }
133}
