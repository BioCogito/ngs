023package org.nmdp.ngs.reads.paired;
024
025import org.biojava.bio.program.fastq.Fastq;
026
027/**
028 * Paired end FASTQ reads listener.
029 */
030public interface PairedEndListener {
031
032    /**
033     * Notify this listener of a paired end read.
034     *
035     * @param left left or first read of a paired end read
036     * @param right right or second read of a paired end read
037     */
038    void paired(Fastq left, Fastq right);
039
040    /**
041     * Notify this listener of an unpaired read.
042     *
043     * @param unpaired unpaired read
044     */
045    void unpaired(Fastq unpaired);
046}
