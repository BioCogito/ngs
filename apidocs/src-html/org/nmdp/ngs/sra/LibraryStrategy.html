023package org.nmdp.ngs.sra;
024
025import java.util.Map;
026
027import com.google.common.collect.ImmutableMap;
028import com.google.common.collect.Maps;
029
030/**
031 * Library strategy.
032 */
033public enum LibraryStrategy {
034    WGS("WGS", "Random sequencing of the whole genome"),
035    WGA("WGA", "Whole genome amplification to replace some instances of random"),
036    WXS("WXS", "Random sequencing of exonic regions selected from the genome"),
037    RNA_SEQ("RNA-SEQ", "Random sequencing of whole transcriptome"),
038    MIRNA_SEQ("miRNA-Seq", "Micro RNA and other small non-coding RNA sequencing"),
039    NCRNA_SEQ("ncRNA-Seq", "Non-coding RNA"),
040    WCS("WCS", "Random sequencing of a whole chromosome or other replicon isolated from a genome"),
041    CLONE("CLONE", "Genomic clone based (hierarchical) sequencing"),
042    POOLCLONE("POOLCLONE", "Shotgun of pooled clones (usually BACs and Fosmids)"),
043    AMPLICON("AMPLICON", "Sequencing of overlapping or distinct PCR or RT-PCR products"),
044    CLONEEND("CLONEEND", "Clone end (5', 3', or both) sequencing"),
045    FINISHING("FINISHING", "Sequencing intended to finish (close) gaps in existing coverage"),
046    CHIP_SEQ("ChIP-Seq", "Direct sequencing of chromatin immunoprecipitates"),
047    MNASE_SEQ("MNase-Seq", "Direct sequencing following MNase digestion"),
048    DNASE_HYPERSENSITIVITY("DNase-Hypersensitivity", "Sequencing of hypersensitive sites, or segments of open chromatin that are more readily cleaved by DNaseI"),
049    BISULFITE_SEQ("Bisulfite-Seq", "Sequencing following treatment of DNA with bisulfite to convert cytosine residues to uracil depending on methylation status"),
050    EST("EST", "Single pass sequencing of cDNA templates"),
051    FL_CDNA("FL-cDNA", "Full-length sequencing of cDNA templates"),
052    CTS("CTS", "Concatenated tag sequencing"),
053    MRE_SEQ("MRE-Seq", "Methylation-Sensitive Restriction Enzyme Sequencing strategy"),
054    MEDIP_SEQ("MeDIP-Seq", "Methylated DNA Immunoprecipitation Sequencing strategy"),
055    MDB_SEQ("MDB-Seq", "Direct sequencing of methylated fractions sequencing strategy"),
056    TN_SEQ("Tn-Seq", "Gene fitness determination through transposon seeding"),
057    VALIDATION("VALIDATION", "Validation"),
058    //FAIRE_SEQ("FAIRE-Seq", "Formaldehyde-Assisted Isolation of Regulatory Elements"),
059    FAIRE_SEQ("FAIRE-seq", "Formaldehyde-Assisted Isolation of Regulatory Elements"),
060    SELEX("SELEX", "Systematic Evolution of Ligands by EXponential enrichment (SELEX) is an in vitro strategy to analyze RNA sequences that perform an activity of interest, most commonly high affinity binding to a ligand"),
061    RIP_SEQ("RIP-Seq", "Direct sequencing of RNA immunoprecipitates (includes CLIP-Seq, HITS-CLIP and PAR-CLI)"),
062    CHIA_PET("ChiA-PET", "Direct sequencing of proximity-ligated chromatin immunoprecipitates"),
063    OTHER("OTHER", "Library strategy not listed");
064
065    private final String description;
066    private final String definition;
067    private static final Map<String, LibraryStrategy> BY_DESCRIPTION;
068
069    static
070    {
071        Map<String, LibraryStrategy> byDescription = Maps.newHashMap();
072        for (LibraryStrategy studyType : values()) {
073            byDescription.put(studyType.getDescription(), studyType);
074        }
075        BY_DESCRIPTION = ImmutableMap.copyOf(byDescription);
076    }
077
078
079    /**
080     * Create a new library strategy with the specified description and definition.
081     *
082     * @param description description
083     * @param definition definition
084     */
085    private LibraryStrategy(final String description, final String definition) {
086        this.description = description;
087        this.definition = definition;
088    }
089
090
091    /**
092     * Return the description of this library strategy.
093     *
094     * @return the description of this library strategy
095     */
096    public String getDescription() {
097        return description;
098    }
099
100    /**
101     * Return the definition of this library strategy.
102     *
103     * @return the definition of this library strategy
104     */
105    public String getDefinition() {
106        return definition;
107    }
108
109    /**
110     * Return a map of library strategies keyed by description.
111     *
112     * @return a map of library strategies keyed by description
113     */
114    public static Map<String, LibraryStrategy> byDescription() {
115        return BY_DESCRIPTION;
116    }
117
118    /**
119     * Return the library strategy with the specified description, or null if no such library strategy exists.
120     *
121     * @param description description
122     * @return the library strategy with the specified description, or null if no such library strategy exists
123     */
124    public static LibraryStrategy fromDescription(final String description) {
125        return BY_DESCRIPTION.get(description);
126    }
127}
