023package org.nmdp.ngs.sra;
024
025import java.util.Map;
026
027import com.google.common.collect.ImmutableMap;
028import com.google.common.collect.Maps;
029
030/**
031 * Library selection.
032 */
033public enum LibrarySelection {
034    RANDOM("RANDOM", "Random selection by shearing or other method"),
035    PCR("PCR", "Source material was selected by designed primers"),
036    RANDOM_PCR("RANDOM PCR", "Source material was selected by randomly generated primers"),
037    RT_PCR("RT-PCR", "Source material was selected by reverse transcription PCR"),
038    HMPR("HMPR", "Hypo-methylated partial restriction digest"),
039    MF("MF", "Methyl Filtrated"),
040    REPEAT_FRACTIONATION("repeat fractionation", "Repeat fractionation (replaces: CF-S, CF-M, CF-H, CF-T)"),
041    SIZE_FRACTIONATION("size fractionation", "Size fractionation"),
042    MSLL("MSLL", "Methylation Spanning Linking Library"),
043    CDNA("cDNA", "complementary DNA"),
044    CHIP("ChIP", "Chromatin immunoprecipitation"),
045    MNASE("MNase", "Micrococcal Nuclease (MNase) digestion"),
046    DNASE("DNAse", "Deoxyribonuclease (MNase) digestion"),
047    HYBRID_SELECTION("Hybrid Selection", "Selection by hybridization in array or solution"),
048    REDUCED_REPRESENTATION("Reduced Representation", "Reproducible genomic subsets, often generated by restriction fragment size selection, containing a manageable number of loci to facilitate re-sampling"),
049    RESTRICTION_DIGEST("Restriction Digest", "DNA fractionation using restriction enzymes"),
050    FIVE_METHYLCYTIDINE_ANTIBODY("5-methylcytidine antibody", "Selection of methylated DNA fragments using an antibody raised against 5-methylcytosine or  5-methylcytidine (m5C)"),
051    MBD2_PROTEIN_METHYL_CPG_BINDING_DOMAIN("MBD2 protein methyl-CpG binding domain", "Enrichment by methyl-CpG binding domain"),
052    CAGE("CAGE", "Cap-analysis gene expression"),
053    RACE("RACE", "Rapid Amplification of cDNA Ends"),
054    MDA("MDA", "Multiple displacement amplification"),
055    PADLOCK_PROBES_CAPTURE_METHOD("padlock probes capture method", "Used in conjuction with Bisulfite-Seq"),
056    OTHER("other", "Other library enrichment, screening, or selection process"),
057    UNSPECIFIED("unspecified", "Library enrichment, screening, or selection is not specified");
058
059    private final String description;
060    private final String definition;
061    private static final Map<String, LibrarySelection> BY_DESCRIPTION;
062
063    static
064    {
065        Map<String, LibrarySelection> byDescription = Maps.newHashMap();
066        for (LibrarySelection studyType : values()) {
067            byDescription.put(studyType.getDescription(), studyType);
068        }
069        BY_DESCRIPTION = ImmutableMap.copyOf(byDescription);
070    }
071
072
073    /**
074     * Create a new library selection with the specified description and definition.
075     *
076     * @param description description
077     * @param definition definition
078     */
079    private LibrarySelection(final String description, final String definition) {
080        this.description = description;
081        this.definition = definition;
082    }
083
084
085    /**
086     * Return the description of this library selection.
087     *
088     * @return the description of this library selection
089     */
090    public String getDescription() {
091        return description;
092    }
093
094    /**
095     * Return the definition of this library selection.
096     *
097     * @return the definition of this library selection
098     */
099    public String getDefinition() {
100        return definition;
101    }
102
103    /**
104     * Return a map of library selections keyed by description.
105     *
106     * @return a map of library selections keyed by description
107     */
108    public static Map<String, LibrarySelection> byDescription() {
109        return BY_DESCRIPTION;
110    }
111
112    /**
113     * Return the library selection with the specified description, or null if no such library selection exists.
114     *
115     * @param description description
116     * @return the library selection with the specified description, or null if no such library selection exists
117     */
118    public static LibrarySelection fromDescription(final String description) {
119        return BY_DESCRIPTION.get(description);
120    }
121}
