023package org.nmdp.ngs.tools;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import static org.dishevelled.compress.Readers.reader;
028import static org.dishevelled.compress.Writers.writer;
029
030import java.io.File;
031import java.io.PrintWriter;
032import java.io.IOException;
033
034import java.util.List;
035
036import java.util.concurrent.Callable;
037
038import com.google.common.collect.ImmutableList;
039
040import org.biojava.bio.program.fastq.Fastq;
041import org.biojava.bio.program.fastq.FastqWriter;
042import org.biojava.bio.program.fastq.SangerFastqWriter;
043import org.biojava.bio.program.fastq.SangerFastqReader;
044import org.biojava.bio.program.fastq.StreamListener;
045
046import org.dishevelled.commandline.ArgumentList;
047import org.dishevelled.commandline.CommandLine;
048import org.dishevelled.commandline.CommandLineParseException;
049import org.dishevelled.commandline.CommandLineParser;
050import org.dishevelled.commandline.Switch;
051import org.dishevelled.commandline.Usage;
052
053import org.dishevelled.commandline.argument.FileArgument;
054import org.dishevelled.commandline.argument.FileListArgument;
055
056/**
057 * Merge two or more files in FASTQ format.
058 */
059public final class MergeFastq implements Callable<Integer> {
060    private final List<File> inputFastqFiles;
061    private final File outputFastqFile;
062    private final FastqWriter fastqWriter = new SangerFastqWriter();
063    private static final String USAGE = "ngs-merge-fastq -i foo_1.fq.gz,bar_1.fq.gz [args]";
064
065
066    /**
067     * Merge two or more files in FASTQ format.
068     *
069     * @param inputFastqFiles list of FASTQ input files, must not be null
070     * @param outputFastqFile FASTQ output file, if any
071     */
072    public MergeFastq(final List<File> inputFastqFiles, final File outputFastqFile) {
073        checkNotNull(inputFastqFiles);
074        this.inputFastqFiles = ImmutableList.copyOf(inputFastqFiles);
075        this.outputFastqFile = outputFastqFile;
076    }
077
078
079    @Override
080    public Integer call() throws Exception {
081        PrintWriter writer = null;
082        try {
083            writer = writer(outputFastqFile);
084
085            Append append = new Append(fastqWriter, writer);
086            SangerFastqReader fastqReader = new SangerFastqReader();
087            for (File inputFastqFile : inputFastqFiles) {
088                fastqReader.stream(reader(inputFastqFile), append);
089            }
090
091            return 0;
092        }
093        finally {
094            try {
095                writer.close();
096            }
097            catch (Exception e) {
098                // ignore
099            }
100        }
101    }
102
103    /**
104     * Append.
105     */
106    private static final class Append implements StreamListener {
107        private final FastqWriter fastqWriter;
108        private final PrintWriter writer;
109
110        private Append(final FastqWriter fastqWriter, final PrintWriter writer) {
111            this.fastqWriter = fastqWriter;
112            this.writer = writer;
113        }
114
115        @Override
116        public void fastq(final Fastq fastq) {
117            try {
118                fastqWriter.append(writer, fastq);
119            }
120            catch (IOException e) {
121                // todo:  rethrow?
122            }
123        }
124    }
125
126
127    /**
128     * Main.
129     *
130     * @param args command line args
131     */
132    public static void main(final String[] args) {
133        Switch about = new Switch("a", "about", "display about message");
134        Switch help = new Switch("h", "help", "display help message");
135        FileListArgument inputFastqFiles = new FileListArgument("i", "input-fastq-files", "list of FASTQ input files", true);
136        FileArgument outputFastqFile = new FileArgument("o", "output-fastq-file", "FASTQ output file, default stdout", false);
137
138        ArgumentList arguments = new ArgumentList(about, help, inputFastqFiles, outputFastqFile);
139        CommandLine commandLine = new CommandLine(args);
140
141        MergeFastq mergeFastq = null;
142        try {
143            CommandLineParser.parse(commandLine, arguments);
144            if (about.wasFound()) {
145                About.about(System.out);
146                System.exit(0);
147            }
148            if (help.wasFound()) {
149                Usage.usage(USAGE, null, commandLine, arguments, System.out);
150                System.exit(0);
151            }
152            mergeFastq = new MergeFastq(inputFastqFiles.getValue(), outputFastqFile.getValue());
153        }
154        catch (CommandLineParseException e) {
155            if (about.wasFound()) {
156                About.about(System.out);
157                System.exit(0);
158            }
159            if (help.wasFound()) {
160                Usage.usage(USAGE, null, commandLine, arguments, System.out);
161                System.exit(0);
162            }
163            Usage.usage(USAGE, e, commandLine, arguments, System.err);
164            System.exit(-1);
165        }
166        try {
167            System.exit(mergeFastq.call());
168        }
169        catch (Exception e) {
170            e.printStackTrace();
171            System.exit(1);
172        }
173    }
174}
