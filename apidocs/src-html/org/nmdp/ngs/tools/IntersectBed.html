074/**
075 * Similar to bedtools2 intersect -v.
076 */
077public final class IntersectBed implements Callable<Integer> {
078    private final File aInputFile;
079    private final File bInputFile;
080    private final File outputFile;
081    private final Strategy strategy;
082    private static final String DEFAULT_STRATEGY = "range-set";
083    private static final String USAGE = "ngs-intersect-bed -b b.bed.gz [args]";
084
085
086    /**
087     * Similar to bedtools2 intersect -v.
088     *
089     * @param aInputFile a input file, if any
090     * @param bInputFile b input file, must not be null
091     * @param outputFile output file, if any
092     * @param strategy strategy, must not be null
093     */
094    public IntersectBed(final File aInputFile, final File bInputFile, final File outputFile, final Strategy strategy) {
095        checkNotNull(bInputFile);
096        checkNotNull(strategy);
097        this.aInputFile = aInputFile;
098        this.bInputFile = bInputFile;
099        this.outputFile = outputFile;
100        this.strategy = strategy;
101    }
102
103
104    @Override
105    public Integer call() throws Exception {
106        BufferedReader a = null;
107        BufferedReader b = null;
108        PrintWriter writer = null;
109
110        try {
111            a = reader(aInputFile);
112            b = reader(bInputFile);
113            writer = writer(outputFile);
114
115            strategy.intersectBed(a, b, writer);
116
117            return 0;
118        }
119        finally {
120            try {
121                a.close();
122            }
123            catch (Exception e) {
124                // ignore
125            }
126            try {
127                b.close();
128            }
129            catch (Exception e) {
130                // ignore
131            }
132            try {
133                writer.close();
134            }
135            catch (Exception e) {
136                // ignore
137            }
138        }
139    }
140
141    /**
142     * Intersect BED strategy.
143     */
144    interface Strategy {
145        void intersectBed(BufferedReader a, BufferedReader b, PrintWriter writer) throws IOException;
146    }
147
148    /** Map of strategies keyed by name. */
149    private static final Map<String, Strategy> strategies = ImmutableMap.<String, Strategy>builder()
150        .put("range-list", new RangeListStrategy())
151        .put("range-set", new RangeSetStrategy())
152        .put("centered-range-tree", new CenteredRangeTreeStrategy())
153        .put("r-tree", new RangeRTreeStrategy())
154        .put("r-star-tree", new RangeRStarTreeStrategy())
155        .build();
156
157    /** Range list strategy. */
158    private static final class RangeListStrategy implements Strategy {
159        @Override
160        public void intersectBed(final BufferedReader a, final BufferedReader b, final PrintWriter writer) throws IOException {
161            // read all of b into memory, group by chromosome
162            final ListMultimap<String, BedRecord> ref = ArrayListMultimap.create();
163            BedReader.stream(b, new BedListener() {
164                @Override
165                public boolean record(final BedRecord rec) {
166                    ref.put(rec.chrom(), rec);
167                    return true;
168                }
169            });
170
171            // calculate coverage range trees
172            final Map<String, RangeTree<Long>> coverage = Maps.newHashMap();
173            for (String chr : ref.keySet()) {
174                List<BedRecord> records = ref.get(chr);
175
176                List<Range<Long>> ranges = Lists.newArrayListWithExpectedSize(records.size());
177                for (BedRecord rec : records) {
178                    ranges.add(rec.toRange());
179                }
180
181                RangeTree<Long> rangeTree = RangeList.create(ranges);
182                coverage.put(chr, rangeTree);
183            }
184            ref.clear();
185
186            // stream records from a, compare to b
187            final PrintWriter w = writer;
188            BedReader.stream(a, new BedListener() {
189                @Override
190                public boolean record(final BedRecord rec) {
191                    String chr = rec.chrom();
192                    if (coverage.containsKey(chr) && !coverage.get(chr).intersects(rec.toRange())) {
193                        BedWriter.write(rec, w);
194                    }
195                    return true;
196                }
197            });
198        }
199    }
200
201    /** Range set strategy. */
202    private static final class RangeSetStrategy implements Strategy {
203        @Override
204        public void intersectBed(final BufferedReader a, final BufferedReader b, final PrintWriter writer) throws IOException {
205            // calculate coverage range sets by chromosome
206            final Map<String, RangeSet<Long>> coverage = Maps.newHashMap();
207            BedReader.stream(b, new BedListener() {
208                @Override
209                public boolean record(final BedRecord rec) {
210                    String chr = rec.chrom();
211                    if (!coverage.containsKey(chr)) {
212                        RangeSet<Long> rangeSet = TreeRangeSet.create();
213                        coverage.put(chr, rangeSet);
214                    }
215                    coverage.get(chr).add(rec.toRange());
216                    return true;
217                }
218            });
219
220            // stream records from a, compare to b
221            final PrintWriter w = writer;
222            BedReader.stream(a, new BedListener() {
223                @Override
224                public boolean record(final BedRecord rec) {
225                    String chr = rec.chrom();
226                    if (coverage.containsKey(chr) && coverage.get(chr).subRangeSet(rec.toRange()).isEmpty()) {
227                        BedWriter.write(rec, w);
228                    }
229                    return true;
230                }
231            });
232        }
233    }

279    /** Range R-Tree strategy. */
280    private static final class RangeRTreeStrategy implements Strategy {
281        @Override
282        public void intersectBed(final BufferedReader a, final BufferedReader b, final PrintWriter writer) throws IOException {
283            // read all of b into memory, group by chromosome
284            final Object key = new Object();
285            final Map<String, RTree<Object>> coverage = Maps.newHashMap();
286            BedReader.stream(b, new BedListener() {
287                @Override
288                public boolean record(final BedRecord rec) {
289                    String chr = rec.chrom();
290                    if (!coverage.containsKey(chr)) {
291                        RTree rtree = RTree.maxChildren(12).create();
292                        coverage.put(chr, rtree);
293                    }
294                    coverage.put(chr, coverage.get(chr).add(key, RangeGeometries.range(rec.toRange())));
295                    return true;
296                }
297            });
298
299            // stream records from a, compare to b
300            final PrintWriter w = writer;
301            BedReader.stream(a, new BedListener() {
302                @Override
303                public boolean record(final BedRecord rec) {
304                    String chr = rec.chrom();
305                    if (coverage.containsKey(chr) && isEmpty(coverage.get(chr).search(RangeGeometries.range(rec.toRange())))) {
306                        BedWriter.write(rec, w);
307                    }
308                    return true;
309                }
310            });
311        }
312    }

349    private static <T> boolean isEmpty(final Observable<T> observable) {
350        return observable.isEmpty().toBlocking().first();
351    }

354    /**
355     * Main.
356     *
357     * @param args command line args
358     */
359    public static void main(final String[] args) {
360        Switch help = new Switch("h", "help", "display help message");
361        FileArgument aInputFile = new FileArgument("a", "a-input-file", "A input BED file, default stdin", false);
362        FileArgument bInputFile = new FileArgument("b", "b-input-file", "B input BED file", true);
363        FileArgument outputFile = new FileArgument("o", "output-file", "output BED file, default stdout", false);
364        StringArgument strategy = new StringArgument("s", "strategy", "strategy { range-list, range-set, centered-range-tree, r-tree, r-star-tree }, default range-set", false);
365
366        ArgumentList arguments = new ArgumentList(help, aInputFile, bInputFile, outputFile, strategy);
367        CommandLine commandLine = new CommandLine(args);
368
369        IntersectBed intersectBed = null;
370        try
371        {
372            CommandLineParser.parse(commandLine, arguments);
373            if (help.wasFound()) {
374                Usage.usage(USAGE, null, commandLine, arguments, System.out);
375                System.exit(0);
376            }
377            intersectBed = new IntersectBed(aInputFile.getValue(), bInputFile.getValue(), outputFile.getValue(), strategies.get(strategy.getValue(DEFAULT_STRATEGY)));
378        }
379        catch (CommandLineParseException | NullPointerException e) {
380            if (help.wasFound()) {
381                Usage.usage(USAGE, null, commandLine, arguments, System.out);
382                System.exit(0);
383            }
384            Usage.usage(USAGE, e, commandLine, arguments, System.err);
385            System.exit(-1);
386        }
387        try {
388            System.exit(intersectBed.call());
389        }
390        catch (Exception e) {
391            e.printStackTrace();
392            System.exit(1);
393        }
394    }
