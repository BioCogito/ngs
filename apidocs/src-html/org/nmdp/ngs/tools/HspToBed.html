023package org.nmdp.ngs.tools;
024
025import static org.dishevelled.compress.Readers.reader;
026import static org.dishevelled.compress.Writers.writer;
027
028import java.io.BufferedReader;
029import java.io.File;
030import java.io.PrintWriter;
031
032import java.util.concurrent.Callable;
033
034import com.google.common.base.Joiner;
035
036import org.dishevelled.commandline.ArgumentList;
037import org.dishevelled.commandline.CommandLine;
038import org.dishevelled.commandline.CommandLineParseException;
039import org.dishevelled.commandline.CommandLineParser;
040import org.dishevelled.commandline.Switch;
041import org.dishevelled.commandline.Usage;
042
043import org.dishevelled.commandline.argument.FileArgument;
044import org.dishevelled.commandline.argument.StringArgument;
045
046import org.nmdp.ngs.align.BedRecord;
047import org.nmdp.ngs.align.BedWriter;
048import org.nmdp.ngs.align.HighScoringPair;
049import org.nmdp.ngs.align.HspListener;
050import org.nmdp.ngs.align.HspReader;
051
052/**
053 * Convert HSPs to BED format.
054 */
055public final class HspToBed implements Callable<Integer> {
056    private final String displayName;
057    private final File hspFile;
058    private final File bedFile;
059    private final boolean reverse;
060    private final boolean transformEvalue;
061    private static final String USAGE = "ngs-hsp-to-bed [args]";
062
063
064    /**
065     * Convert HSPs to BED format.
066     *
067     * @param displayName query display name, if any
068     * @param hspFile input HSP file, from e.g. blastn -outfmt 7, if any
069     * @param bedFile output BED file, if any
070     * @param reverse true to reverse query and target in BED file
071     * @param transformEvalue true to transform e-value to BED score [0..1000]
072     */
073    public HspToBed(final String displayName, final File hspFile, final File bedFile, final boolean reverse, final boolean transformEvalue) {
074        this.displayName = displayName;
075        this.hspFile = hspFile;
076        this.bedFile = bedFile;
077        this.reverse = reverse;
078        this.transformEvalue = transformEvalue;
079    }
080
081
082    @Override
083    public Integer call() throws Exception {
084        BufferedReader reader = null;
085        PrintWriter writer = null;
086        try {
087            reader = reader(hspFile);
088            writer = writer(bedFile);
089
090            final PrintWriter w = writer;
091            HspReader.stream(reader, new HspListener() {
092                    @Override
093                    public boolean hsp(final HighScoringPair hsp) {
094                        BedWriter.write((reverse ? toReverseBedRecord(hsp) : toBedRecord(hsp)), w);
095                        return true;
096                    }
097                });
098
099            return 0;
100        }
101        finally {
102            try {
103                reader.close();
104            }
105            catch (Exception e) {
106                // empty
107            }
108            try {
109                writer.close();
110            }
111            catch (Exception e) {
112                // empty
113            }
114        }
115    }
116
117    private BedRecord toBedRecord(final HighScoringPair hsp) {
118        String chrom = hsp.target();
119        // hsps are 1-based, fully closed; bed records are 0-based, closed open
120        long start = hsp.targetStart() - 1L;
121        long end = hsp.targetEnd();
122        String name = Joiner.on(":").join((displayName != null ? displayName : hsp.source()), hsp.sourceStart(), hsp.sourceEnd(), (hsp.sourceStart() <= hsp.sourceEnd() ? "+" : "-"), hsp.percentIdentity(), hsp.alignmentLength(), hsp.mismatches(), hsp.gapOpens(), hsp.evalue(), hsp.bitScore());
123        String score = String.valueOf((transformEvalue ? transform(hsp.evalue()) : hsp.evalue()));
124        String strand = hsp.targetStart() <= hsp.targetEnd() ? "+" : "-";
125
126        return new BedRecord(chrom, start, end, name, score, strand);
127    }
128
129    private BedRecord toReverseBedRecord(final HighScoringPair hsp) {
130        String chrom = displayName != null ? displayName : hsp.source();
131        // hsps are 1-based, fully closed; bed records are 0-based, closed open
132        long start = hsp.sourceStart() - 1L;
133        long end = hsp.sourceEnd();
134        String name = Joiner.on(":").join(hsp.target(), hsp.targetStart(), hsp.targetEnd(), (hsp.targetStart() <= hsp.targetEnd() ? "+" : "-"), hsp.percentIdentity(), hsp.alignmentLength(), hsp.mismatches(), hsp.gapOpens(), hsp.evalue(), hsp.bitScore());
135        String score = String.valueOf((transformEvalue ? transform(hsp.evalue()) : hsp.evalue()));
136        String strand = hsp.sourceStart() <= hsp.sourceEnd() ? "+" : "-";
137
138        return new BedRecord(chrom, start, end, name, score, strand);
139    }
140
141    private static int transform(final double evalue) {
142        double transformedEvalue = -Math.log10(evalue) + 2.0d;
143        if (Double.isInfinite(transformedEvalue)) {
144            transformedEvalue = 257.0d; // 255.0 + 2.0d
145        }
146        transformedEvalue = lerp(transformedEvalue, 0.0d, 257.0d, 0.0d, 1000.0d);
147        return (int) transformedEvalue;
148    }
149
150    private static double lerp(final double value, final double sourceMinimum, final double sourceMaximum, final double targetMinimum, final double targetMaximum) {
151        return targetMinimum + (targetMaximum - targetMinimum) * ((value - sourceMinimum) / (sourceMaximum - sourceMinimum));
152    }
153
154
155    /**
156     * Main.
157     *
158     * @param args command line args
159     */
160    public static void main(final String[] args) {
161        Switch about = new Switch("a", "about", "display about message");
162        Switch help = new Switch("h", "help", "display help message");
163        StringArgument displayName = new StringArgument("d", "display-name", "query display name", false);
164        FileArgument hspFile = new FileArgument("i", "hsp-file", "input HSP file, from e.g. blastn -outfmt 7, default stdin", false);
165        FileArgument bedFile = new FileArgument("o", "bed-file", "output BED file, default stdout", false);
166        Switch reverse = new Switch("r", "reverse", "reverse query and target in BED file");
167        Switch transformEvalue = new Switch("t", "transform-evalue", "transform e-value to BED score [0..1000]");
168
169        ArgumentList arguments = new ArgumentList(about, help, displayName, hspFile, bedFile, reverse, transformEvalue);
170        CommandLine commandLine = new CommandLine(args);
171
172        HspToBed hspToBed = null;
173        try {
174            CommandLineParser.parse(commandLine, arguments);
175            if (about.wasFound()) {
176                About.about(System.out);
177                System.exit(0);
178            }
179            if (help.wasFound()) {
180                Usage.usage(USAGE, null, commandLine, arguments, System.out);
181                System.exit(0);
182            }
183            hspToBed = new HspToBed(displayName.getValue(), hspFile.getValue(), bedFile.getValue(), reverse.wasFound(), transformEvalue.wasFound());
184        }
185        catch (CommandLineParseException e) {
186            Usage.usage(USAGE, e, commandLine, arguments, System.err);
187            System.exit(-1);
188        }
189        try {
190            System.exit(hspToBed.call());
191        }
192        catch (Exception e) {
193            e.printStackTrace();
194            System.exit(1);
195        }
196    }
