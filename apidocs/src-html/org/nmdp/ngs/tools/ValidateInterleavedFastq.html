023package org.nmdp.ngs.tools;
024
025import static org.dishevelled.compress.Readers.reader;
026
027import java.io.BufferedReader;
028import java.io.File;
029
030import java.util.concurrent.Callable;
031
032import org.dishevelled.commandline.ArgumentList;
033import org.dishevelled.commandline.CommandLine;
034import org.dishevelled.commandline.CommandLineParseException;
035import org.dishevelled.commandline.CommandLineParser;
036import org.dishevelled.commandline.Switch;
037import org.dishevelled.commandline.Usage;
038
039import org.dishevelled.commandline.argument.FileArgument;
040
041import org.nmdp.ngs.reads.paired.PairedEndAdapter;
042import org.nmdp.ngs.reads.paired.PairedEndFastqReader;
043
044/**
045 * Validate a file in interleaved FASTQ format.
046 */
047public final class ValidateInterleavedFastq implements Callable<Integer> {
048    private final File inputFastqFile;
049    private static final String USAGE = "ngs-validate-interleaved-fastq [args]";
050
051
052    /**
053     * Validate a file in interleaved FASTQ format.
054     *
055     * @param inputFastqFile input interleaved FASTQ file, if any
056     */
057    public ValidateInterleavedFastq(final File inputFastqFile) {
058        this.inputFastqFile = inputFastqFile;
059    }
060
061
062    @Override
063    public Integer call() throws Exception {
064        BufferedReader reader = null;
065        try {
066            reader = reader(inputFastqFile);
067            PairedEndFastqReader.streamInterleaved((Readable) reader, new PairedEndAdapter());
068
069            return 0;
070        }
071        finally {
072            try {
073                reader.close();
074            }
075            catch (Exception e) {
076                // ignore
077            }
078        }
079    }
080
081
082    /**
083     * Main.
084     *
085     * @param args command line args
086     */
087    public static void main(final String[] args) {
088        Switch about = new Switch("a", "about", "display about message");
089        Switch help = new Switch("h", "help", "display help message");
090        FileArgument inputFastqFile = new FileArgument("i", "input-fastq-file", "input interleaved FASTQ file, default stdin", false);
091
092        ArgumentList arguments = new ArgumentList(about, help, inputFastqFile);
093        CommandLine commandLine = new CommandLine(args);
094
095        ValidateInterleavedFastq validateInterleavedFastq = null;
096        try
097        {
098            CommandLineParser.parse(commandLine, arguments);
099            if (about.wasFound()) {
100                About.about(System.out);
101                System.exit(0);
102             }
103            if (help.wasFound()) {
104                Usage.usage(USAGE, null, commandLine, arguments, System.out);
105                System.exit(0);
106            }
107            validateInterleavedFastq = new ValidateInterleavedFastq(inputFastqFile.getValue());
108        }
109        catch (CommandLineParseException | IllegalArgumentException e) {
110            Usage.usage(USAGE, e, commandLine, arguments, System.err);
111            System.exit(-1);
112        }
113        try {
114            System.exit(validateInterleavedFastq.call());
115        }
116        catch (Exception e) {
117            e.printStackTrace();
118            System.exit(1);
119        }
120    }
121}
