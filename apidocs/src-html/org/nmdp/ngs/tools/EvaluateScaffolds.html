067/**
068 * Evaluate assembly scaffolds against a reference sequence.
069 */
070@SuppressWarnings("deprecation")
071public final class EvaluateScaffolds implements Callable<Integer> {
072    private final File referenceFastaFile;
073    private final File scaffoldsFastaFile;
074    private final File evalFile;
075    private static final String USAGE = "ngs-evaluate-scaffolds -r reference.fa.gz -s scaffolds.fa.gz";
076
077
078    /**
079     * Evaluate assembly scaffolds against a reference sequence.
080     *
081     * @param referenceFastaFile input reference FASTA file, must not be null
082     * @param scaffoldsFastaFile input scaffolds FASTA file, must not be null
083     * @param evalFile output eval file, if any
084     */
085    public EvaluateScaffolds(final File referenceFastaFile, final File scaffoldsFastaFile, final File evalFile) {
086        checkNotNull(referenceFastaFile);
087        checkNotNull(scaffoldsFastaFile);
088        this.referenceFastaFile = referenceFastaFile;
089        this.scaffoldsFastaFile = scaffoldsFastaFile;
090        this.evalFile = evalFile;
091    }
092
093
094    @Override
095    public Integer call() throws Exception {
096        PrintWriter writer = null;
097        try {
098            writer = writer(evalFile);
099
100            Sequence reference = readReference();
101            List<Sequence> scaffolds = readScaffolds();
102
103            writer.println("#reference length = " + reference.length());
104            writer.println("#scaffold count = " + scaffolds.size());
105            writer.println("#scaffold lengths = " + dumpLengths(scaffolds));
106
107            RangeSet<Long> ranges = TreeRangeSet.create();
108            for (HighScoringPair hsp : blastn(referenceFastaFile, scaffoldsFastaFile)) {
109                if (reference.getName().equals(hsp.target())) {
110                    writer.println("#" + hsp.toString());
111                    if (hsp.targetStart() <= hsp.targetEnd()) { // strands match
112                        ranges.add(Range.closed(hsp.targetStart(), hsp.targetEnd()));
113                    }
114                    else {
115                        ranges.add(Range.closed(hsp.targetEnd(), hsp.targetStart()));
116                    }
117                }
118            }
119
120            writer.println("#coalesced intervals = " + ranges);
121
122            long breadthOfCoverage = 0;
123            for (Range<Long> range : ranges.asRanges()) {
124                breadthOfCoverage += ContiguousSet.create(range, DiscreteDomain.longs()).size();
125            }
126            double normalizedBreadthOfCoverage = (double) breadthOfCoverage / (double) reference.length();
127            writer.println("#breadth-of-coverage = " + breadthOfCoverage);
128            writer.println("#normalized breadth-of-coverage = " + normalizedBreadthOfCoverage);
129
130            StringBuilder sb = new StringBuilder();
131            sb.append(referenceFastaFile.getName());
132            sb.append("\t");
133            sb.append(scaffoldsFastaFile.getName());
134            sb.append("\t");
135            sb.append(reference.length());
136            sb.append("\t");
137            sb.append(scaffolds.size());
138            sb.append("\t");
139            sb.append(ranges.asRanges().size());
140            sb.append("\t");
141            sb.append(breadthOfCoverage);
142            sb.append("\t");
143            sb.append(normalizedBreadthOfCoverage);
144            sb.append("\t");
145            writer.println(sb.toString());
146
147            return 0;
148        }
149        finally {
150            try {
151                writer.close();
152            }
153            catch (Exception e) {
154                // ignore
155            }
156        }
157    }
158
159    private Sequence readReference() throws IOException, BioException {
160        Sequence sequence = null;
161        BufferedReader reader = null;
162        try {
163            reader = reader(referenceFastaFile);
164            SequenceIterator sequences = SeqIOTools.readFastaDNA(reader);
165            if (!sequences.hasNext()) {
166                throw new IOException("reference FASTA file was empty");
167            }
168            sequence = sequences.nextSequence();
169            if (sequences.hasNext()) {
170                throw new IOException("reference FASTA file contains more than one sequence");
171            }
172        }
173        finally {
174            try {
175                reader.close();
176            }
177            catch (Exception e) {
178                // ignore
179            }
180        }
181        return sequence;
182    }
183
184    private List<Sequence> readScaffolds() throws IOException, BioException {
185        BufferedReader reader = null;
186        List<Sequence> scaffolds = Lists.newLinkedList();
187        try {
188            reader = reader(scaffoldsFastaFile);
189
190            for (SequenceIterator sequences = SeqIOTools.readFastaDNA(reader); sequences.hasNext(); ) {
191                Sequence sequence = sequences.nextSequence();
192                scaffolds.add(sequence);
193            }
194        }
195        finally {
196            try {
197                reader.close();
198            }
199            catch (Exception e) {
200                // ignore
201            }
202        }
203        return scaffolds;
204    }
205
206    private static String dumpLengths(final List<Sequence> scaffolds) {
207        StringBuilder sb = new StringBuilder();
208        Iterator<Sequence> iterator = scaffolds.iterator();
209        sb.append(iterator.next().length());
210        while (iterator.hasNext()) {
211            sb.append(", ");
212            sb.append(iterator.next().length());
213        }
214        return sb.toString();
215    }

218    /**
219     * Main.
220     *
221     * @param args command line args
222     */
223    public static void main(final String[] args) {
224        Switch about = new Switch("a", "about", "display about message");
225        Switch help = new Switch("h", "help", "display help message");
226        FileArgument referenceFastaFile = new FileArgument("r", "reference", "input reference FASTA file", true);
227        FileArgument scaffoldsFastaFile = new FileArgument("s", "scaffolds", "input scaffolds FASTA file", true);
228        FileArgument evalFile = new FileArgument("e", "eval", "output eval file, default stdout", false);
229
230        ArgumentList arguments = new ArgumentList(about, help, referenceFastaFile, scaffoldsFastaFile, evalFile);
231        CommandLine commandLine = new CommandLine(args);
232
233        EvaluateScaffolds evaluateScaffolds = null;
234        try {
235            CommandLineParser.parse(commandLine, arguments);
236            if (about.wasFound()) {
237                About.about(System.out);
238                System.exit(0);
239            }
240            if (help.wasFound()) {
241                Usage.usage(USAGE, null, commandLine, arguments, System.out);
242                System.exit(0);
243            }
244            evaluateScaffolds = new EvaluateScaffolds(referenceFastaFile.getValue(), scaffoldsFastaFile.getValue(), evalFile.getValue());
245        }
246        catch (CommandLineParseException e) {
247            if (about.wasFound()) {
248                About.about(System.out);
249                System.exit(0);
250            }
251            if (help.wasFound()) {
252                Usage.usage(USAGE, null, commandLine, arguments, System.out);
253                System.exit(0);
254            }
255            Usage.usage(USAGE, e, commandLine, arguments, System.err);
256            System.exit(-1);
257        }
258        try {
259            System.exit(evaluateScaffolds.call());
260        }
261        catch (Exception e) {
262            e.printStackTrace();
263            System.exit(1);
264        }
265    }
