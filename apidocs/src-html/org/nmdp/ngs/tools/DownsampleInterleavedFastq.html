023package org.nmdp.ngs.tools;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import static org.dishevelled.compress.Readers.reader;
028import static org.dishevelled.compress.Writers.writer;
029
030import java.io.BufferedReader;
031import java.io.File;
032import java.io.PrintWriter;
033import java.io.IOException;
034
035import java.util.concurrent.Callable;
036
037import org.apache.commons.math3.distribution.BinomialDistribution;
038
039import org.apache.commons.math3.random.MersenneTwister;
040import org.apache.commons.math3.random.RandomGenerator;
041
042import org.biojava.bio.program.fastq.Fastq;
043import org.biojava.bio.program.fastq.FastqWriter;
044import org.biojava.bio.program.fastq.SangerFastqWriter;
045
046import org.dishevelled.commandline.ArgumentList;
047import org.dishevelled.commandline.CommandLine;
048import org.dishevelled.commandline.CommandLineParseException;
049import org.dishevelled.commandline.CommandLineParser;
050import org.dishevelled.commandline.Switch;
051import org.dishevelled.commandline.Usage;
052
053import org.dishevelled.commandline.argument.DoubleArgument;
054import org.dishevelled.commandline.argument.FileArgument;
055import org.dishevelled.commandline.argument.IntegerArgument;
056
057import org.nmdp.ngs.reads.paired.PairedEndAdapter;
058import org.nmdp.ngs.reads.paired.PairedEndFastqReader;
059
060/**
061 * Downsample sequences from a file in interleaved FASTQ format.
062 */
063public final class DownsampleInterleavedFastq implements Callable<Integer> {
064    private final File inputFastqFile;
065    private final File outputFastqFile;
066    private final BinomialDistribution distribution;
067    private final FastqWriter fastqWriter = new SangerFastqWriter();
068    private static final String USAGE = "ngs-downsample-interleaved-fastq -p 0.5 [args]";
069
070
071    /**
072     * Downsample sequences from a file in interleaved FASTQ format.
073     *
074     * @param inputFastqFile input interleaved FASTQ file, if any
075     * @param outputFastqFile output interleaved FASTQ file, if any
076     * @param distribution binomial distribution, must not be null
077     */
078    public DownsampleInterleavedFastq(final File inputFastqFile, final File outputFastqFile, final BinomialDistribution distribution) {
079        checkNotNull(distribution);
080        this.inputFastqFile = inputFastqFile;
081        this.outputFastqFile = outputFastqFile;
082        this.distribution = distribution;
083    }
084
085
086    @Override
087    public Integer call() throws Exception {
088        BufferedReader reader = null;
089        PrintWriter writer = null;
090        try {
091            reader = reader(inputFastqFile);
092            writer = writer(outputFastqFile);
093
094            final PrintWriter w = writer;
095            PairedEndFastqReader.streamInterleaved(reader, new PairedEndAdapter() {
096                    @Override
097                    public void paired(final Fastq left, final Fastq right) {
098                        if (distribution.sample() > 0) {
099                            try {
100                                fastqWriter.append(w, left);
101                                fastqWriter.append(w, right);
102                            }
103                            catch (IOException e) {
104                                throw new RuntimeException("could not write FASTQ", e);
105                            }
106                        }
107                    }
108                });
109
110            return 0;
111        }
112        finally {
113            try {
114                reader.close();
115            }
116            catch (Exception e) {
117                // ignore
118            }
119            try {
120                writer.close();
121            }
122            catch (Exception e) {
123                // ignore
124            }
125        }
126    }
127
128
129    /**
130     * Main.
131     *
132     * @param args command line args
133     */
134    public static void main(final String[] args) {
135        Switch about = new Switch("a", "about", "display about message");
136        Switch help = new Switch("h", "help", "display help message");
137        FileArgument inputFastqFile = new FileArgument("i", "input-fastq-file", "input interleaved FASTQ file, default stdin", false);
138        FileArgument outputFastqFile = new FileArgument("o", "output-fastq-file", "output interleaved FASTQ file, default stdout", false);
139        DoubleArgument probability = new DoubleArgument("p", "probability", "probability a FASTQ record will be removed, [0.0-1.0]", true);
140        IntegerArgument seed = new IntegerArgument("z", "seed", "random number seed, default relates to current time", false);
141
142        ArgumentList arguments = new ArgumentList(about, help, inputFastqFile, outputFastqFile, probability, seed);
143        CommandLine commandLine = new CommandLine(args);
144
145        DownsampleInterleavedFastq downsampleInterleavedFastq = null;
146        try
147        {
148            CommandLineParser.parse(commandLine, arguments);
149            if (about.wasFound()) {
150                About.about(System.out);
151                System.exit(0);
152            }
153            if (help.wasFound()) {
154                Usage.usage(USAGE, null, commandLine, arguments, System.out);
155                System.exit(0);
156            }
157
158            RandomGenerator random = seed.wasFound() ? new MersenneTwister(seed.getValue()) : new MersenneTwister();
159            BinomialDistribution distribution = new BinomialDistribution(random, 1, probability.getValue());
160
161            downsampleInterleavedFastq = new DownsampleInterleavedFastq(inputFastqFile.getValue(), outputFastqFile.getValue(), distribution);
162        }
163        catch (CommandLineParseException e) {
164            if (about.wasFound()) {
165                About.about(System.out);
166                System.exit(0);
167            }
168            if (help.wasFound()) {
169                Usage.usage(USAGE, null, commandLine, arguments, System.out);
170                System.exit(0);
171            }
172            Usage.usage(USAGE, e, commandLine, arguments, System.err);
173            System.exit(-1);
174        }
175        catch (IllegalArgumentException e) {
176            Usage.usage(USAGE, e, commandLine, arguments, System.err);
177            System.exit(-1);
178        }
179        try {
180            System.exit(downsampleInterleavedFastq.call());
181        }
182        catch (Exception e) {
183            e.printStackTrace();
184            System.exit(1);
185        }
186    }
187}
