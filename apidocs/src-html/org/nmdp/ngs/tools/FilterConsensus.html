084/**
085 * Filter consensus sequences into subregions of research or clinical interest.
086 */
087public final class FilterConsensus implements Callable<Integer> {
088    private final File bamFile;
089    private final File bedFile;
090    private final File outputFile;
091    private final String gene;
092    private final boolean cdna;
093    private final boolean removeGaps;
094    private final double minimumBreadth;
095    private final int expectedPloidy;
096    static final double DEFAULT_MINIMUM_BREADTH = 0.5d;
097    static final int DEFAULT_EXPECTED_PLOIDY = 2;
098    static final String USAGE = "ngs-filter-consensus -i input.bam -x genome.bed -g HLA-A [args]";
099
100
101    /**
102     * Filter consensus sequences into subregions of research or clinical interest.
103     *
104     * @param bamFile input BAM file of consensus sequences, must not be null
105     * @param bedFile input BED file of genomic regions, must not be null
106     * @param outputFile output FASTA file, if any
107     * @param gene gene, must not be null
108     * @param cdna cdna, true to output cdna from the same contig (phased consensus sequence)
109     *    in FASTA format (for interpretation)
110     * @param removeGaps remove alignment gaps in the filtered consensus sequence
111     * @param minimumBreadth minimum breadth, must be in range [0.0 - 1.0]
112     * @param expectedPloidy expected ploidy, must be &gt; 0
113     */
114    public FilterConsensus(final File bamFile,
115                           final File bedFile,
116                           final File outputFile,
117                           final String gene,
118                           final boolean cdna,
119                           final boolean removeGaps,
120                           final double minimumBreadth,
121                           final int expectedPloidy) {
122        checkNotNull(bamFile);
123        checkNotNull(bedFile);
124        checkNotNull(gene);
125        checkArgument(minimumBreadth >= 0.0d && minimumBreadth <= 1.0d, "minimum breadth must be in range [0.0 - 1.0]");
126        checkArgument(expectedPloidy > 0, "expected ploidy must be > 0");
127        this.bamFile = bamFile;
128        this.bedFile = bedFile;
129        this.outputFile = outputFile;
130        this.gene = gene;
131        this.cdna = cdna;
132        this.removeGaps = removeGaps;
133        this.minimumBreadth = minimumBreadth;
134        this.expectedPloidy = expectedPloidy;
135    }
136
137
138    @Override
139    public Integer call() throws Exception {
140        PrintWriter writer = null;
141        try {
142            writer = writer(outputFile);
143
144            // map of region alleles from BED-formatted genomic regions file keyed by "exon" as integer
145            Map<Integer, Allele> exons = readBedFile(bedFile);
146
147            // todo:  refactor this block into separate static methods
148            // map of overlapping alignment alleles from BAM-formatted consensus sequence file keyed by "exon" as integer
149            ListMultimap<Integer, Allele> regions = ArrayListMultimap.create();
150
151            for (SAMRecord record : new SAMFileReader(bamFile)) {
152                List<Edit> edits = cigarToEditList(record);
153
154                String referenceName = record.getReferenceName();
155                int start = record.getAlignmentStart();
156                int end = record.getAlignmentEnd();
157                Locus alignment = new Locus(referenceName, start, end);
158                SymbolList sequence = DNATools.createDNA(record.getReadString());
159
160                for (Edit edit : edits) {
161                    sequence.edit(edit);
162                }
163
164                String name = record.getReadName();
165                Allele contig = Allele.builder()
166                    .withContig(referenceName)
167                    .withStart(start)
168                    .withEnd(end)
169                    .withSequence(sequence)
170                    .build();
171
172                for (Map.Entry<Integer, Allele> entry : exons.entrySet()) {
173                    int index = entry.getKey();
174                    Allele exon = entry.getValue();
175
176                    String chr = exon.getContig();
177                    int min = exon.getMin();
178                    int max = exon.getMax();
179                    String range = chr + ":" + min + "-" + max;
180
181                    if (alignment.overlaps(exon)) {
182                        Allele xover = exon.doubleCrossover(contig);
183
184                        Allele clipped = xover.leftHardClip("-").rightHardClip("-");
185                        clipped.setName(">" + name + "|gene=" + gene + "|exon=" + index + "|location=" + range + "|" + (max - min));
186                        regions.put(index, clipped);
187
188                        int offset = 0;
189                        for (Edit edit : edits) {
190                            if (edit.replacement.equals(SymbolList.EMPTY_LIST) && exon.contains(edit.pos)) {
191                                exon.sequence.edit(new Edit(edit.pos + offset, 0, edit.replacement));
192                                offset += edit.length;
193                            }
194                        }
195                    }
196                }
197            }
198
199            // todo: improve this data structure
200            Map<String, ListMultimap<Integer, Allele>> contigs = new HashMap<String, ListMultimap<Integer, Allele>>();
201
202            for (int index : regions.keySet()) {
203                for (Allele allele : regions.get(index)) {
204                    int sequenceLength = allele.sequence.seqString().length();
205                    List<String> fields = Splitter.on("|").splitToList(allele.getName());
206
207                    Double locusLength = Double.parseDouble(fields.get(fields.size() - 1));
208
209                    if (sequenceLength / locusLength >= minimumBreadth) {
210                        if (!contigs.containsKey(fields.get(0))) {
211                            contigs.put(fields.get(0), ArrayListMultimap.<Integer, Allele>create());
212                        }
213                        contigs.get(fields.get(0)).put(index, allele);
214
215                        if (!cdna) {
216                            writer.println(allele.getName() + "|" + sequenceLength);
217                            writer.println(allele.sequence.seqString().toUpperCase());
218                        }
219                    }
220                }
221            }
222
223            if (cdna) {
224                Map<String, String> cdnas = new HashMap<String, String>();
225                for (String contig : contigs.keySet()) {
226                    StringBuilder sb = new StringBuilder();
227
228                    for (int index : contigs.get(contig).keySet()) {
229                        List<Allele> list = contigs.get(contig).get(index);
230                        Collections.sort(list, new Comparator<Allele>() {
231                                @Override
232                                public int compare(final Allele first, final Allele second) {
233                                    return first.sequence.seqString().length() - second.sequence.seqString().length();
234                                }
235                            });
236                        Allele best = list.get(0);
237                        sb.append(best.sequence.seqString());
238                    }
239
240                    String cdnaSequence = sb.toString();
241                    if (removeGaps) {
242                        cdnaSequence = cdnaSequence.replaceAll("-", "");
243                    }
244
245                    // todo:  use strand from genomic region file
246                    if (!(gene.equals("HLA-A") || gene.equals("HLA-DPB1"))) {
247                        cdnaSequence = DNATools.reverseComplement(DNATools.createDNA(cdnaSequence)).seqString();
248                    }
249
250                    cdnas.put(contig, cdnaSequence.toUpperCase());
251                }
252
253                List<Map.Entry<String, String>> list = new ArrayList<Map.Entry<String, String>>(cdnas.entrySet());
254                Collections.sort(list, new Comparator<Map.Entry<String, String>>() {
255                    @Override
256                    public int compare(final Map.Entry<String, String> first, final Map.Entry<String, String> second) {
257                        return second.getValue().length() - first.getValue().length();
258                    }
259                });
260
261                for (Map.Entry<String, String> entry : list.subList(0, expectedPloidy > list.size() ? list.size() : expectedPloidy)) {
262                  writer.println(entry.getKey() + "\n" + entry.getValue());
263                }
264            }
265
266            return 0;
267        }
268        finally {
269            try {
270                writer.close();
271            }
272            catch (Exception e) {
273                // empty
274            }
275        }
276    }

279    static List<Edit> cigarToEditList(final SAMRecord record) throws IllegalSymbolException {
280        int position = 0;
281        List<Edit> edits = new ArrayList<Edit>();
282        for (CigarElement element : record.getCigar().getCigarElements()) {
283            position += element.getLength();
284
285            String operator = element.getOperator().toString();
286            if ("I".equals(operator) || "S".equals(operator)) {
287                edits.add(new Edit(position - element.getLength() + 1, element.getLength(), SymbolList.EMPTY_LIST));
288                position -= element.getLength();
289            }
290            else if ("D".equals(operator)) {
291                SymbolList replace = DNATools.createDNA(Joiner.on("").join(Collections.nCopies(element.getLength(), "-")));
292                edits.add(new Edit(position + element.getLength() - 1, 0, replace));
293            }
294        }
295        return edits;
296    }

298    static Map<Integer, Allele> readBedFile(final File bedFile) throws IOException {
299        BufferedReader reader = null;
300        final Map<Integer, Allele> exons = new HashMap<Integer, Allele>();
301        try {
302            reader = reader(bedFile);
303            BedReader.stream(reader, new BedListener() {
304                    @Override
305                     public boolean record(final BedRecord rec) {
306                        try {
307                            int exon = Integer.parseInt(rec.name());
308                            Locus locus = FeatureParser.parseLocus(rec.chrom() + ":" + rec.start() + "-" + rec.end());
309                            exons.put(exon, Allele.builder()
310                                      .withContig(locus.getContig())
311                                      .withStart(locus.getMin())
312                                      .withEnd(locus.getMax())
313                                      .build());
314                        }
315                        catch (AlleleException | NumberFormatException | NullPointerException | ParseException e) {
316                            throw new ConversionException(rec, e);
317                        }
318                        return true;
319                    }
320                });
321        }
322        catch (ConversionException e) {
323            throw new IOException("could not convert BED record \"" + e.rec() + "\" to allele", e.getCause());
324        }
325        finally {
326            try {
327                reader.close();
328            }
329            catch (Exception e) {
330                // ignore
331            }
332        }
333        return exons;
334    }
335
336    private static final class ConversionException extends RuntimeException {
337        private final BedRecord rec;
338
339        private ConversionException(final BedRecord rec, final Exception cause) {
340            super(cause);
341            this.rec = rec;
342        }
343
344        private BedRecord rec() {
345            return rec;
346        }
347    }

349    /**
350     * Main.
351     *
352     * @param args command line args
353     */
354    public static void main(final String[] args) {
355        Switch about = new Switch("a", "about", "display about message");
356        Switch help = new Switch("h", "help", "display help message");
357        FileArgument bamFile = new FileArgument("i", "bam-file", "input BAM file of consensus sequences", true);
358        FileArgument bedFile = new FileArgument("x", "bed-file", "input BED file of genomic regions", true);
359        FileArgument outputFile = new FileArgument("o", "output-file", "output FASTA file, default stdout", false);
360        StringArgument gene = new StringArgument("g", "gene", "gene name, written to the FASTA headers", true);
361        Switch cdna = new Switch("c", "cdna", "output cDNA from the same contig (phased consensus sequence) in FASTA format");
362        Switch removeGaps = new Switch("r", "remove-gaps", "remove alignment gaps in the filtered consensus sequence");
363        DoubleArgument minimumBreadth = new DoubleArgument("b", "minimum-breadth-of-coverage", "filter contigs less than minimum, default " + DEFAULT_MINIMUM_BREADTH, false);
364        IntegerArgument expectedPloidy = new IntegerArgument("p", "expected-ploidy", "filter contigs more than expected ploidy, default " + DEFAULT_EXPECTED_PLOIDY, false);
365
366        ArgumentList arguments = new ArgumentList(about, help, bamFile, bedFile, outputFile, gene, cdna, removeGaps, minimumBreadth, expectedPloidy);
367        CommandLine commandLine = new CommandLine(args);
368
369        FilterConsensus filterConsensus = null;
370        try
371        {
372            CommandLineParser.parse(commandLine, arguments);
373            if (about.wasFound()) {
374                About.about(System.out);
375                System.exit(0);
376            }
377            if (help.wasFound()) {
378                Usage.usage(USAGE, null, commandLine, arguments, System.out);
379                System.exit(0);
380            }
381
382            if (!bamFile.getValue().exists()) {
383                throw new IllegalArgumentException("-i, --bam-file must be a file that exists");
384            }
385            if (!bedFile.getValue().exists()) {
386                throw new IllegalArgumentException("-x, --bed-file must be a file that exists");
387            }
388
389            filterConsensus = new FilterConsensus(bamFile.getValue(),
390                                                  bedFile.getValue(),
391                                                  outputFile.getValue(),
392                                                  gene.getValue(),
393                                                  cdna.wasFound(),
394                                                  removeGaps.wasFound(),
395                                                  minimumBreadth.getValue(DEFAULT_MINIMUM_BREADTH),
396                                                  expectedPloidy.getValue(DEFAULT_EXPECTED_PLOIDY));
397        }
398        catch (CommandLineParseException | IllegalArgumentException e) {
399            if (about.wasFound()) {
400                About.about(System.out);
401                System.exit(0);
402            }
403            if (help.wasFound()) {
404                Usage.usage(USAGE, null, commandLine, arguments, System.out);
405                System.exit(0);
406            }
407            Usage.usage(USAGE, e, commandLine, arguments, System.err);
408            System.exit(-1);
409        }
410        try {
411            System.exit(filterConsensus.call());
412        }
413        catch (Exception e) {
414            e.printStackTrace();
415            System.exit(1);
416        }
417    }
