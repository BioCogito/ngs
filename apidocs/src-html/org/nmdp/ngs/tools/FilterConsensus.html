080/**
081 * Filter consensus sequences into subregions of research or clinical interest.
082 */
083public final class FilterConsensus implements Callable<Integer> {
084    private final File inputBamFile;
085    private final File inputGenomicFile;
086    private final File outputFile;
087    private final String gene;
088    private final boolean cdna;
089    private final boolean removeGaps;
090    private final double minimumBreadth;
091    private final int expectedPloidy;
092    static final double DEFAULT_MINIMUM_BREADTH = 0.5d;
093    static final int DEFAULT_EXPECTED_PLOIDY = 2;
094    static final String USAGE = "ngs-filter-consensus -i input.bam -x genome.txt -g HLA-A [args]";
095
096
097    /**
098     * Filter consensus sequences into subregions of research or clinical interest.
099     *
100     * @param inputBamFile input BAM file, must not be null
101     * @param inputGenomicFile input genomic region file, must not be null
102     * @param outputFile output FASTA file, if any
103     * @param gene gene, must not be null
104     * @param cdna cdna, true to output cdna from the same contig (phased consensus sequence)
105     *    in FASTA format (for interpretation)
106     * @param removeGaps remove alignment gaps in the filtered consensus sequence
107     * @param minimumBreadth minimum breadth, must be in range [0.0 - 1.0]
108     * @param expectedPloidy expected ploidy, must be &gt; 0
109     */
110    public FilterConsensus(final File inputBamFile,
111                           final File inputGenomicFile,
112                           final File outputFile,
113                           final String gene,
114                           final boolean cdna,
115                           final boolean removeGaps,
116                           final double minimumBreadth,
117                           final int expectedPloidy) {
118        checkNotNull(inputBamFile);
119        checkNotNull(inputGenomicFile);
120        checkNotNull(gene);
121        checkArgument(minimumBreadth >= 0.0d && minimumBreadth <= 1.0d, "minimum breadth must be in range [0.0 - 1.0]");
122        checkArgument(expectedPloidy > 0, "expected ploidy must be > 0");
123        this.inputBamFile = inputBamFile;
124        this.inputGenomicFile = inputGenomicFile;
125        this.outputFile = outputFile;
126        this.gene = gene;
127        this.cdna = cdna;
128        this.removeGaps = removeGaps;
129        this.minimumBreadth = minimumBreadth;
130        this.expectedPloidy = expectedPloidy;
131    }
132
133
134    @Override
135    public Integer call() throws Exception {
136        PrintWriter writer = null;
137        try {
138            writer = writer(outputFile);
139
140            // map of region alleles from genomic file keyed by "exon" as integer
141            Map<Integer, Allele> exons = readGenomicFile(inputGenomicFile);
142
143            // todo:  refactor this block into separate static methods
144            // map of overlapping alignment alleles from BAM file keyed by "exon" as integer
145            ListMultimap<Integer, Allele> regions = ArrayListMultimap.create();
146
147            for (SAMRecord record : new SAMFileReader(inputBamFile)) {
148                List<Edit> edits = cigarToEditList(record);
149
150                String referenceName = record.getReferenceName();
151                int start = record.getAlignmentStart();
152                int end = record.getAlignmentEnd();
153                Locus alignment = new Locus(referenceName, start, end);
154                SymbolList sequence = DNATools.createDNA(record.getReadString());
155
156                for (Edit edit : edits) {
157                    sequence.edit(edit);
158                }
159
160                String name = record.getReadName();
161                Allele contig = Allele.builder()
162                    .withContig(referenceName)
163                    .withStart(start)
164                    .withEnd(end)
165                    .withSequence(sequence)
166                    .build();
167
168                for (Map.Entry<Integer, Allele> entry : exons.entrySet()) {
169                    int index = entry.getKey();
170                    Allele exon = entry.getValue();
171
172                    String chr = exon.getContig();
173                    int min = exon.getMin();
174                    int max = exon.getMax();
175                    String range = chr + ":" + min + "-" + max;
176
177                    if (alignment.overlaps(exon)) {
178                        Allele xover = exon.doubleCrossover(contig);
179
180                        Allele clipped = xover.leftHardClip("-").rightHardClip("-");
181                        clipped.setName(">" + name + "|gene=" + gene + "|exon=" + index + "|location=" + range + "|" + (max - min));
182                        regions.put(index, clipped);
183
184                        int offset = 0;
185                        for (Edit edit : edits) {
186                            if (edit.replacement.equals(SymbolList.EMPTY_LIST) && exon.contains(edit.pos)) {
187                                exon.sequence.edit(new Edit(edit.pos + offset, 0, edit.replacement));
188                                offset += edit.length;
189                            }
190                        }
191                    }
192                }
193            }
194
195            // todo: improve this data structure
196            Map<String, ListMultimap<Integer, Allele>> contigs = new HashMap<String, ListMultimap<Integer, Allele>>();
197
198            for (int index : regions.keySet()) {
199                for (Allele allele : regions.get(index)) {
200                    int sequenceLength = allele.sequence.seqString().length();
201                    List<String> fields = Splitter.on("|").splitToList(allele.getName());
202
203                    Double locusLength = Double.parseDouble(fields.get(fields.size() - 1));
204
205                    if (sequenceLength/locusLength >= minimumBreadth) {
206                        if (!contigs.containsKey(fields.get(0))) {
207                            contigs.put(fields.get(0), ArrayListMultimap.<Integer, Allele>create());
208                        }
209                        contigs.get(fields.get(0)).put(index, allele);
210
211                        if (!cdna) {
212                            writer.println(allele.getName() + "|" + sequenceLength);
213                            writer.println(allele.sequence.seqString().toUpperCase());
214                        }
215                    }
216                }
217            }
218
219            if (cdna) {
220                Map<String, String> cdnas = new HashMap<String, String>();
221                for (String contig : contigs.keySet()) {
222                    StringBuilder sb = new StringBuilder();
223
224                    for (int index : contigs.get(contig).keySet()) {
225                        List<Allele> list = contigs.get(contig).get(index);
226                        Collections.sort(list, new Comparator<Allele>() {
227                                @Override
228                                public int compare(final Allele first, final Allele second) {
229                                    return first.sequence.seqString().length() - second.sequence.seqString().length();
230                                }
231                            });
232                        Allele best = list.get(0);
233                        sb.append(best.sequence.seqString());
234                    }
235
236                    String cdnaSequence = sb.toString();
237                    if (removeGaps) {
238                        cdnaSequence = cdnaSequence.replaceAll("-", "");
239                    }
240
241                    // todo:  use strand from genomic region file
242                    if (!(gene.equals("HLA-A") || gene.equals("HLA-DPB1"))) {
243                        cdnaSequence = DNATools.reverseComplement(DNATools.createDNA(cdnaSequence)).seqString();
244                    }
245
246                    cdnas.put(contig, cdnaSequence.toUpperCase());
247                }
248                
249                List<Map.Entry<String, String>> list = new ArrayList<Map.Entry<String, String>>(cdnas.entrySet());
250                Collections.sort(list, new Comparator<Map.Entry<String, String>>() {
251                    @Override
252                    public int compare(final Map.Entry<String, String> first, final Map.Entry<String, String> second) {
253                        return second.getValue().length() - first.getValue().length();
254                    }
255                });
256                
257                for (Map.Entry<String, String> entry : list.subList(0, expectedPloidy > list.size() ? list.size() : expectedPloidy)) {
258                  writer.println(entry.getKey() + "\n" + entry.getValue());
259                }
260            }
261
262            return 0;
263        }
264        finally {
265            try {
266                writer.close();
267            }
268            catch (Exception e) {
269                // empty
270            }
271        }
272    }

275    static List<Edit> cigarToEditList(final SAMRecord record) throws IllegalSymbolException {
276        int position = 0;
277        List<Edit> edits = new ArrayList<Edit>();
278        for (CigarElement element : record.getCigar().getCigarElements()) {
279            position += element.getLength();
280
281            String operator = element.getOperator().toString();
282            if ("I".equals(operator) || "S".equals(operator)) {
283                edits.add(new Edit(position - element.getLength() + 1, element.getLength(), SymbolList.EMPTY_LIST));
284                position -= element.getLength();
285            }
286            else if ("D".equals(operator)) {
287                SymbolList replace = DNATools.createDNA(Joiner.on("").join(Collections.nCopies(element.getLength(), "-")));
288                edits.add(new Edit(position + element.getLength() - 1, 0, replace));
289            }
290        }
291        return edits;
292    }

294    // todo:  use BED format; extract BED format reader from e.g. LiftoverBed
295    static Map<Integer, Allele> readGenomicFile(final File inputGenomicFile) throws IOException {
296        int lineNumber = 0;
297        BufferedReader reader = null;
298        Map<Integer, Allele> exons = new HashMap<Integer, Allele>();
299        try {
300            reader = reader(inputGenomicFile);
301
302            while (reader.ready()) {
303                String line = reader.readLine();
304                if (line == null) {
305                    break;
306                }
307                String[] tokens = line.split("\\s+");
308                if (tokens.length != 2) {
309                    throw new IOException("invalid genomic file format at line " + lineNumber + ", invalid number of tokens");
310                }
311
312                try {
313                    int exon = Integer.parseInt(tokens[0]);
314                    Locus locus = FeatureParser.parseLocus(tokens[1]);
315                    exons.put(exon, Allele.builder()
316                              .withContig(locus.getContig())
317                              .withStart(locus.getMin())
318                              .withEnd(locus.getMax() + 1)
319                              .build());
320                }
321                catch (NumberFormatException | AlleleException | ParseException e) {
322                    throw new IOException("invalid genomic file format at line " + lineNumber + ", caught " + e.getMessage());
323                }
324
325                lineNumber++;
326            }
327        }
328        finally {
329            try {
330                reader.close();
331            }
332            catch (Exception e) {
333                // ignore
334            }
335        }
336        return exons;
337    }

340    /**
341     * Main.
342     *
343     * @param args command line args
344     */
345    public static void main(final String[] args) {
346        Switch about = new Switch("a", "about", "display about message");
347        Switch help = new Switch("h", "help", "display help message");
348        FileArgument inputBamFile = new FileArgument("i", "input-bam-file", "input BAM file", true);
349        FileArgument inputGenomicFile = new FileArgument("x", "input-genomic-range-file", "input file of genomic ranges, space-delimited exon chrom:start-end", true);
350        FileArgument outputFile = new FileArgument("o", "output-file", "output FASTA file, default stdout", false);
351        StringArgument gene = new StringArgument("g", "gene", "gene name, written to the FASTA headers", true);
352        Switch cdna = new Switch("c", "cdna", "output cDNA from the same contig (phased consensus sequence) in FASTA format");
353        Switch removeGaps = new Switch("r", "remove-gaps", "remove alignment gaps in the filtered consensus sequence");
354        DoubleArgument minimumBreadth = new DoubleArgument("b", "minimum-breadth-of-coverage", "filter contigs less than minimum, default " + DEFAULT_MINIMUM_BREADTH, false);
355        IntegerArgument expectedPloidy = new IntegerArgument("p", "expected-ploidy", "filter contigs more than expected ploidy, default " + DEFAULT_EXPECTED_PLOIDY, false);
356
357        ArgumentList arguments = new ArgumentList(about, help, inputBamFile, inputGenomicFile, outputFile, gene, cdna, removeGaps, minimumBreadth, expectedPloidy);
358        CommandLine commandLine = new CommandLine(args);
359
360        FilterConsensus filterConsensus = null;
361        try
362        {
363            CommandLineParser.parse(commandLine, arguments);
364            if (about.wasFound()) {
365                About.about(System.out);
366                System.exit(0);
367            }
368            if (help.wasFound()) {
369                Usage.usage(USAGE, null, commandLine, arguments, System.out);
370                System.exit(0);
371            }
372
373            if (!inputBamFile.getValue().exists()) {
374                throw new IllegalArgumentException("-i, --input-bam-file must be a file that exists");
375            }
376            if (!inputGenomicFile.getValue().exists()) {
377                throw new IllegalArgumentException("-x, --input-genomic-range-file must be a file that exists");
378            }
379
380            filterConsensus = new FilterConsensus(inputBamFile.getValue(),
381                                                  inputGenomicFile.getValue(),
382                                                  outputFile.getValue(),
383                                                  gene.getValue(),
384                                                  cdna.wasFound(),
385                                                  removeGaps.wasFound(),
386                                                  minimumBreadth.getValue(DEFAULT_MINIMUM_BREADTH),
387                                                  expectedPloidy.getValue(DEFAULT_EXPECTED_PLOIDY));
388        }
389        catch (CommandLineParseException | IllegalArgumentException e) {
390            if (about.wasFound()) {
391                About.about(System.out);
392                System.exit(0);
393            }
394            if (help.wasFound()) {
395                Usage.usage(USAGE, null, commandLine, arguments, System.out);
396                System.exit(0);
397            }
398            Usage.usage(USAGE, e, commandLine, arguments, System.err);
399            System.exit(-1);
400        }
401        try {
402            System.exit(filterConsensus.call());
403        }
404        catch (Exception e) {
405            e.printStackTrace();
406            System.exit(1);
407        }
408    }
