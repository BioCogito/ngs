023package org.nmdp.ngs.tools;
024
025import static org.dishevelled.compress.Readers.reader;
026import static org.dishevelled.compress.Writers.writer;
027
028import java.io.BufferedReader;
029import java.io.File;
030import java.io.PrintWriter;
031
032import java.util.concurrent.Callable;
033
034import org.biojava.bio.program.fastq.Fastq;
035import org.biojava.bio.program.fastq.FastqReader;
036import org.biojava.bio.program.fastq.SangerFastqReader;
037import org.biojava.bio.program.fastq.StreamListener;
038
039import org.dishevelled.commandline.ArgumentList;
040import org.dishevelled.commandline.CommandLine;
041import org.dishevelled.commandline.CommandLineParseException;
042import org.dishevelled.commandline.CommandLineParser;
043import org.dishevelled.commandline.Switch;
044import org.dishevelled.commandline.Usage;
045
046import org.dishevelled.commandline.argument.FileArgument;
047
048/**
049 * Output description lines from sequences in FASTQ format.
050 */
051public final class FastqDescription implements Callable<Integer> {
052    private final File fastqFile;
053    private final File descriptionFile;
054    private final FastqReader fastqReader = new SangerFastqReader();
055    private static final String USAGE = "ngs-fastq-description [args]";
056
057
058    /**
059     * Output description lines from sequences in FASTQ format.
060     *
061     * @param fastqFile input FASTQ file, if any
062     * @param descriptionFile output file of description lines, if any
063     */
064    public FastqDescription(final File fastqFile, final File descriptionFile) {
065        this.fastqFile = fastqFile;
066        this.descriptionFile = descriptionFile;
067    }
068
069
070    @Override
071    public Integer call() throws Exception {
072        BufferedReader reader = null;
073        PrintWriter writer = null;
074        try {
075            reader = reader(fastqFile);
076            writer = writer(descriptionFile);
077
078            final PrintWriter w = writer;
079            fastqReader.stream(reader, new StreamListener() {
080                    @Override
081                    public void fastq(final Fastq fastq)
082                    {
083                        w.println(fastq.getDescription());
084                    }
085                });
086
087            return 0;
088        }
089        finally {
090            try {
091                reader.close();
092            }
093            catch (Exception e) {
094                // ignore
095            }
096            try {
097                writer.close();
098            }
099            catch (Exception e) {
100                // ignore
101            }
102        }
103    }
104
105    /**
106     * Main.
107     *
108     * @param args command line args
109     */
110    public static void main(final String[] args) {
111        Switch about = new Switch("a", "about", "display about message");
112        Switch help = new Switch("h", "help", "display help message");
113        FileArgument fastqFile = new FileArgument("i", "fastq-file", "input FASTQ file, default stdin", false);
114        FileArgument descriptionFile = new FileArgument("o", "description-file", "output file of description lines, default stdout", false);
115
116        ArgumentList arguments = new ArgumentList(about, help, fastqFile, descriptionFile);
117        CommandLine commandLine = new CommandLine(args);
118
119        FastqDescription fastqDescription = null;
120        try
121        {
122            CommandLineParser.parse(commandLine, arguments);
123            if (about.wasFound()) {
124                About.about(System.out);
125                System.exit(0);
126            }
127            if (help.wasFound()) {
128                Usage.usage(USAGE, null, commandLine, arguments, System.out);
129                System.exit(0);
130            }
131            fastqDescription = new FastqDescription(fastqFile.getValue(), descriptionFile.getValue());
132        }
133        catch (CommandLineParseException e) {
134            Usage.usage(USAGE, e, commandLine, arguments, System.err);
135            System.exit(-1);
136        }
137        try {
138            System.exit(fastqDescription.call());
139        }
140        catch (Exception e) {
141            e.printStackTrace();
142            System.exit(1);
143        }
144    }
145}
