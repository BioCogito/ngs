023package org.nmdp.ngs.tools;
024
025import static com.google.common.base.Preconditions.checkArgument;
026import static com.google.common.base.Preconditions.checkNotNull;
027
028import static org.dishevelled.compress.Writers.writer;
029
030import java.io.File;
031import java.io.PrintWriter;
032
033import java.util.concurrent.Callable;
034
035import org.apache.commons.math3.distribution.IntegerDistribution;
036import org.apache.commons.math3.distribution.NormalDistribution;
037import org.apache.commons.math3.distribution.RealDistribution;
038import org.apache.commons.math3.distribution.UniformIntegerDistribution;
039
040import org.apache.commons.math3.random.MersenneTwister;
041import org.apache.commons.math3.random.RandomGenerator;
042
043import org.dishevelled.commandline.ArgumentList;
044import org.dishevelled.commandline.CommandLine;
045import org.dishevelled.commandline.CommandLineParseException;
046import org.dishevelled.commandline.CommandLineParser;
047import org.dishevelled.commandline.Switch;
048import org.dishevelled.commandline.Usage;
049
050import org.dishevelled.commandline.argument.DoubleArgument;
051import org.dishevelled.commandline.argument.FileArgument;
052import org.dishevelled.commandline.argument.IntegerArgument;
053import org.dishevelled.commandline.argument.StringArgument;
054
055import org.nmdp.ngs.align.BedRecord;
056import org.nmdp.ngs.align.BedWriter;
057
058/**
059 * Generate records in BED format.
060 */
061public final class GenerateBed implements Callable<Integer> {
062    private final File bedFile;
063    private final int n;
064    private final int size;
065    private final String chrom;
066    private final RandomGenerator random;
067    private final RealDistribution length;
068    private final IntegerDistribution location;
069    private static final int DEFAULT_N = 1000;
070    private static final int DEFAULT_SIZE = 100000;
071    private static final String DEFAULT_CHROM = "1";
072    private static final double NO_VARIATION = 1.0E-100;
073    private static final double DEFAULT_MEAN_LENGTH = 100.0d;
074    private static final double DEFAULT_LENGTH_VARIATION = 10.0d;
075    private static final String USAGE = "ngs-generate-bed [args]";
076
077
078    /**
079     * Generate records in BED format.
080     *
081     * @param bedFile output BED file, if any
082     * @param n number of BED records to generate, must be at least zero
083     * @param size chromosome size, must be at least zero
084     * @param chrom chromosome name, must not be null
085     * @param random random generator, must not be null
086     * @param length length distribution, must not be null
087     */
088    public GenerateBed(final File bedFile, final int n, final int size, final String chrom, final RandomGenerator random, final RealDistribution length) {
089        checkArgument(n >= 0, "n must be at least zero");
090        checkArgument(size >= 0, "size must be at least zero");
091        checkNotNull(chrom);
092        checkNotNull(random);
093        checkNotNull(length);
094        this.bedFile = bedFile;
095        this.n = n;
096        this.size = size;
097        this.chrom = chrom;
098        this.random = random;
099        this.length = length;
100
101        int flanking = (int) (length.getNumericalMean() + length.getNumericalVariance());
102        location = new UniformIntegerDistribution(this.random, 1 - flanking, size + flanking);
103    }
104
105
106    @Override
107    public Integer call() throws Exception {
108        PrintWriter writer = null;
109        try {
110            writer = writer(bedFile);
111
112            for (int i = 0; i < n; i++) {
113                // sample location and length
114                long start = (long) location.sample();
115                long end = start + 1L + (long) length.sample();
116
117                // trim to chrom ends
118                long adjStart = Math.min((long) (size - 1), Math.max(0L, start));
119                long adjEnd = Math.max(adjStart + 1L, Math.min((long) (size - 1), end));
120
121                // and write
122                BedWriter.write(new BedRecord(chrom, adjStart, adjEnd, "record" + i, "0", "+"), writer);
123            }
124
125            return 0;
126        }
127        finally {
128            try {
129                writer.close();
130            }
131            catch (Exception e) {
132                // ignore
133            }
134        }
135    }
136
137
138    /**
139     * Main.
140     *
141     * @param args command line arguments
142     */
143    public static void main(final String[] args) {
144        Switch about = new Switch("a", "about", "display about message");
145        Switch help = new Switch("h", "help", "display help message");
146        FileArgument bedFile = new FileArgument("o", "bed-file", "output BED file, default stdout", false);
147        IntegerArgument n = new IntegerArgument("n", "n", "number of BED records to generate, default " + DEFAULT_N, false);
148        IntegerArgument size = new IntegerArgument("s", "size", "chromosome size, default " + DEFAULT_SIZE, false);
149        StringArgument chrom = new StringArgument("c", "chrom", "chromosome name, default " + DEFAULT_CHROM, false);
150        DoubleArgument meanLength = new DoubleArgument("l", "mean-length", "mean length, default " + DEFAULT_MEAN_LENGTH, false);
151        DoubleArgument lengthVariation = new DoubleArgument("v", "length-variation", "length variation, default " + DEFAULT_LENGTH_VARIATION, false);
152        IntegerArgument seed = new IntegerArgument("z", "seed", "random number seed, default relates to current time", false);
153
154        ArgumentList arguments = new ArgumentList(about, help, bedFile, n, size, chrom, meanLength, lengthVariation, seed);
155        CommandLine commandLine = new CommandLine(args);
156
157        GenerateBed generateBed = null;
158        try
159        {
160            CommandLineParser.parse(commandLine, arguments);
161            if (about.wasFound()) {
162                About.about(System.out);
163                System.exit(0);
164            }
165            if (help.wasFound()) {
166                Usage.usage(USAGE, null, commandLine, arguments, System.out);
167                System.exit(0);
168            }
169
170            RandomGenerator random = seed.wasFound() ? new MersenneTwister(seed.getValue()) : new MersenneTwister();
171            double lv = Math.max(NO_VARIATION, lengthVariation.getValue(DEFAULT_LENGTH_VARIATION));
172            RealDistribution length = new NormalDistribution(random, meanLength.getValue(DEFAULT_MEAN_LENGTH), lv, NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
173
174            generateBed = new GenerateBed(bedFile.getValue(), n.getValue(DEFAULT_N), size.getValue(DEFAULT_SIZE), chrom.getValue(DEFAULT_CHROM), random, length);
175        }
176        catch (CommandLineParseException e) {
177            Usage.usage(USAGE, e, commandLine, arguments, System.err);
178            System.exit(-1);
179        }
180        try {
181            System.exit(generateBed.call());
182        }
183        catch (Exception e) {
184            e.printStackTrace();
185            System.exit(1);
186        }
187    }
