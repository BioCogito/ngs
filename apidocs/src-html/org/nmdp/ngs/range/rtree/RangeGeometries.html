023package org.nmdp.ngs.range.rtree;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import com.github.davidmoten.rtree.geometry.Geometries;
028import com.github.davidmoten.rtree.geometry.Point;
029import com.github.davidmoten.rtree.geometry.Rectangle;
030
031import com.google.common.collect.BoundType;
032import com.google.common.collect.Range;
033
034/**
035 * Geometries for use in building R-Trees and R*-Trees from Guava ranges.
036 *
037 * <pre>
038 * import static org.nmdp.ngs.range.rtree.RangeGeometries.*;
039 *
040 * RTree&lt;String&gt; rtree = RTree.create();
041 * rtree = rtree.add("foo", closed(10, 20));
042 * rtree = rtree.add("bar", closedOpen(14, 28));
043 * rtree = rtree.add("baz", open(18, 36));
044 *
045 * for (String result : rtree.search(singleton(20)).toBlocking().toIterable()) {
046 *   // ...
047 * }
048 * </pre>
049 */
050public final class RangeGeometries {
051
052    /**
053     * Private no-arg constructor.
054     */
055    private RangeGeometries() {
056        // empty
057    }
058
059
060    /**
061     * Create and return a new point geometry from the specified singleton value.
062     *
063     * @param value singleton value, must not be null
064     * @return a new point geometry from the specified singleton value
065     */
066    public static <N extends Number & Comparable<? super N>> Point singleton(final N value) {
067        checkNotNull(value);
068        return Geometries.point(value.doubleValue(), 0.5d);
069    }
070
071    /**
072     * Create and return a new rectangle geometry from the specified closed range <code>[lower..upper]</code>.
073     *
074     * @param lower lower endpoint, must not be null
075     * @param upper upper endpoint, must not be null
076     * @return a new rectangle geometry from the specified closed range
077     */
078    public static <N extends Number & Comparable<? super N>> Rectangle closed(final N lower, final N upper) {
079        checkNotNull(lower);
080        checkNotNull(upper);
081        return range(Range.closed(lower, upper));
082    }
083
084    /**
085     * Create and return a new rectangle geometry from the specified closed open range <code>[lower..upper)</code>.
086     *
087     * @param lower lower endpoint, must not be null
088     * @param upper upper endpoint, must not be null
089     * @return a new rectangle geometry from the specified closed range
090     */
091    public static <N extends Number & Comparable<? super N>> Rectangle closedOpen(final N lower, final N upper) {
092        checkNotNull(lower);
093        checkNotNull(upper);
094        return range(Range.closedOpen(lower, upper));
095    }
096
097    /**
098     * Create and return a new rectangle geometry from the specified open closed range <code>(lower..upper]</code>
099     *
100     * @param lower lower endpoint, must not be null
101     * @param upper upper endpoint, must not be null
102     * @return a new rectangle geometry from the specified closed range
103     */
104    public static <N extends Number & Comparable<? super N>> Rectangle openClosed(final N lower, final N upper) {
105        checkNotNull(lower);
106        checkNotNull(upper);
107        return range(Range.openClosed(lower, upper));
108    }
109    
110    /**
111     * Create and return a new rectangle geometry from the specified open range <code>(lower..upper)</code>.
112     *
113     * @param lower lower endpoint, must not be null
114     * @param upper upper endpoint, must not be null
115     * @return a new rectangle geometry from the specified closed range
116     */
117    public static <N extends Number & Comparable<? super N>> Rectangle open(final N lower, final N upper) {
118        checkNotNull(lower);
119        checkNotNull(upper);
120        return range(Range.open(lower, upper));
121    }
122
123    /**
124     * Create and return a new rectangle geometry from the specified range.
125     *
126     * @param range range, must not be null, must not be empty, and must have lower and upper bounds
127     * @return a new rectangle geometry from the specified range
128     */
129    public static <N extends Number & Comparable<? super N>> Rectangle range(final Range<N> range) {
130        checkNotNull(range);
131        if (range.isEmpty()) {
132            throw new IllegalArgumentException("range must not be empty");
133        }
134        if (!range.hasLowerBound() || !range.hasUpperBound()) {
135            throw new IllegalArgumentException("range must have lower and upper bounds");
136        }
137        Number lowerEndpoint = range.lowerEndpoint();
138        BoundType lowerBoundType = range.lowerBoundType();
139        Number upperEndpoint = range.upperEndpoint();
140        BoundType upperBoundType = range.upperBoundType();
141
142        /*
143
144          Since we are representing genomic coordinate systems, the expectation is
145          that endpoints are instance of Integer, Long, or BigInteger; thus for open
146          lower and upper bounds we can safely add or substract 1.0 respectively.
147
148          Then by convention a rectangle with y1 0.0 and height of 1.0 is used.
149
150          closed(10, 20) --> (10.0, 0.0, 20.0, 1.0)
151          closedOpen(10, 20) --> (10.0, 0.0, 19.0, 1.0)
152          openClosed(10, 20) --> (11.0, 0.0, 20.0, 1.0)
153          open(10, 20) --> (11.0, 0.0, 19.0, 1.0);
154
155          closed(10, 11) --> (10.0, 0.0, 11.0, 1.0)
156          closedOpen(10, 11) --> (10.0, 0.0, 10.0, 1.0)
157          openClosed(10, 11) --> (11.0, 0.0, 11.0, 1.0)
158          open(10, 11) --> empty, throw exception
159
160          closed(10, 10) --> (10.0, 0.0, 10.0, 1.0)
161          closedOpen(10, 10) --> empty, throw exception
162          openClosed(10, 10) --> empty, throw exception
163          open(10, 10) --> empty, throw exception
164
165        */
166        double x1 = lowerBoundType == BoundType.OPEN ? lowerEndpoint.doubleValue() + 1.0d : lowerEndpoint.doubleValue();
167        double y1 = 0.0d;
168        double x2 = upperBoundType == BoundType.OPEN ? upperEndpoint.doubleValue() - 1.0d : upperEndpoint.doubleValue();
169        double y2 = 1.0d;
170        return Geometries.rectangle(x1, y1, x2, y2);
171    }
172}
