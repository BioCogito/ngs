023package org.nmdp.ngs.range;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.math.BigInteger;
028
029import com.google.common.collect.BoundType;
030import com.google.common.collect.ComparisonChain;
031import com.google.common.collect.Ordering;
032import com.google.common.collect.Range;
033
034/**
035 * Utility methods on ranges.
036 */
037public final class Ranges {
038
039    /**
040     * Private no-arg constructor.
041     */
042    private Ranges() {
043        // empty
044    }
045
046    /**
047     * Return the center of the specified range.
048     *
049     * @param <C> range endpoint type
050     * @param range range, must not be null
051     * @return the center of the specified range
052     */
053    public static <C extends Comparable> C center(final Range<C> range) {
054        checkNotNull(range);
055        if (!range.hasLowerBound() && !range.hasUpperBound()) {
056            throw new IllegalStateException("cannot find the center of a range without bounds");
057        }
058        if (!range.hasLowerBound()) {
059            return range.upperEndpoint();
060        }
061        if (!range.hasUpperBound()) {
062            return range.lowerEndpoint();
063        }
064        C lowerEndpoint = range.lowerEndpoint();
065        C upperEndpoint = range.upperEndpoint();
066
067        if (upperEndpoint instanceof Integer) {
068            Integer upper = (Integer) upperEndpoint;
069            Integer lower = (Integer) lowerEndpoint;
070            return (C) Integer.valueOf((upper.intValue() + lower.intValue()) / 2);
071        }
072        if (upperEndpoint instanceof Long) {
073            Long upper = (Long) upperEndpoint;
074            Long lower = (Long) lowerEndpoint;
075            return (C) Long.valueOf((upper.longValue() + lower.longValue()) / 2L);
076        }
077        if (upperEndpoint instanceof BigInteger) {
078            BigInteger upper = (BigInteger) upperEndpoint;
079            BigInteger lower = (BigInteger) lowerEndpoint;
080            BigInteger two = BigInteger.valueOf(2L);
081            return (C) upper.subtract(lower).divide(two);
082        }
083
084        // todo:  could potentially calculate the center of any range with a discrete domain
085        throw new IllegalStateException("cannot find the center of a range whose endpoint type is not Integer, Long, or BigInteger");
086    }
087
088    /**
089     * Return true if the specified ranges intersect.
090     *
091     * @param <C> range endpoint type
092     * @param range0 first range, must not be null
093     * @param range1 second range, must not be null
094     * @return true if the specified ranges intersect
095     */
096    public static <C extends Comparable> boolean intersect(final Range<C> range0, final Range<C> range1) {
097        checkNotNull(range0);
098        checkNotNull(range1);
099        return range0.isConnected(range1) && !range0.intersection(range1).isEmpty();
100    }
101
102    /**
103     * Return true if the specified range is strictly less than the specified value.
104     *
105     * @param <C> range endpoint type
106     * @param range range, must not be null
107     * @param value value, must not be null
108     * @return true if the specified range is strictly less than the specified value
109     */
110    public static <C extends Comparable> boolean isLessThan(final Range<C> range, final C value) {
111        checkNotNull(range);
112        checkNotNull(value);
113
114        if (!range.hasUpperBound()) {
115            return false;
116        }
117        if (range.upperBoundType() == BoundType.OPEN && range.upperEndpoint().equals(value)) {
118            return true;
119        }
120        return range.upperEndpoint().compareTo(value) < 0;
121    }
122
123    /**
124     * Return true if the specified range is strictly greater than the specified value.
125     *
126     * @param <C> range endpoint type
127     * @param range range, must not be null
128     * @param value value, must not be null
129     * @return true if the specified range is strictly greater than the specified value
130     */
131    public static <C extends Comparable> boolean isGreaterThan(final Range<C> range, final C value) {
132        checkNotNull(range);
133        checkNotNull(value);
134
135        if (!range.hasLowerBound()) {
136            return false;
137        }
138        if (range.lowerBoundType() == BoundType.OPEN && range.lowerEndpoint().equals(value)) {
139            return true;
140        }
141        return range.lowerEndpoint().compareTo(value) > 0;
142    }
143
144
145    /**
146     * Return an ordering by lower endpoint over ranges.
147     *
148     * @param <C> range endpoint type
149     * @return an ordering by lower endpoint over ranges
150     */
151    public static <C extends Comparable> Ordering<Range<C>> orderingByLowerEndpoint() {
152        return new Ordering<Range<C>>() {
153            @Override
154            public int compare(final Range<C> left, final Range<C> right) {
155                return ComparisonChain.start()
156                    .compare(left.hasLowerBound(), right.hasLowerBound())
157                    .compare(left.lowerEndpoint(), right.lowerEndpoint())
158                    .result();
159            }
160        };
161    }
162
163    /**
164     * Return a reverse ordering by lower endpoint over ranges.
165     *
166     * @param <C> range endpoint type
167     * @return a reverse ordering by lower endpoint over ranges
168     */
169    public static <C extends Comparable> Ordering<Range<C>> reverseOrderingByLowerEndpoint() {
170        Ordering<Range<C>> orderingByLowerEndpoint = orderingByLowerEndpoint();
171        return orderingByLowerEndpoint.reverse();
172    }
173
174    /**
175     * Return an ordering by upper endpoint over ranges.
176     *
177     * @param <C> range endpoint type
178     * @return an ordering by upper endpoint over ranges
179     */
180    public static <C extends Comparable> Ordering<Range<C>> orderingByUpperEndpoint() {
181        return new Ordering<Range<C>>() {
182            @Override
183            public int compare(final Range<C> left, final Range<C> right) {
184                return ComparisonChain.start()
185                    .compare(left.hasUpperBound(), right.hasUpperBound())
186                    .compare(left.upperEndpoint(), right.upperEndpoint())
187                    .result();
188            }
189        };
190    }
191
192    /**
193     * Return a reverse ordering by upper endpoint over ranges.
194     *
195     * @param <C> range endpoint type
196     * @return a reverse ordering by upper endpoint over ranges
197     */
198    public static <C extends Comparable> Ordering<Range<C>> reverseOrderingByUpperEndpoint() {
199        Ordering<Range<C>> orderingByUpperEndpoint = orderingByUpperEndpoint();
200        return orderingByUpperEndpoint.reverse();
201    }
202}
