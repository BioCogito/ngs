023package org.nmdp.ngs.feature;
024
025import org.biojava.bio.seq.Sequence;
026
027import org.biojava.bio.symbol.RangeLocation;
028
029import com.google.common.base.Objects;
030
031/**
032 * Locus.
033 */
034public class Locus extends RangeLocation implements Comparable<Locus> {
035    protected String contig;
036
037    public Locus(final String contig, final int start, final int end) {
038        super(start, end);
039        this.contig = contig;
040    }
041  
042    public Locus(final String contig, final RangeLocation range) {
043        super(range.getMin(), range.getMax());
044        this.contig = contig;
045    }
046
047    public final int getStart() {
048        return getMin();
049    }
050
051    public final int getEnd() {
052        return getMax();
053    }
054
055    public boolean isPointLocation() {
056        return this.getStart() == this.getEnd();
057    }
058  
059    public final static class Util {
060        // TODO: use SymbolList instead of String to handle IUPAC ambiguity
061        /**
062         * @param pattern sequence, either inserted or deleted
063         */
064        private static String rotateLeft(final String pattern) {
065            return pattern.substring(pattern.length() - 1) + pattern.substring(0, pattern.length() - 1);
066        }
067
068        /**
069         * @param pattern sequence, either inserted or deleted
070         */
071        private static String rotateRight(final String pattern) {
072            return pattern.substring(1, pattern.length()) + pattern.substring(0, 1);
073        }
074
075        /**
076         * @param reference contig
077         * @param position of the reference contig immediately preceeding the inserted
078         * or deleted pattern. This is to maintain consistency with VCF positioning
079         * but is inconsistent with other variant nomenclatures (eg HGVS)
080         * @param pattern sequence, either inserted or deleted
081         */
082        public static int pushLeft(final int position, final String pattern, final String reference) {
083            int lower = position;
084
085            String base = reference.substring(lower, lower + 1);
086            String rotated = rotateLeft(pattern);
087
088            if ((base + pattern).equals(rotated + base)) {
089                lower = pushLeft(position - 1, rotated, reference);
090            }
091
092            return lower;
093        }
094
095        public static int pushRight(final int position, final String pattern, final String reference) {
096            int upper = position;
097
098            String base = reference.substring(upper + 1, upper + 2);
099            String rotated = rotateRight(pattern);
100
101            if ((pattern + base).equals(base + rotated)) {
102                upper = pushRight(position + 1, rotated, reference);
103            }
104
105            return upper;
106        }
107    }
108  
109    public String getContig() {
110        return contig;
111    }
112
113    public Locus intersection(final Locus right) {
114        if(!this.contig.equals(right.contig) ) {
115            return new Locus("", 0, 0);
116        }
117    
118        if (this.getEnd() == right.getStart() || this.getStart() == right.getEnd()) {
119            return new Locus(contig, 0, 0);
120        }
121    
122        if (this.overlaps(right)) {
123            return new Locus(contig, (RangeLocation) super.intersection(right));
124        }
125
126        return new Locus(contig, 0, 0);
127    }
128
129    public Locus union(final Locus right) {
130        if (!this.contig.equals(right.contig)) {
131            return new Locus(contig, 0, 0);
132        }
133        else {
134            if (this.overlaps(right)) {
135                return new Locus(contig, (RangeLocation) super.union(right));
136            }
137            return new Locus(contig, 0, 0);
138        }
139    }
140
141    public final int length() {
142        return this.getEnd() - this.getStart();
143    }
144
145    public boolean isEmpty() {
146        return this.getStart() == 0 && this.getEnd() == 0;
147    }
148
149    public boolean overlaps(final Locus right) {
150        if (!this.contig.equals(right.contig)) {
151            return false;
152        }
153    
154        if (this.getEnd() == right.getStart() || this.getStart() == right.getEnd()) {
155            return false;
156        }
157
158        return super.overlaps(right);
159    }
160
161    @Override
162    public boolean equals(final Object right) {
163        if (!super.equals(right)) {
164            return false;
165        }
166
167        Locus locus = (Locus) right;
168        return this.contig.equals(locus.contig);
169    }
170    /**
171     *  
172     */
173    @Override
174    public int compareTo(Locus right) {
175      if(this.contig.equals(right.contig)) {
176        return this.getMin() - right.getMin();
177      }
178      return this.contig.compareTo(right.contig);
179    }
180    
181    @Override
182    public int hashCode() {
183      return Objects.hashCode(this.getStart(), this.getEnd(), contig);
184    }
185  
186    @Override
187    public String toString() {
188        return contig + super.toString();
189    }
190}
