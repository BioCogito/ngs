039/**
040 * Allele class for extending assembly-specific genomic coordinates (Locus objects) with sequence data.
041 */
042public final class Allele extends Locus {
043    /**
044     * Name of the allele.
045     */
046    String name;
047    /**
048     * Enumeration of possible lesion types.
049     */
050    public enum Lesion {
051        /**
052         * Unknown or undetermined.
053         */
054        UNKNOWN,
055        /**
056         * Matches the specified reference sequence at the corresponding genomic location.
057         */
058        MATCH,
059        /**
060         * A single basepair mismatch with respect to the reference sequence.
061         */
062        SUBSTITUTION,
063        /**
064         * One or more gaps in the sequence with respect to the reference.
065         */
066        INSERTION,
067        /**
068         * One or more gaps in the reference with respect to the sequence.
069         */
070        DELETION
071    }
072    /**
073     * Sequence of acceptable characters (alphabet) at the corresponding genomic location.
074     */
075    public final SymbolList sequence;
076    /**
077     * Lesion represented by the sequence with respect to a specified reference sequence. The reference sequence is assigned with proper Allele object instantiation using a Builder.
078     */
079    public final Lesion lesion;

081     * Construct a new allele.

089    Allele(final String name, final String contig, final int start, final int end, final SymbolList sequence, final Lesion lesion) {
090        super(contig, start, end);
091        this.name = name;
092        this.sequence = sequence;

207            this.reference = "";

212            return this;

214        /**
215         * 
216         * @return new Allele parameterized using with methods
217         * @throws AlleleException if Allele cannot be built because the assigned sequence contains illegal symbols (eg non-nucleic acid)
218         */
219        public Allele build() throws AlleleException {
220
221            if (sequence == null || sequence == SymbolList.EMPTY_LIST) {
222                String gaps = Joiner.on("").join(Collections.nCopies(end - start, "-"));
223
224                try {
225                    this.sequence = DNATools.createDNA(gaps);
226                }
227                catch (IllegalSymbolException ex) {
228                    throw new AlleleException("cannot build allele: " + ex.getMessage());
229                }
230            }
231
232            if (reference != null && !reference.isEmpty()) {
233                if (lesion == Lesion.INSERTION || lesion == Lesion.DELETION) {
234                    if (end == start) {
235                        start = Locus.Util.pushLeft(start, sequence.seqString(), reference);
236                        end = Locus.Util.pushRight(end, sequence.seqString(), reference);
237                    }
238                }
239            }
240
241            return new Allele(name, contig, start, end, sequence, lesion);
242        }
243    }
244    /**
245     * Define allele equivalence as the same genomic location (contig, start, and end) as well as the same lesion and sequence.
246     * @param right allele
247     * @return true if this is equal to right, false otherwise
248     */
249    @Override
250    public boolean equals(final Object right) {
251        if (!super.equals(right)) {
252            return false;
253        }
254
255        Allele allele = (Allele) right;
256        return allele.lesion == this.lesion && allele.sequence.equals(this.sequence);
257    }

259     * 
260     * @return allele hash code
261     */

263    public int hashCode() {
264      return super.hashCode() + Objects.hashCode(name, lesion) + sequence.hashCode();
265    }
266    /**
267     * A method to simulate double crossover between Allele objects whereby the sequence from one allele is joined to the other within a specific region of overlap.
268     * @param right allele
269     * @return a new crossed-over allele. This implementation treats the original alleles as immutable thus favoring programmatic convenience over genetic reality.
270     * @throws IllegalSymbolException 
271     * @throws IndexOutOfBoundsException
272     * @throws IllegalAlphabetException

274    public Allele doubleCrossover(final Allele right) throws IllegalSymbolException, IndexOutOfBoundsException, IllegalAlphabetException {

284            int target = homologue.getStart() - right.getStart() + 1;
285            int from = homologue.getStart() - this.getStart() + 1;

311    }
312    /**
313     * A method to simulate merging of two alleles strictly, meaning the sequence in the overlapping regions must be equal.
314     * @param right allele
315     * @param minimumOverlap for the merge to occur
316     * @return new merged allele.
317     * @throws IllegalSymbolException
318     * @throws IndexOutOfBoundsException
319     * @throws IllegalAlphabetException
320     * @throws AlleleException 
321     */
322    public Allele merge(final Allele right, final long minimumOverlap) throws IllegalSymbolException, IndexOutOfBoundsException, IllegalAlphabetException, AlleleException {
323      
324        Allele.Builder builder = Allele.builder();
325        Locus overlap = intersection(right);
326
327        // System.out.println("overlap = " + overlap);
328        // System.out.println("overlap.length() " + overlap.length() + " < " + startimumOverlap + "??");
329
330        if (overlap.length() < minimumOverlap) {
331            return builder.reset().build();
332        }
333
334        Allele bit = builder  
335            .withContig(overlap.getContig())
336            .withStart(overlap.getStart())
337            .withEnd(overlap.getEnd())
338            .withSequence(SymbolList.EMPTY_LIST)
339            .withLesion(Lesion.UNKNOWN)
340            .build();
341
342        // System.out.println("bit = " + bit + " " + bit.sequence.seqString());
343
344        Allele a = bit.doubleCrossover(right);
345
346        // System.out.println("a = " + a + " " + a.sequence.seqString());
347
348        Allele b = bit.doubleCrossover(this);
349
350        // System.out.println("b = " + b + " " + b.sequence.seqString());   
351
352        if (a.sequence.seqString().equals(b.sequence.seqString())) {
353            Locus union = union(right);
354            return builder
355                .withName(right.getName())
356                .withContig(union.getContig())
357                .withStart(union.getStart())
358                .withEnd(union.getEnd())
359                .withSequence(SymbolList.EMPTY_LIST)
360                .withLesion(Lesion.UNKNOWN)
361                .build()
362
363                .doubleCrossover(right)
364                .doubleCrossover(this);
365        }
366        return builder.reset().build(); 
367    }
368    /**
369     * A method to simulate hard clipping (removal) of leftmost sequence, for example synthesized sequence representing molecular barcodes or target capture probes (primers)
370     * @param pattern sequence (will be strictly matched, no regular expressions)
371     * @return new Allele with clipped sequence
372     * @throws IllegalAlphabetException
373     * @throws AlleleException
374     * @throws IllegalSymbolException 
375     */
376    public Allele leftHardClip(final String pattern) throws IllegalAlphabetException, AlleleException, IllegalSymbolException {
377        int start = this.getStart();
378
379        SymbolList copy = DNATools.createDNA(sequence.seqString());
380        while (copy.seqString().startsWith(pattern)) {
381            copy.edit(new Edit(1, pattern.length(), SymbolList.EMPTY_LIST));
382            start += pattern.length();
383        }
384
385        return builder()
386            .withContig(this.getContig())
387            .withStart(start)
388            .withEnd(this.getEnd())
389            .withSequence(copy)
390            .withLesion(this.lesion)
391            .build();
392    }
393    /**
394     * A method to simulate hard clipping (removal) of rightmost sequence, for example synthesized sequence representing molecular barcodes or target capture probes (primers)
395     * @param pattern sequence (will be strictly matched, no regular expressions)
396     * @return new Allele with clipped sequence
397     * @throws IllegalAlphabetException
398     * @throws AlleleException
399     * @throws IllegalSymbolException 
400     */
401    public Allele rightHardClip(final String pattern) throws IllegalSymbolException, IndexOutOfBoundsException, IllegalAlphabetException, AlleleException {
402        int end = this.getEnd();
403
404        // System.out.println("end before = " + end);
405
406        SymbolList copy = DNATools.createDNA(sequence.seqString());
407        while (copy.seqString().endsWith(pattern)) {
408            copy.edit(new Edit(copy.length() - pattern.length() + 1, pattern.length(), SymbolList.EMPTY_LIST));
409
410            end -= pattern.length();
411
412            // System.out.println("end during " + end);
413            // System.out.println("editing copy " + copy.seqString());
414        }
415
416        // System.out.println("end after = " + end);
417        // System.out.println("edited copy = " + copy.seqString());
418
419        return builder()
420            .withContig(this.getContig())
421            .withStart(this.getStart())
422            .withEnd(end)
423            .withSequence(copy)
424            .withLesion(this.lesion)
425            .build();
426    }
