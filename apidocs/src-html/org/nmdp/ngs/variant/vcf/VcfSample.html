023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.util.ArrayList;
028import java.util.List;
029
030import javax.annotation.concurrent.Immutable;
031
032/**
033 * VCF sample.
034 */
035@Immutable
036public final class VcfSample {
037    /** Id. */
038    private final String id;
039
040    /** Array of VCF genomes. */
041    private final VcfGenome[] genomes;
042
043
044    /**
045     * Create a new VCF sample.
046     *
047     * @param id id, must not be null
048     * @param genomes VCF genomes, must not be null
049     */
050    VcfSample(final String id, final VcfGenome... genomes) {
051        checkNotNull(id);
052        checkNotNull(genomes);
053        this.id = id;
054        this.genomes = genomes;
055    }
056
057
058    /**
059     * Return the id for this VCF sample.
060     *
061     * @return the id for this VCF sample.
062     */
063    public String getId() {
064        return id;
065    }
066
067    /**
068     * Return the VCF genomes for this VCF sample.
069     *
070     * @return the VCF genomes for this VCF sample
071     */
072    public VcfGenome[] getGenomes() {
073        return genomes;
074    }
075
076
077    /**
078     * Create and return a new VCF sample builder.
079     *
080     * @return a new VCF sample builder
081     */
082    public static Builder builder() {
083        return new Builder();
084    }
085
086    /**
087     * VCF sample builder.
088     */
089    public static final class Builder {
090        /** Id. */
091        private String id;
092
093        /** List of VCF genomes. */
094        private final List<VcfGenome> genomes = new ArrayList<VcfGenome>();
095
096
097        /**
098         * Private no-arg constructor.
099         */
100        private Builder() {
101            // empty
102        }
103
104
105        /**
106         * Return this VCF sample builder configured with the specified id.
107         *
108         * @param id id
109         * @return this VCF sample builder configured with the specified id
110         */
111        public Builder withId(final String id) {
112            this.id = id;
113            return this;
114        }
115
116        /**
117         * Return this VCF sample builder configured with the specified VCF genome.
118         *
119         * @param genome VCF genome, must not be null
120         * @return this VCF sample builder configured with the specified VCF genome
121         */
122        public Builder withGenome(final VcfGenome genome) {
123            checkNotNull(genome);
124            genomes.add(genome);
125            return this;
126        }
127
128        /**
129         * Return this VCF sample builder configured with the specified VCF genomes.
130         *
131         * @param genomes one or more VCF genomes, must not contain null
132         * @return this VCF sample builder configured with the specified VCF genomes
133         */
134        public Builder withGenomes(final VcfGenome... genomes) {
135            checkNotNull(genomes);
136            for (VcfGenome genome : genomes) {
137                withGenome(genome);
138            }
139            return this;
140        }
141
142        /**
143         * Return this VCF sample builder configured with the specified VCF genomes.
144         *
145         * @param genomes one or more VCF genomes, must not be null and must not contain null values
146         * @return this VCF sample builder configured with the specified VCF genomes
147         */
148        public Builder withGenomes(final List<VcfGenome> genomes) {
149            checkNotNull(genomes);
150            for (VcfGenome genome : genomes) {
151                withGenome(genome);
152            }
153            return this;
154        }
155
156        /**
157         * Reset this VCF sample builder.
158         *
159         * @return this VCF sample builder
160         */
161        public Builder reset() {
162            id = null;
163            genomes.clear();
164            return this;
165        }
166
167        /**
168         * Create and return a new VCF sample populated from the configuration of this VCF sample builder.
169         *
170         * @return a new VCF sample populated from the configuration of this VCF sample builder
171         */
172        public VcfSample build() {
173            return new VcfSample(id, genomes.toArray(new VcfGenome[genomes.size()]));
174        }
175    }
176}
