023package org.nmdp.ngs.variant.vcf;
024
025import javax.annotation.concurrent.Immutable;
026
027import com.google.common.collect.ArrayListMultimap;
028import com.google.common.collect.ImmutableListMultimap;
029import com.google.common.collect.ListMultimap;
030
031/**
032 * VCF genotype.
033 */
034@Immutable
035public final class VcfGenotype {
036    /** Genotype fields. */
037    private final ListMultimap<String, String> fields;
038
039
040    /**
041     * Create a new VCF genotype with the specified genotype fields.
042     *
043     * @param fields genotype fields, must not be null
044     */
045    VcfGenotype(final ListMultimap<String, String> fields) {
046        this.fields = ImmutableListMultimap.copyOf(fields);
047    }
048
049
050    /**
051     * Return the value of the GT genotype field for this VCF genotype or <code>null</code> if no such field exists.
052     *
053     * @return the value of the GT genotype field for this VCF genotype or <code>null</code> if no such field exists
054     */
055    public String getGt() {
056        return fields.get("GT").isEmpty() ? null : fields.get("GT").get(0);
057    }
058
059    /**
060     * Return the genotype fields for this VCF genotype.
061     *
062     * @return the genotype fields for this VCF genotype
063     */
064    public ListMultimap<String, String> getFields() {
065        return fields;
066    }
067
068    /**
069     * Create and return a new VCF genotype builder.
070     *
071     * @return a new VCF genotype builder
072     */
073    public static Builder builder() {
074        return new Builder();
075    }
076
077    /**
078     * VCF genotype builder.
079     */
080    public static final class Builder {
081        /** Genotype fields. */
082        private ListMultimap<String, String> fields = ArrayListMultimap.create();
083
084
085        /**
086         * Private no-arg constructor.
087         */
088        private Builder() {
089            // empty
090        }
091
092
093        /**
094         * Return this VCF genotype builder configured with the specified GT genotype field.
095         *
096         * @param gt GT genotype field
097         * @return this VCF record builder configured with the specified GT genotype field
098         */
099        public Builder withGt(final String gt) {
100            if (gt == null) {
101                fields.removeAll("GT");
102            }
103            else {
104                fields.put("GT", gt);
105            }
106            return this;
107        }
108
109        /**
110         * Return this VCF genotype builder configured with the specified genotype field.
111         *
112         * @param id genotype field id
113         * @param values genotype field values
114         * @return this VCF record builder configured with the specified genotype field
115         */
116        public Builder withField(final String id, final String... values) {
117            if (values != null) {
118                for (String value : values) {
119                    fields.put(id, value);
120                }
121            }
122            return this;
123        }
124
125        /**
126         * Return this VCF genotype builder configured with the specified genotype fields.
127         *
128         * @param fields genotype fields, must not be null
129         * @return this VCF record builder configured with the specified genotype fields
130         */
131        public Builder withFields(final ListMultimap<String, String> fields) {
132            this.fields.putAll(fields);
133            return this;
134        }
135
136        /**
137         * Reset this VCF genotype builder.
138         *
139         * @return this VCF genotype builder
140         */
141        public Builder reset() {
142            fields.clear();
143            return this;
144        }
145
146        /**
147         * Create and return a new VCF genotype populated from the configuration of this VCF genotype builder.
148         *
149         * @return a new VCF genotype populated from the configuration of this VCF genotype builder
150         */
151        public VcfGenotype build() {
152            // check GT cardinality constraint
153            if (fields.containsKey("GT")) {
154                String gt = fields.get("GT").get(0);
155                fields.removeAll("GT");
156                if (gt != null) {
157                    fields.put("GT", gt);
158                }
159            }
160            return new VcfGenotype(fields);
161        }
162    }
163}
