023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.io.IOException;
028
029import java.util.ArrayList;
030import java.util.HashMap;
031import java.util.List;
032import java.util.Map;
033
034import com.google.common.collect.ArrayListMultimap;
035import com.google.common.collect.ListMultimap;
036
037/**
038 * VCF sample parser.
039 */
040public final class VcfSampleParser {
041
042    /**
043     * Private no-arg constructor.
044     */
045    private VcfSampleParser() {
046        // empty
047    }
048
049
050    /**
051     * Read zero or more VCF samples from the specified readable.
052     *
053     * @param readable readable to read from, must not be null
054     * @return zero or more VCF samples read from the specified readable
055     * @throws IOException if an I/O error occurs
056     */
057    public static Iterable<VcfSample> samples(final Readable readable) throws IOException {
058        checkNotNull(readable);
059        ParseListener parseListener = new ParseListener();
060        VcfParser.parse(readable, parseListener);
061        return parseListener.getSamples().values();
062    }
063
064    /**
065     * Parse listener.
066     */
067    static final class ParseListener extends VcfParseAdapter {
068        /** VCF samples keyed by name. */
069        private Map<String, VcfSample> samples = new HashMap<String, VcfSample>();
070
071        /*
072          ##SAMPLE=<ID=S_ID,Genomes=G1_ID;G2_ID; ...;GK_ID,Mixture=N1;N2; ...;NK,Description=S1;S2; ...; SK >
073
074          ##SAMPLE=<ID=Blood,Genomes=Germline,Mixture=1.,Description="Patient germline genome">
075          ##SAMPLE=<ID=TissueSample,Genomes=Germline;Tumor,Mixture=.3,.7,Description="Patient germline genome;Patient tumor genome">
076        */
077
078        @Override
079        public void meta(final String meta) throws IOException {
080            if (meta.startsWith("##SAMPLE=")) {
081                ListMultimap<String, String> values = ArrayListMultimap.create();
082                String[] tokens = meta.substring(10).split(",");
083                for (String token : tokens) {
084                    String[] metaTokens = token.split("=");
085                    String key = metaTokens[0];
086                    String[] valueTokens = metaTokens[1].split(";");
087                    for (String valueToken : valueTokens) {
088                        values.put(key, valueToken.replace("\"", "").replace(">", ""));
089                    }
090                }
091
092                String id = values.get("ID").get(0);
093                List<String> genomeIds = values.get("Genomes");
094                List<String> mixtures = values.get("Mixture");
095                List<String> descriptions = values.get("Description");
096
097                List<VcfGenome> genomes = new ArrayList<VcfGenome>(genomeIds.size());
098                for (int i = 0, size = genomeIds.size(); i < size; i++) {
099                    genomes.add(new VcfGenome(genomeIds.get(i), Double.parseDouble(mixtures.get(i)), descriptions.get(i)));
100                }
101                samples.put(id, new VcfSample(id, genomes.toArray(new VcfGenome[genomes.size()])));
102            }
103        }
104
105        @Override
106        public void samples(final String... samples) throws IOException {
107            for (String sample : samples) {
108                // add if missing in meta lines
109                if (!this.samples.containsKey(sample)) {
110                    this.samples.put(sample, new VcfSample(sample, new VcfGenome[0]));
111                }
112            }
113        }
114
115        @Override
116        public boolean complete() throws IOException {
117            return false;
118        }
119
120        /**
121         * Return the VCF samples keyed by name.
122         *
123         * @return the VCF samples keyed by name
124         */
125        Map<String, VcfSample> getSamples() {
126            return samples;
127        }
128    }
129}
