023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.io.IOException;
028
029import java.util.ArrayList;
030import java.util.List;
031
032/**
033 * VCF header parser.
034 */
035public final class VcfHeaderParser {
036
037    /**
038     * Private no-arg constructor.
039     */
040    private VcfHeaderParser() {
041        // empty
042    }
043
044
045    /**
046     * Read the VCF header from the specified readable.
047     *
048     * @param readable readable to read from, must not be null
049     * @return the VCF header read from the specified readable
050     * @throws IOException if an I/O error occurs
051     */
052    public static VcfHeader header(final Readable readable) throws IOException {
053        checkNotNull(readable);
054        ParseListener parseListener = new ParseListener();
055        VcfParser.parse(readable, parseListener);
056        return parseListener.getHeader();
057    }
058
059    /**
060     * Parse listener.
061     */
062    static final class ParseListener extends VcfParseAdapter {
063        /** File format. */
064        private String fileFormat;
065
066        /** List of meta-information header lines. */
067        private List<String> meta = new ArrayList<String>();
068
069
070        @Override
071        public void meta(final String meta) throws IOException {
072            this.meta.add(meta.trim());
073            if (meta.startsWith("##fileformat=")) {
074                fileFormat = meta.substring(13).trim();
075            }
076        }
077
078        @Override
079        public boolean complete() throws IOException {
080            return false;
081        }
082
083        /**
084         * Return the VCF header.
085         *
086         * @return the VCF header
087         * @throws IOException if an I/O error occurs
088         */
089        VcfHeader getHeader() throws IOException {
090            if (fileFormat == null) {
091                throw new IOException("could not read header, required fileformat meta-information header line not found");
092            }
093            return new VcfHeader(fileFormat, meta);
094        }
095    };
096}
