023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import static com.google.common.collect.Maps.newHashMap;
028
029import java.io.IOException;
030
031import java.util.Collection;
032import java.util.Map;
033
034import com.google.common.io.CharStreams;
035import com.google.common.io.LineProcessor;
036
037/**
038 * Low-level VCF parser.
039 */
040public final class VcfParser {
041
042    /**
043     * Private no-arg constructor.
044     */
045    private VcfParser() {
046        // empty
047    }
048
049
050    /**
051     * Parse the specified readable.
052     *
053     * @param readable readable, must not be null
054     * @param listener low-level event based parser callback, must not be null
055     * @throws IOException if an I/O error occurs
056     */
057    public static void parse(final Readable readable, final VcfParseListener listener) throws IOException {
058        checkNotNull(readable);
059        VcfLineProcessor lineProcessor = new VcfLineProcessor(listener);
060        CharStreams.readLines(readable, lineProcessor);
061    }
062
063    /**
064     * VCF line processor.
065     */
066    private static final class VcfLineProcessor implements LineProcessor<Object> {
067        /** Line number. */
068        private long lineNumber = 0;
069
070        /** VCF parse listener. */
071        private final VcfParseListener listener;
072
073        /** Map of sample names keyed by column. */
074        private final Map<Integer, String> samples = newHashMap();
075
076
077        /**
078         * Create a new VCF line processor.
079         *
080         * @param listener VCF parse listener
081         */
082        private VcfLineProcessor(final VcfParseListener listener) {
083            checkNotNull(listener);
084            this.listener = listener;
085        }
086
087
088        @Override
089        public Object getResult() {
090            return null;
091        }
092
093        @Override
094        public boolean processLine(final String line) throws IOException
095        {
096            lineNumber++;
097            // consider using guava Splitter
098            String[] tokens = line.split("\t");
099
100            if (tokens[0].startsWith("##")) {
101                // meta-information lines
102                listener.meta(line);
103            }
104            else if (tokens[0].startsWith("#CHROM")) {
105                // header line
106                if (tokens.length > 8) {
107                    for (int column = 9, columns = tokens.length; column < columns; column++) {
108                        samples.put(column, tokens[column]);
109                    }
110                }
111                Collection<String> values = samples.values();
112                listener.samples(values.toArray(new String[values.size()]));
113            }
114            else {
115                // data lines
116                listener.lineNumber(lineNumber);
117                if (tokens.length < 8) {
118                    throw new IOException("invalid data line at line number " + lineNumber + ", expected 8 tokens, found " + tokens.length);
119                }
120
121                listener.chrom(tokens[0]);
122
123                try {
124                    listener.pos(Long.parseLong(tokens[1]));
125                }
126                catch (NumberFormatException e) {
127                    throw new IOException("invalid pos at line number " + lineNumber, e);
128                }
129
130                String[] idTokens = tokens[2].split(";");
131                listener.id(isMissingValue(idTokens) ? new String[0] : idTokens);
132
133                listener.ref(tokens[3]);
134
135                // todo: check for symbolic alleles
136                String[] altTokens = tokens[4].split(",");
137                listener.alt(isMissingValue(altTokens) ? new String[0] : altTokens);
138
139                try {
140                    listener.qual(isMissingValue(tokens[5]) ? Double.NaN : Double.parseDouble(tokens[5]));
141                }
142                catch (NumberFormatException e) {
143                    throw new IOException("invalid qual at line number " + lineNumber, e);
144                }
145
146                String[] filterTokens = tokens[6].split(";");
147                listener.filter(isMissingValue(filterTokens) ? new String[0] : filterTokens);
148
149                String[] infoTokens = tokens[7].split(";");
150                if (!isMissingValue(infoTokens)) {
151                    for (String infoToken : infoTokens) {
152                        String[] entryTokens = infoToken.split("=");
153                        if (entryTokens.length == 1) {
154                            listener.info(entryTokens[0]);
155                        }
156                        else if (entryTokens.length == 2) {
157                            String infoId = entryTokens[0];
158                            String value = entryTokens[1];
159                            if (isMissingValue(value)) {
160                                listener.info(infoId);
161                            }
162                            else {
163                                listener.info(infoId, value.split(","));
164                            }
165                        }
166                    }
167                }
168
169                if (tokens.length > 8) {
170                    String[] formatTokens = tokens[8].split(":");
171                    listener.format(formatTokens);
172
173                    for (int column = 9, columns = tokens.length; column < columns; column++) {
174                        String[] genotypeTokens = tokens[column].split(":");
175
176                        if (genotypeTokens.length > formatTokens.length) {
177                            throw new IOException("invalid genotype fields at line number " + lineNumber + ", too many genotype fields");
178                        }
179                        if (formatTokens.length > 0 && "GT".equals(formatTokens[0])) {
180                            if (genotypeTokens.length == 0) {
181                                throw new IOException("invalid genotype fields at line number " + lineNumber + ", missing genotype (GT) field");
182                            }
183                            String gt = genotypeTokens[0];
184                            listener.genotype(samples.get(column), "GT", isMissingGenotypeValue(gt) ? null : gt);
185                        }
186                        for (int i = 1, size = Math.min(formatTokens.length, genotypeTokens.length); i < size; i++) {
187                            if (!isMissingValue(genotypeTokens[i])) {
188                                listener.genotype(samples.get(column), formatTokens[i], genotypeTokens[i].split(","));
189                            }
190                        }
191                    }
192                }
193
194                if (!listener.complete()) {
195                    return false;
196                }
197            }
198            return true;
199        }
200    }
201
202    /**
203     * Return true if the specified value is the missing value (<code>"."</code>).
204     *
205     * @param value value
206     * @return true if the specified value is the missing value (<code>"."</code>)
207     */
208    static boolean isMissingValue(final String value) {
209        return ".".equals(value);
210    }
211
212    /**
213     * Return true if the specified value is the missing value (<code>"."</code>) or a genotype (GT) of only
214     * missing values (e.g. <code>"./."</code> for diploid).
215     *
216     * @param value value
217     * @return true if the specified value is the missing value (<code>"."</code>) or a genotype (GT) of only
218     *   missing values (e.g. <code>"./."</code> for diploid)
219     */
220    static boolean isMissingGenotypeValue(final String value) {
221        return isMissingValue(value) || "./.".equals(value);
222    }
223
224    /**
225     * Return true if the specified array of values is the missing value (<code>"."</code>).
226     *
227     * @param values array of values
228     * @return true if the specified array of values is the missing value (<code>"."</code>)
229     */
230    static boolean isMissingValue(final String[] values) {
231        return (values.length == 1 && ".".equals(values[0]));
232    }
233}
