023package org.nmdp.ngs.hml;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.net.URL;
028
029import java.io.BufferedReader;
030import java.io.File;
031import java.io.FileReader;
032import java.io.InputStream;
033import java.io.InputStreamReader;
034import java.io.IOException;
035
036import java.util.ArrayList;
037import java.util.List;
038
039import com.google.common.base.Charsets;
040
041import com.google.common.io.Resources;
042
043import org.biojava.bio.BioException;
044
045import org.biojava.bio.seq.DNATools;
046import org.biojava.bio.seq.SequenceIterator;
047
048import org.biojava.bio.seq.io.SeqIOTools;
049
050import org.biojava.bio.symbol.IllegalSymbolException;
051import org.biojava.bio.symbol.SymbolList;
052
053import org.nmdp.ngs.hml.jaxb.Hml;
054import org.nmdp.ngs.hml.jaxb.Hmlid;
055import org.nmdp.ngs.hml.jaxb.Sequence;
056
057/**
058 * Static utility methods for creating HML model classes.
059 */
060public final class HmlUtils {
061
062    /**
063     * Private no-arg constructor.
064     */
065    private HmlUtils() {
066        // empty
067    }
068
069
070    /**
071     * Create and return a new HML Sequence from the specified sequence.
072     *
073     * @param sequence sequence, must not be null, and alphabet must be DNA
074     * @return a new HML Sequence created from the specified sequence.
075     */
076    public static Sequence createSequence(final org.biojava.bio.seq.Sequence sequence) {
077        checkNotNull(sequence);
078
079        Sequence s = new Sequence();
080        if (!DNATools.getDNA().equals(sequence.getAlphabet())) {
081            throw new IllegalArgumentException("alphabet must be DNA");
082        }
083        s.setValue(sequence.seqString());
084        return s;
085    }
086
087    /**
088     * Create and return zero or more DNA HML Sequences read from the specified reader in FASTA format.
089     *
090     * @param reader reader to read from, must not be null
091     * @return zero or more DNA HML Sequences read from the specified reader in FASTA format
092     * @throws IOException if an I/O error occurs
093     */
094    public static Iterable<Sequence> createSequences(final BufferedReader reader) throws IOException {
095        checkNotNull(reader);
096        List<Sequence> sequences = new ArrayList<Sequence>();
097        for (SequenceIterator it = SeqIOTools.readFastaDNA(reader); it.hasNext(); ) {
098            try {
099                sequences.add(createSequence(it.nextSequence()));
100            }
101            catch (BioException e) {
102                throw new IOException("could not read DNA sequences", e);
103            }
104        }
105        return sequences;
106    }
107
108
109    /**
110     * Create and return zero or more DNA HML Sequences read from the specified file in FASTA format.
111     *
112     * @param file file to read from, must not be null
113     * @return zero or more DNA HML Sequences read from the specified file in FASTA format
114     * @throws IOException if an I/O error occurs
115     */
116    public static Iterable<Sequence> createSequences(final File file) throws IOException {
117        checkNotNull(file);
118        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
119            return createSequences(reader);
120        }
121    }
122
123    /**
124     * Create and return zero or more DNA HML Sequences read from the specified URL in FASTA format.
125     *
126     * @param url URL to read from, must not be null
127     * @return zero or more DNA HML Sequences read from the specified URL in FASTA format
128     * @throws IOException if an I/O error occurs
129     */
130    public static Iterable<Sequence> createSequences(final URL url) throws IOException {
131        checkNotNull(url);
132        try (BufferedReader reader = Resources.asCharSource(url, Charsets.UTF_8).openBufferedStream()) {
133            return createSequences(reader);
134        }
135    }
136
137    /**
138     * Create and return zero or more DNA HML Sequences read from the specified input stream in FASTA format.
139     *
140     * @param inputStream input stream to read from, must not be null
141     * @return zero or more DNA HML Sequences read from the specified input stream in FASTA format
142     * @throws IOException if an I/O error occurs
143     */
144    public static Iterable<Sequence> createSequences(final InputStream inputStream) throws IOException {
145        checkNotNull(inputStream);
146        try (BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream))) {
147            return createSequences(reader);
148        }
149    }
150
151    /**
152     * Return the Hmlid element from the specified HML document, if any.
153     *
154     * @param hml HML document, must not be null
155     * @return the Hmlid element from the specified HML document, if any
156     */
157    public static Hmlid getHmlid(final Hml hml) {
158        checkNotNull(hml);
159        return hml.getHmlid();
160    }
161
162    /**
163     * Convert the specified HML Sequence element into a DNA symbol list.
164     *
165     * @param sequence HML Sequence element, must not be null
166     * @return the specified HML Sequence element converted into a DNA symbol list
167     * @throws IllegalSymbolException if an illegal symbol is found
168     */
169    public static SymbolList toDnaSymbolList(final Sequence sequence) throws IllegalSymbolException {
170        checkNotNull(sequence);
171        return DNATools.createDNA(sequence.getValue().replaceAll("\\s+", ""));
172    }
173}
