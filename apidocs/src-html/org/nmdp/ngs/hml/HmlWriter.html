023package org.nmdp.ngs.hml;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.net.URL;
028
029import java.io.BufferedWriter;
030import java.io.File;
031import java.io.FileWriter;
032import java.io.IOException;
033import java.io.OutputStream;
034import java.io.OutputStreamWriter;
035import java.io.Writer;
036
037import javax.xml.XMLConstants;
038
039import javax.xml.bind.JAXBContext;
040import javax.xml.bind.JAXBException;
041import javax.xml.bind.Marshaller;
042
043import javax.xml.validation.Schema;
044import javax.xml.validation.SchemaFactory;
045
046import org.nmdp.ngs.hml.jaxb.Hml;
047
048import org.xml.sax.SAXException;
049
050/**
051 * Writer for HML xml.
052 */
053public final class HmlWriter {
054
055    /**
056     * Private no-arg constructor.
057     */
058    private HmlWriter() {
059        // empty
060    }
061
062
063    /**
064     * Write the specified HML to the specified writer.
065     *
066     * @param data HML to write, must not be null
067     * @param writer writer to write to, must not be null
068     * @throws IOException if an I/O error occurs
069     */
070    public static void write(final Hml data, final Writer writer) throws IOException {
071        checkNotNull(data);
072        checkNotNull(writer);
073
074        try {
075            JAXBContext context = JAXBContext.newInstance(Hml.class);
076            Marshaller marshaller = context.createMarshaller();
077            SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
078            URL schemaURL = HmlReader.class.getResource("/org/nmdp/ngs/hml/xsd/hml-1.0.1.xsd");
079            Schema schema = schemaFactory.newSchema(schemaURL);
080            marshaller.setSchema(schema);
081            marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
082            marshaller.marshal(data, writer);
083        }
084        catch (JAXBException | SAXException e) {
085            e.printStackTrace();
086            throw new IOException("could not marshal HML", e);
087        }
088    }
089
090    /**
091     * Write the specified HML to the specified file.
092     *
093     * @param data HML to write, must not be null
094     * @param file file to write to, must not be null
095     * @throws IOException if an I/O error occurs
096     */
097    public static void write(final Hml data, final File file) throws IOException {
098        checkNotNull(data);
099        checkNotNull(file);
100        try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) {
101            write(data, writer);
102        }
103    }
104
105    /**
106     * Write the specified HML to the specified output stream.
107     *
108     * @param data HML to write, must not be null
109     * @param outputStream output stream to write to, must not be null
110     * @throws IOException if an I/O error occurs
111     */
112    public static void write(final Hml data, final OutputStream outputStream) throws IOException {
113        checkNotNull(data);
114        checkNotNull(outputStream);
115        try (BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(outputStream))) {
116            write(data, writer);
117        }
118    }
119}
