023package org.nmdp.ngs.fca;
024
025import java.util.List;
026import java.util.ArrayList;
027
028import com.tinkerpop.blueprints.Direction;
029import com.tinkerpop.blueprints.Edge;
030import com.tinkerpop.blueprints.Graph;
031import com.tinkerpop.blueprints.Vertex;
032
033import org.dishevelled.bitset.MutableBitSet;
034
035/**
036 * Concept lattice.
037 *
038 * @param <G> object type
039 * @param <M> attribute type
040 */
041public class ConceptLattice<G, M> extends AbstractLattice<Concept> {
042    protected List<G> objects;
043    protected List<M> attributes;
044
045    public ConceptLattice(final Graph lattice, final List<M> attributes) {
046        super(lattice);
047        objects = new ArrayList<>();
048        this.attributes = attributes;
049        MutableBitSet ones = new MutableBitSet(this.attributes.size());
050        ones.set(0, this.attributes.size());
051        top = lattice.addVertex(null);
052        top.setProperty("label", new Concept(new MutableBitSet(), ones));
053        top.setProperty("color", color);
054        direction = Partial.Order.Direction.FORWARD;
055        size = 1;
056        order = 0;
057        bottom = top;
058    }
059
060    public Concept insert(final G object, final List<M> attributes) {
061        objects.add(object);
062        MutableBitSet intent = Concept.encode(attributes, this.attributes);
063        Concept proposed = new Concept(new MutableBitSet(), intent);
064        Vertex added = super.addIntent(proposed, top);
065
066        List<G> list = new ArrayList<G>();
067        list.add(object);
068        MutableBitSet extent = Concept.encode(list, objects);
069
070        List queue = new ArrayList();
071        added.setProperty("color", ++color);
072        queue.add(added);
073
074        while (!queue.isEmpty()) {
075            Vertex visiting = (Vertex) queue.remove(0);
076            Concept visitingConcept = visiting.getProperty(LABEL);
077            visitingConcept.extent().or(extent);
078
079            for (Edge edge : visiting.getEdges(Direction.BOTH)) {
080                Vertex target = edge.getVertex(Direction.OUT);
081
082                if ((int) target.getProperty("color") != color) {
083                    if (filter(visiting, target)) {
084                        target.setProperty("color", color);
085                        queue.add(target);
086                    }
087                }
088            }
089        }
090        return added.getProperty(LABEL);
091    }
092
093    // todo:  this should be typed
094    public final List getObjects() {
095        return objects;
096    }
097
098    public List<M> getAttributes() {
099        return attributes;
100    }
101
102    @Override
103    public final Concept join(final Concept left, final Concept right) {
104        MutableBitSet bits = (MutableBitSet) new MutableBitSet().or(left.intent()).or(right.intent());
105        Concept query = new Concept(new MutableBitSet(), bits);
106        return supremum(query, top).getProperty(LABEL);
107    }
108
109    public Concept greatestLowerBound(final List query) {
110        // todo:  support this method
111        throw new UnsupportedOperationException("Not supported yet.");
112    }
113
114    @Override
115    public String toString() {
116        StringBuilder sb = new StringBuilder("digraph {\n");
117        for (Vertex vertex : lattice.getVertices()) {
118            for (Edge edge : vertex.getEdges(Direction.BOTH)) {
119                Vertex target = edge.getVertex(Direction.OUT);
120                if (!vertex.getProperty("label").equals(target.getProperty("label"))) {
121                    Concept vertexConcept = vertex.getProperty("label");
122                    Concept targetConcept = target.getProperty("label");
123                    if (!filter(vertex, target)) {
124                        sb.append("  \"" + Concept.decode(vertexConcept.extent(), objects) + Concept.decode(vertexConcept.intent(), attributes) + "\" -> \"" + Concept.decode(targetConcept.extent(), objects) + Concept.decode(targetConcept.intent(), attributes) + "\"[label=\"" + edge.getLabel() + "\"]\n");
125                    }
126                }
127            }
128        }
129        sb.append("}");
130        return sb.toString();
131    }
132}
