025import static com.google.common.base.Preconditions.checkNotNull;
026
027import java.util.ArrayList;
028import java.util.Iterator;
029import java.util.List;
030
031import com.tinkerpop.blueprints.Graph;
032import com.tinkerpop.blueprints.Edge;
033import com.tinkerpop.blueprints.Vertex;

036/**
037 * Abstract lattice.
038 *
039 * @param <T> type
040 */
041public abstract class AbstractLattice<T extends Partial> implements Lattice<T> {
042    protected Graph lattice;
043    protected Vertex bottom;
044    protected Vertex top;
045
046    protected Partial.Order.Direction direction;
047    protected int color;
048    protected int size;
049    protected int order;
050
051    protected static final String LABEL = "label";
052
053    protected AbstractLattice(final Graph lattice) {
054        checkNotNull(lattice);
055        this.lattice = lattice;
056    }
057
058    protected boolean filter(final Vertex source, final Vertex target) {
059        T sourceConcept = source.getProperty(LABEL);
060        T targetConcept = target.getProperty(LABEL);
061        return filter(sourceConcept, targetConcept);
062    }
063
064    private boolean filter(final Vertex source, final T right) {
065        T sourceConcept = source.getProperty(LABEL);
066        return filter(sourceConcept, right);
067    }
068
069    private boolean filter(final T left, final Vertex target) {
070        T targetConcept = target.getProperty(LABEL);
071        return filter(left, targetConcept);
072    }
073
074    private boolean filter(final T right, final T left) {
075        return right.relation(left).greaterOrEqual();
076    }
077
078    /**
079     * Find the supremum or least upper bound.
080     *
081     * @param proposed proposed
082     * @param generator starting point and tracer
083     * @return vertex whose label-concept represents the supremum
084     */
085    // todo:  modifies generator parameter
086    protected Vertex supremum(final T proposed, Vertex generator) {
087        boolean max = true;
088        while (max) {
089            max = false;
090            for (Edge edge : generator.getEdges(Direction.BOTH)) {
091                Vertex target = edge.getVertex(Direction.OUT);
092
093                if (filter(target, generator)) {
094                    continue;
095                }
096                //Concept proposed = new Concept(new MutableBitSet(), proposed.intent);
097
098                if (filter(target, proposed)) {
099                    generator = target;
100                    max = true;
101                    break;
102                }
103            }
104        }
105        return generator;
106    }
107
108    private Vertex add(final T label) {
109        Vertex child = lattice.addVertex(null);
110        child.setProperty("label", label);
111        child.setProperty("color", color);
112        ++size;
113        return child;
114    }
115
116    private Edge addUndirectedEdge(final Vertex source, final Vertex target, final String weight) {
117        T sourceConcept = source.getProperty("label");
118        T targetConcept = target.getProperty("label");
119        Partial.Order.Direction direction = this.direction;
120
121        if (targetConcept.relation(sourceConcept).gte()) {
122            direction = Partial.Order.Direction.REVERSE;
123        }
124
125        lattice.addEdge(null, source, target, weight);
126        Edge edge = lattice.addEdge(null, target, source, weight);
127        ++order;
128        return edge;
129    }
130
131    private void removeUndirectedEdge(final Vertex source, final Vertex target) {
132        for (Edge edge : source.getEdges(Direction.BOTH)) {
133            if (edge.getVertex(Direction.OUT).equals(target)) {
134                lattice.removeEdge(edge);
135                break;
136            }
137
138            if (edge.getVertex(Direction.IN).equals(target)) {
139                lattice.removeEdge(edge);
140                break;
141            }
142        }
143        --order;
144    }
145
146    /**
147     * Find the supremum or least upper bound.
148     *
149     * @param proposed proposed
150     * @param generator starting point and tracer
151     * @return vertex whose label-concept represents the supremum
152     */
153    // todo:  modifies generator parameter
154    protected Vertex addIntent(T proposed, Vertex generator) {
155        generator = supremum(proposed, generator);
156
157        if (filter(generator, proposed) && filter(proposed, generator)) {
158            return generator;
159        }
160        List parents = new ArrayList<>();
161        for (Vertex target : lattice.getVertices()) {
162            if (filter(target, generator)) {
163                continue;
164            }
165
166            Vertex candidate = target;
167            if (!filter(target, proposed) && !filter(proposed, target)) {
168                T targetConcept = target.getProperty(LABEL);
169                // MutableBitSet meet = (MutableBitSet) new MutableBitSet().or(targetConcept.intent()).and(proposed.intent());
170                T intersect = (T) targetConcept.intersect(proposed);
171                candidate = addIntent(intersect, candidate);
172            }
173
174            boolean add = true;
175            List doomed = new ArrayList<>();
176            for (Iterator it = parents.iterator(); it.hasNext();) {
177                Vertex parent = (Vertex) it.next();
178
179                if (filter(parent, candidate)) {
180                    add = false;
181                    break;
182                }
183                else if (filter(candidate, parent)) {
184                    doomed.add(parent);
185                }
186            }
187
188            for (Iterator it = doomed.iterator(); it.hasNext();) {
189                Vertex vertex = (Vertex) it.next();
190                parents.remove(vertex);
191            }
192
193            if (add) {
194                parents.add(candidate);
195            }
196        }
197
198        T generatorConcept = generator.getProperty(LABEL);
199
200        Vertex child = add((T) proposed.union(generatorConcept));
201        addUndirectedEdge(generator, child, "");
202        bottom = filter(bottom, proposed) ? child : bottom;
203
204        for (Iterator it = parents.iterator(); it.hasNext();) {
205            Vertex parent = (Vertex) it.next();
206            if (!parent.equals(generator)) {
207                removeUndirectedEdge(parent, generator);
208                addUndirectedEdge(parent, child, "");
209            }
210        }
211        return child;
212    }
213
214    public int size() {
215        return size;
216    }
217
218    public int order() {
219        return order;
220    }
221
222    @Override
223    public final T bottom() {
224        return bottom.getProperty(LABEL);
225    }
226
227    @Override
228    public final T top() {
229        return top.getProperty(LABEL);
230    }
231
232    @Override
233    public T join(final T left, final T right) {
234        return supremum((T) left.union(right), top).getProperty(LABEL);
235    }
236
237    @Override
238    public T meet(final T left, final T right) {
239        return supremum((T) left.intersect(right), top).getProperty(LABEL);
240    }
241
242    @Override
243    public double measure(final T left, final T right) {
244        //Concept concept = leastUpperBound(right);
245        //Concept meet = left.meet(leastUpperBound(left), concept);
246        // T meet = this.meet(left, right);
247        return (double) join(left, right).measure() / meet(right, top()).measure();
248    }
