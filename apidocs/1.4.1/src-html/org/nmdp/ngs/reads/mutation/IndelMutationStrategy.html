023package org.nmdp.ngs.reads.mutation;
024
025import static com.google.common.base.Preconditions.checkArgument;
026import static com.google.common.base.Preconditions.checkNotNull;
027
028import org.apache.commons.math3.random.RandomGenerator;
029
030import org.biojava.bio.symbol.Symbol;
031import org.biojava.bio.symbol.SymbolList;
032
033import org.nmdp.ngs.reads.MutationStrategy;
034
035/**
036 * Indel mutation strategy.
037 */
038public final class IndelMutationStrategy implements MutationStrategy {
039    /** Random. */
040    private final RandomGenerator random;
041
042    /** Insertion mutation strategy. */
043    private final InsertionMutationStrategy insertion;
044
045    /** Insertion rate. */
046    private final double insertionRate;
047
048    /** Deletion mutation strategy. */
049    private final DeletionMutationStrategy deletion;
050
051
052    /**
053     * Create a new indel mutation strategy with the specified parameters.
054     *
055     * @param random random, must not be null
056     * @param insertion insertion mutation strategy, must not be null
057     * @param insertionRate insertion rate, insertionRate and deletionRate must sum to 1.0d
058     * @param deletion deletion mutation strategy, must not be null
059     * @param deletionRate deletion rate, insertionRate and deletionRate must sum to 1.0d
060     */
061    public IndelMutationStrategy(final RandomGenerator random, final InsertionMutationStrategy insertion, final double insertionRate, final DeletionMutationStrategy deletion, final double deletionRate) {
062        checkNotNull(random);
063        checkNotNull(insertion);
064        checkNotNull(deletion);
065        checkArgument(1.0d - insertionRate - deletionRate < 0.01d, "insertionRate and deletionRate must sum to 1.0d");
066        this.random = random;
067        this.insertion = insertion;
068        this.insertionRate = insertionRate;
069        this.deletion = deletion;
070    }
071
072
073    @Override
074    public SymbolList mutate(final Symbol symbol) {
075        return random.nextDouble() < insertionRate ? insertion.mutate(symbol) : deletion.mutate(symbol);
076    }
077}
