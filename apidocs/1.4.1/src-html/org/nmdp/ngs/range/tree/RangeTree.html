023package org.nmdp.ngs.range.tree;
024
025import java.util.Set;
026
027import com.google.common.collect.Range;
028
029/**
030 * Range tree.
031 *
032 * @param <C> range endpoint type
033 */
034public interface RangeTree<C extends Comparable> {
035
036    /**
037     * Return the number of ranges in this range tree.
038     *
039     * @return the number of ranges in this range tree
040     */
041    int size();
042
043    /**
044     * Return true if the number of ranges in this range tree is zero.
045     *
046     * @return true if the number of ranges in this range tree is zero
047     */
048    boolean isEmpty();
049
050    /**
051     * Return true if the specified location intersects with any ranges in this range tree.
052     *
053     * @param location location to intersect
054     * @return true if the specified location intersects with any ranges in this range tree
055     */
056    boolean contains(C location);
057
058    /**
059     * Return the number of ranges in this range tree at the specified location.
060     *
061     * @param location location
062     * @return the number of ranges in this range tree at the specified location
063     */
064    int count(C location);
065
066    /**
067     * Return the ranges in this range tree at the specified location, if any.
068     *
069     * @param location location
070     * @return the ranges in this range tree at the specified location, if any
071     */
072    Iterable<Range<C>> query(C location);
073
074    /**
075     * Return the number of ranges in this range tree that intersect the specified query range.
076     *
077     * @param query range to intersect, must not be null
078     * @return the number of ranges in this range tree that intersect the specified query range
079     */
080    int count(Range<C> query);
081
082    /**
083     * Return the ranges in this range tree that intersect the specified query range, if any.
084     *
085     * @param query range to intersect, must not be null
086     * @return the ranges in this range tree that intersect the specified query range, if any
087     */
088    Iterable<Range<C>> intersect(Range<C> query);
089
090    /**
091     * Return true if the specified query range intersects with any ranges in this range tree.
092     *
093     * @param query range to intersect, must not be null
094     * @return true if the specified query range intersects with any ranges in this range tree
095     */
096    boolean intersects(Range<C> query);
097
098    /**
099     * Return the intersection of the ranges in this range tree with the specified query list of
100     * ranges as intersecting pairs of ranges, if any.
101     *
102     * @param query list of ranges to intersect, must not be null
103     * @return the intersection of the ranges in this range tree with the specified query list of
104     *    ranges as intersecting pairs of ranges, if any
105     */
106    Iterable<Set<Range<C>>> intersect(Iterable<Range<C>> query);
107
108    /**
109     * Return true if any range in the specified query list of ranges intersects
110     * with any ranges in this range tree.
111     *
112     * @param query list of ranges to intersect, must not be null
113     * @return true if any range in the specified query list of ranges intersects
114     *    with any ranges in this range tree
115     */
116    boolean intersects(Iterable<Range<C>> query);
117
118    /*
119
120       Additional queries:
121
122       Iterable<Range<C>> closest(Range<C> query);
123       Iterable<Set<Range<C>>> closest(Iterable<Range<C>> query);
124
125       bedtools closest options:
126         -s Force strandedness. That is, find the closest feature in B overlaps A on the
127              same strand. By default, this is disabled.
128         -d In addition to the closest feature in B, report its distance to A as an extra
129              column. The reported distance for overlapping features will be 0.
130         -t How ties for closest feature should be handled. This occurs when two features
131              in B have exactly the same overlap with a feature in A. By default, all such
132              features in B are reported.
133
134              Here are the other choices controlling how ties are handled:
135                all- Report all ties (default).
136                first- Report the first tie that occurred in the B file.
137                last- Report the last tie that occurred in the B file.
138
139       see
140       http://bedtools.readthedocs.org/en/latest/content/tools/closest.html
141
142     */
143}
