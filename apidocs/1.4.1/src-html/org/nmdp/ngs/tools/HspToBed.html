023package org.nmdp.ngs.tools;
024
025import static org.dishevelled.compress.Readers.reader;
026import static org.dishevelled.compress.Writers.writer;
027
028import java.io.BufferedReader;
029import java.io.File;
030import java.io.PrintWriter;
031import java.io.IOException;
032
033import org.dishevelled.commandline.ArgumentList;
034import org.dishevelled.commandline.CommandLine;
035import org.dishevelled.commandline.CommandLineParseException;
036import org.dishevelled.commandline.CommandLineParser;
037import org.dishevelled.commandline.Switch;
038import org.dishevelled.commandline.Usage;
039
040import org.dishevelled.commandline.argument.FileArgument;
041import org.dishevelled.commandline.argument.StringArgument;
042
043/**
044 * Convert HSPs to BED format.
045 */
046public final class HspToBed implements Runnable {
047    private final String displayName;
048    private final File hspFile;
049    private final File bedFile;
050    private final boolean reverse;
051    private final boolean transformEvalue;
052    private static final String USAGE = "java HspToBed [options] -i result.txt -o result.bed";
053
054
055    /**
056     * Convert HSPs to BED format.
057     *
058     * @param displayName query display name, if any
059     * @param hspFile input HSP file, from e.g. blastn -outfmt 7, if any
060     * @param bedFile output BED file, if any
061     * @param reverse true to reverse query and target in BED file
062     * @param transformEvalue true to transform e-value to BED score [0..1000]
063     */
064    public HspToBed(final String displayName, final File hspFile, final File bedFile, final boolean reverse, final boolean transformEvalue) {
065        this.displayName = displayName;
066        this.hspFile = hspFile;
067        this.bedFile = bedFile;
068        this.reverse = reverse;
069        this.transformEvalue = transformEvalue;
070    }
071
072
073    @Override
074    public void run() {
075        BufferedReader reader = null;
076        PrintWriter writer = null;
077        try {
078            reader = reader(hspFile);
079            writer = writer(bedFile);
080
081            while (reader.ready()) {
082                String line = reader.readLine();
083                if (line == null) {
084                    break;
085                }
086                if (line.startsWith("#")) {
087                    continue;
088                }
089
090                String[] tokens = line.split("\t");
091                String query = tokens[0];
092                String chr = tokens[1];
093                // todo:  was this left out on purpose?
094                //double identity = Double.parseDouble(tokens[2]);
095                int alignmentLength = Integer.parseInt(tokens[3]);
096                int mismatches = Integer.parseInt(tokens[4]);
097                int gapOpens = Integer.parseInt(tokens[5]);
098                long queryStart = Long.parseLong(tokens[6]);
099                long queryEnd = Long.parseLong(tokens[7]);
100                long start = Long.parseLong(tokens[8]);
101                long end = Long.parseLong(tokens[9]);
102                double evalue = Double.parseDouble(tokens[10]);
103                double bitscore = Double.parseDouble(tokens[11].trim());
104
105                if (reverse) {
106                    writer.print(displayName != null ? displayName : query);
107                    writer.print("\t");
108                    writer.print(queryStart);
109                    writer.print("\t");
110                    writer.print(queryEnd);
111                    writer.print("\t");
112
113                    writer.print(chr);
114                    writer.print(":");
115                    writer.print(start);
116                    writer.print(":");
117                    writer.print(end);
118                    writer.print(":");
119                    writer.print(start <= end ? "+" : "-");
120                    writer.print(":");
121                    writer.print(alignmentLength);
122                    writer.print(":");
123                    writer.print(mismatches);
124                    writer.print(":");
125                    writer.print(gapOpens);
126                    writer.print(":");
127                    writer.print(evalue);
128                    writer.print(":");
129                    writer.print(bitscore);
130                    writer.print("\t");
131
132                    if (transformEvalue) {
133                        writer.print(transform(evalue));
134                    }
135                    else {
136                        writer.print(evalue);
137                    }
138
139                    writer.print("\t");
140                    writer.print(queryStart <= queryEnd ? "+" : "-");
141                    writer.print("\n");
142                }
143                else {
144                    writer.print(chr);
145                    writer.print("\t");
146                    writer.print(start);
147                    writer.print("\t");
148                    writer.print(end);
149                    writer.print("\t");
150                    writer.print(displayName != null ? displayName : query);
151                    writer.print(":");
152                    writer.print(queryStart);
153                    writer.print(":");
154                    writer.print(queryEnd);
155                    writer.print(":");
156                    writer.print(queryStart <= queryEnd ? "+" : "-");
157                    writer.print(":");
158                    writer.print(alignmentLength);
159                    writer.print(":");
160                    writer.print(mismatches);
161                    writer.print(":");
162                    writer.print(gapOpens);
163                    writer.print(":");
164                    writer.print(evalue);
165                    writer.print(":");
166                    writer.print(bitscore);
167                    writer.print("\t");
168
169                    if (transformEvalue) {
170                        writer.print(transform(evalue));
171                    }
172                    else {
173                        writer.print(evalue);
174                    }
175
176                    writer.print("\t");
177                    writer.print(start <= end ? "+" : "-");
178                    writer.print("\n");
179                }
180            }
181        }
182        catch (IOException e) {
183            e.printStackTrace();
184            System.exit(-1);
185        }
186        finally {
187            try {
188                reader.close();
189            }
190            catch (Exception e) {
191                // empty
192            }
193            try {
194                writer.close();
195            }
196            catch (Exception e) {
197                // empty
198            }
199        }
200    }
201
202    private static int transform(final double evalue) {
203        double transformedEvalue = -Math.log10(evalue) + 2.0d;
204        if (Double.isInfinite(transformedEvalue)) {
205            transformedEvalue = 257.0d; // 255.0 + 2.0d
206        }
207        transformedEvalue = lerp(transformedEvalue, 0.0d, 257.0d, 0.0d, 1000.0d);
208        return (int) transformedEvalue;
209    }
210
211    private static double lerp(final double value, final double sourceMinimum, final double sourceMaximum, final double targetMinimum, final double targetMaximum) {
212        return targetMinimum + (targetMaximum - targetMinimum) * ((value - sourceMinimum) / (sourceMaximum - sourceMinimum));
213    }
214
215
216    /**
217     * Main.
218     *
219     * @param args command line args
220     */
221    public static void main(final String[] args) {
222        Switch help = new Switch("h", "help", "display help message");
223        StringArgument displayName = new StringArgument("d", "display-name", "query display name", false);
224        FileArgument hspFile = new FileArgument("i", "hsp-file", "input HSP file, from e.g. blastn -outfmt 7, default stdin", false);
225        FileArgument bedFile = new FileArgument("o", "bed-file", "output BED file, default stdout", false);
226        Switch reverse = new Switch("r", "reverse", "reverse query and target in BED file");
227        Switch transformEvalue = new Switch("t", "transform-evalue", "transform e-value to BED score [0..1000]");
228
229        ArgumentList arguments = new ArgumentList(help, displayName, hspFile, bedFile, reverse, transformEvalue);
230        CommandLine commandLine = new CommandLine(args);
231        try {
232            CommandLineParser.parse(commandLine, arguments);
233            if (help.wasFound()) {
234                Usage.usage(USAGE, null, commandLine, arguments, System.out);
235                System.exit(-2);
236            }
237            new HspToBed(displayName.getValue(), hspFile.getValue(), bedFile.getValue(), reverse.wasFound(), transformEvalue.wasFound()).run();
238        }
239        catch (CommandLineParseException e) {
240            Usage.usage(USAGE, e, commandLine, arguments, System.err);
241            System.exit(-1);
242        }
243    }
