025import static com.google.common.base.Preconditions.checkArgument;
026import static com.google.common.base.Preconditions.checkNotNull;
027
028import static org.dishevelled.compress.Sources.charSource;
029import static org.dishevelled.compress.Writers.writer;
030
031import java.io.File;
032import java.io.PrintWriter;
033import java.io.IOException;
034
035import java.util.Collections;
036import java.util.List;
037
038import java.util.regex.Pattern;
039
040import com.google.common.collect.Lists;
041import com.google.common.collect.Ordering;
042
043import org.biojava.bio.program.fastq.Fastq;
044import org.biojava.bio.program.fastq.SangerFastqReader;
045import org.biojava.bio.program.fastq.StreamListener;
046
047import org.dishevelled.commandline.ArgumentList;
048import org.dishevelled.commandline.CommandLine;
049import org.dishevelled.commandline.CommandLineParseException;
050import org.dishevelled.commandline.CommandLineParser;
051import org.dishevelled.commandline.Switch;
052import org.dishevelled.commandline.Usage;
053
054import org.dishevelled.commandline.argument.FileArgument;
055import org.dishevelled.commandline.argument.IntegerArgument;
056
057/**
058 * Convert sequences in FASTQ format to SSAKE import format.
059 */
060public final class FastqToSsake implements Runnable {
061    private final int insertSize;
062    private final File firstFastqFile;
063    private final File secondFastqFile;
064    private final File ssakeFile;
065    private final File unpairedFile;
066    public static final int DEFAULT_INSERT_SIZE = 500;
067    private static final Pattern LEFT = Pattern.compile("^.* 1.*$");
068    private static final Pattern RIGHT = Pattern.compile("^.* 2.*$");
069    private static final String USAGE = "java FastqToSsake -1 foo_1.fq -2 foo_2.fq [args]\n\n   Note:  the contents of both FASTQ files are read into RAM.\n   Increase RAM to the JVM using e.g. -Xms2g -Xmx8g if necessary.";
070
071
072    /**
073     * Convert sequences in FASTQ format to SSAKE import format.
074     *
075     * @param firstFastqFile first FASTQ input file, must not be null
076     * @param secondFastqFile second FASTQ input file, must not be null
077     * @param ssakeFile output SSAKE file, if any
078     * @param insertSize insert size, must be at least 0
079     * @param unpairedFile file to write unpaired read names to, if any
080     */
081    public FastqToSsake(final File firstFastqFile, final File secondFastqFile, final File ssakeFile, final int insertSize, final File unpairedFile) {
082        checkNotNull(firstFastqFile);
083        checkNotNull(secondFastqFile);
084        checkArgument((insertSize > -1), "insertSize must be at least 0");
085        this.insertSize = insertSize;
086        this.firstFastqFile = firstFastqFile;
087        this.secondFastqFile = secondFastqFile;
088        this.ssakeFile = ssakeFile;
089        this.unpairedFile = unpairedFile;
090    }
091
092
093    @Override
094    public void run() {
095        PrintWriter ssakeWriter = null;
096        PrintWriter unpairedWriter = null;
097        try {
098            ssakeWriter = writer(ssakeFile);
099            unpairedWriter = (unpairedFile == null) ? null : writer(unpairedFile);
100
101            // read both FASTQ files into RAM (ick)
102            final List<Fastq> reads = Lists.newArrayList();
103            SangerFastqReader fastqReader = new SangerFastqReader();
104            fastqReader.stream(charSource(firstFastqFile), new StreamListener() {
105                    @Override
106                    public void fastq(final Fastq fastq) {
107                        reads.add(fastq);
108                    }
109                });
110            fastqReader.stream(charSource(secondFastqFile), new StreamListener() {
111                    @Override
112                    public void fastq(final Fastq fastq) {
113                        reads.add(fastq);
114                    }
115                });
116
117            // .. and sort by description
118            Collections.sort(reads, new Ordering<Fastq>() {
119                    @Override
120                    public int compare(final Fastq left, final Fastq right) {
121                        return left.getDescription().compareTo(right.getDescription());
122                    }
123                });
124
125            for (int i = 0, size = reads.size(); i < size; ) {
126                Fastq left = reads.get(i);
127                if ((i + 1) == size) {
128                    if (unpairedWriter != null) {
129                        unpairedWriter.println(left.getDescription());
130                    }
131                    break;
132                }
133                Fastq right = reads.get(i + 1);
134
135                if (isLeft(left)) {
136                    if (isRight(right)) {
137                        // write paired reads to SSAKE-hacked-up version of FASTA format
138                        StringBuilder sb = new StringBuilder(512);
139                        sb.append(">");
140                        sb.append(left.getDescription());
141                        sb.append(":");
142                        sb.append(right.getDescription());
143                        sb.append(":");
144                        sb.append(insertSize);
145                        sb.append("\n");
146                        sb.append(left.getSequence());
147                        sb.append(":");
148                        sb.append(right.getSequence());
149                        ssakeWriter.println(sb.toString());
150                        i += 2;
151                    }
152                    else {
153                        if (unpairedWriter != null) {
154                            unpairedWriter.println(right.getDescription());
155                        }
156                        i++;
157                    }
158                }
159                else {
160                    if (unpairedWriter != null) {
161                        unpairedWriter.println(left.getDescription());
162                    }
163                    i++;
164                }
165            }
166        }
167        catch (IOException e) {
168            e.printStackTrace();
169            System.exit(1);
170        }
171        finally {
172            try {
173                ssakeWriter.close();
174            }
175            catch (Exception e) {
176                // ignore
177            }
178            try {
179                unpairedWriter.close();
180            }
181            catch (Exception e) {
182                // ignore
183            }
184        }
185    }
186
187    static boolean isLeft(final Fastq fastq) {
188        return LEFT.matcher(fastq.getDescription()).matches();
189    }
190
191    static boolean isRight(final Fastq fastq) {
192        return RIGHT.matcher(fastq.getDescription()).matches();
193    }
194
195    /**
196     * Main.
197     *
198     * @param args command line args
199     */
200    public static void main(final String[] args) {
201        Switch help = new Switch("h", "help", "display help message");
202        FileArgument firstFastqFile = new FileArgument("1", "first-fastq-file", "first FASTQ input file", true);
203        FileArgument secondFastqFile = new FileArgument("2", "second-fastq-file", "second FASTQ input file", true);
204        FileArgument ssakeFile = new FileArgument("o", "ssake-file", "output SSAKE file, default stdout", false);
205        FileArgument unpairedFile = new FileArgument("u", "unpaired-file", "write unpaired read names to file", false);
206        IntegerArgument insertSize = new IntegerArgument("s", "insert-size", "insert size, must be at least zero, default " + DEFAULT_INSERT_SIZE, false);
207
208        ArgumentList arguments = new ArgumentList(help, firstFastqFile, secondFastqFile, ssakeFile, insertSize, unpairedFile);
209        CommandLine commandLine = new CommandLine(args);
210        try
211        {
212            CommandLineParser.parse(commandLine, arguments);
213            if (help.wasFound()) {
214                Usage.usage(USAGE, null, commandLine, arguments, System.out);
215                System.exit(-2);
216            }
217            new FastqToSsake(firstFastqFile.getValue(), secondFastqFile.getValue(), ssakeFile.getValue(), insertSize.getValue(DEFAULT_INSERT_SIZE), unpairedFile.getValue()).run();
218        }
219        catch (CommandLineParseException e) {
220            Usage.usage(USAGE, e, commandLine, arguments, System.err);
221            System.exit(-1);
222        }
223    }
224}
