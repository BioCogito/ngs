023package org.nmdp.ngs.sra;
024
025import java.util.Map;
026
027import com.google.common.collect.ImmutableMap;
028import com.google.common.collect.Maps;
029
030/**
031 * File type.
032 */
033public enum FileType {
034    SRF("srf"),
035    SFF("sff"),
036    FASTQ("fastq"),
037    CRAM("cram"),
038    BAM("bam"),
039    ILLUMINA_NATIVE_QSEQ("Illumina_native_qseq"),
040    ILLUMINA_NATIVE_SCARF("Illumina_native_scarf"),
041    ILLUMINA_NATIVE_FASTQ("Illumina_native_fastq"),
042    SOLID_NATIVE_CSFASTA("SOLiD_native_csfasta"),
043    SOLID_NATIVE_QUAL("SOLiD_native_qual"),
044    PACBIO_HDF5("PacBio_HDF5"),
045    COMPLETE_GENOMICS_NATIVE("CompleteGenomics_native");
046
047    private final String description;
048    private static final Map<String, FileType> BY_DESCRIPTION;
049
050    static
051    {
052        Map<String, FileType> byDescription = Maps.newHashMap();
053        for (FileType fileType : values()) {
054            byDescription.put(fileType.getDescription(), fileType);
055        }
056        BY_DESCRIPTION = ImmutableMap.copyOf(byDescription);
057    }
058
059
060    /**
061     * Create a new file type with the specified description.
062     *
063     * @param description description
064     */
065    private FileType(final String description) {
066        this.description = description;
067    }
068
069
070    /**
071     * Return the description of this file type.
072     *
073     * @return the description of this file type
074     */
075    public String getDescription() {
076        return description;
077    }
078
079    /**
080     * Return a map of file types keyed by description.
081     *
082     * @return a map of file types keyed by description
083     */
084    public static Map<String, FileType> byDescription() {
085        return BY_DESCRIPTION;
086    }
087
088    /**
089     * Return the file type with the specified descripton, or null if no such file type exists.
090     *
091     * @param description description
092     * @return the file type with the specified descripton, or null if no such file type exists
093     */
094    public static FileType fromDescription(final String description) {
095        return BY_DESCRIPTION.get(description);
096    }
097}
