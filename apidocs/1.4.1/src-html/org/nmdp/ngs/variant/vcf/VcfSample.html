023package org.nmdp.ngs.variant.vcf;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import javax.annotation.concurrent.Immutable;
028
029/**
030 * VCF sample.
031 */
032@Immutable
033public final class VcfSample {
034    /** Id. */
035    private final String id;
036
037    /** Array of VCF genomes. */
038    private final VcfGenome[] genomes;
039
040
041    /**
042     * Create a new VCF sample.
043     *
044     * @param id id, must not be null
045     * @param genomes VCF genomes, must not be null
046     */
047    VcfSample(final String id, final VcfGenome... genomes) {
048        checkNotNull(id);
049        checkNotNull(genomes);
050        this.id = id;
051        this.genomes = genomes;
052    }
053
054
055    /**
056     * Return the id for this VCF sample.
057     *
058     * @return the id for this VCF sample.
059     */
060    public String getId() {
061        return id;
062    }
063
064    /**
065     * Return the VCF genomes for this VCF sample.
066     *
067     * @return the VCF genomes for this VCF sample
068     */
069    public VcfGenome[] getGenomes() {
070        return genomes;
071    }
072}
