060/**
061 * Convert first and second sequence files in FASTQ format to interleaved FASTQ format.
062 */
063public final class InterleaveFastq3 implements Runnable {
064    private final File firstFastqFile;
065    private final File secondFastqFile;
066    private final File pairedFile;
067    private final File unpairedFile;
068    private final FastqWriter fastqWriter = new SangerFastqWriter();
069    private static final String USAGE = "java InterleaveFastq -1 foo_1.fq.gz -2 foo_2.fq.gz --paired foo.paired.fq.gz --unpaired foo.unpaired.fq.gz";
070
071
072    /**
073     * Convert first and second sequence files in FASTQ format to interleaved FASTQ format.
074     *
075     * @param firstFastqFile first FASTQ input file, must not be null
076     * @param secondFastqFile second FASTQ input file, must not be null
077     * @param pairedFile output interleaved paired FASTQ file, must not be null
078     * @param unpairedFile output interleaved unpaired FASTQ file, must not be null
079     */
080    public InterleaveFastq3(final File firstFastqFile, final File secondFastqFile, final File pairedFile, final File unpairedFile) {
081        checkNotNull(firstFastqFile);
082        checkNotNull(secondFastqFile);
083        checkNotNull(pairedFile);
084        checkNotNull(unpairedFile);
085        this.firstFastqFile = firstFastqFile;
086        this.secondFastqFile = secondFastqFile;
087        this.pairedFile = pairedFile;
088        this.unpairedFile = unpairedFile;
089    }
090
091
092    @Override
093    public void run() {
094        BufferedReader firstReader = null;
095        BufferedReader secondReader = null;
096        PrintWriter pairedWriter = null;
097        PrintWriter unpairedWriter = null;
098        try {
099            firstReader = reader(firstFastqFile);
100            secondReader = reader(secondFastqFile);
101            pairedWriter = writer(pairedFile);
102            unpairedWriter = writer(unpairedFile);
103
104            final PrintWriter pw = pairedWriter;
105            final PrintWriter uw = unpairedWriter;
106
107            final FastqWriter pairedFastqWriter = new SangerFastqWriter();
108            final FastqWriter unpairedFastqWriter = new SangerFastqWriter();
109
110            PairedEndFastqReader.streamPaired(firstReader, secondReader, new PairedEndAdapter() {
111                    @Override
112                    public void paired(final Fastq left, final Fastq right) {
113                        try {
114                            pairedFastqWriter.append(pw, left);
115                            pairedFastqWriter.append(pw, right);
116                        }
117                        catch (IOException e) {
118                            throw new RuntimeException("could not write paired end reads", e);
119                        }
120                    }
121
122                    @Override
123                    public void unpaired(final Fastq unpaired) {
124                        try {
125                            unpairedFastqWriter.append(uw, unpaired);
126                        }
127                        catch (IOException e) {
128                            throw new RuntimeException("could not write unpaired read", e);
129                        }
130                    }
131                });
132        }
133        catch (IOException e) {
134            e.printStackTrace();
135            System.exit(1);
136        }
137        catch (RuntimeException e) {
138            e.printStackTrace();
139            System.exit(2);
140        }
141        finally {
142            try {
143                firstReader.close();
144            }
145            catch (Exception e) {
146                // ignore
147            }
148            try {
149                secondReader.close();
150            }
151            catch (Exception e) {
152                // ignore
153            }
154            try {
155                pairedWriter.close();
156            }
157            catch (Exception e) {
158                // ignore
159            }
160            try {
161                unpairedWriter.close();
162            }
163            catch (Exception e) {
164                // ignore
165            }
166        }
167    }
168
169
170    /**
171     * Main.
172     *
173     * @param args command line args
174     */
175    public static void main(final String[] args) {
176        Switch help = new Switch("h", "help", "display help message");
177        FileArgument firstFastqFile = new FileArgument("1", "first-fastq-file", "first FASTQ input file", true);
178        FileArgument secondFastqFile = new FileArgument("2", "second-fastq-file", "second FASTQ input file", true);
179        FileArgument pairedFile = new FileArgument("p", "paired-file", "output interleaved paired FASTQ file", true);
180        FileArgument unpairedFile = new FileArgument("u", "unpaired-file", "output interleaved unpaired FASTQ file", true);
181
182        ArgumentList arguments = new ArgumentList(help, firstFastqFile, secondFastqFile, pairedFile, unpairedFile);
183        CommandLine commandLine = new CommandLine(args);
184        try {
185            CommandLineParser.parse(commandLine, arguments);
186            if (help.wasFound()) {
187                Usage.usage(USAGE, null, commandLine, arguments, System.out);
188                System.exit(-2);
189            }
190            new InterleaveFastq3(firstFastqFile.getValue(), secondFastqFile.getValue(), pairedFile.getValue(), unpairedFile.getValue()).run();
191        }
192        catch (CommandLineParseException e) {
193            Usage.usage(USAGE, e, commandLine, arguments, System.err);
194            System.exit(-1);
195        }
196    }
