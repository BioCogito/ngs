082/**
083 * Generate paired-end next generation sequencing (NGS/HTS) reads.
084 */
085@SuppressWarnings("deprecation")
086public final class GeneratePairedEndReads implements Runnable {
087    private final File referenceFile;
088    private final File firstReadFile;
089    private final File secondReadFile;
090    private final RandomGenerator random;
091    private final RealDistribution length;
092    private final RealDistribution insertSize;
093    private final QualityStrategy quality;
094    private final CoverageStrategy coverage;
095    private final double mutationRate;
096    private final MutationStrategy mutation;
097
098    private static final double DEFAULT_MEAN_LENGTH = 60.0d;
099    private static final double DEFAULT_LENGTH_VARIATION = 10.0d;
100    private static final double DEFAULT_MEAN_INSERT_SIZE = 500.0d;
101    private static final double DEFAULT_INSERT_SIZE_VARIATION = 50.0d;
102    private static final int DEFAULT_MINIMUM_COVERAGE = 20;
103    private static final double DEFAULT_MEAN_QUALITY = 25.0d;
104    private static final double DEFAULT_QUALITY_VARIATION = 5.0d;
105    private static final double DEFAULT_MEAN_QUALITY_WEIGHT = 0.9d;
106    private static final double DEFAULT_QUALITY_WEIGHT_VARIATION = 0.1d;
107    private static final double DEFAULT_EXTEND_INSERTION_RATE = 0.4d;
108    private static final int DEFAULT_MAXIMUM_INSERTION_LENGTH = 8;
109    private static final double DEFAULT_INSERTION_RATE = 0.5d;
110    private static final double DEFAULT_DELETION_RATE = 0.5d;
111    private static final double DEFAULT_SUBSTITUTION_RATE = 0.4d;
112    private static final double DEFAULT_INDEL_RATE = 0.6d;
113    private static final double DEFAULT_AMBIGUOUS_RATE = 0.0d;
114    private static final double DEFAULT_MUTATION_RATE = 0.05d;
115    private static final double NO_VARIATION = 1.0E-100;
116    private static final CoverageStrategy DEFAULT_COVERAGE = new MinimumCoverageStrategy(DEFAULT_MINIMUM_COVERAGE);
117    private static final MutationStrategy DEFAULT_MUTATION = new IdentityMutationStrategy();
118    private static final String USAGE = "java GeneratePairedEndReads [args] -1 out_1.fq -2 out_2.fq";
119
120
121    private GeneratePairedEndReads(final File referenceFile,
122                                   final File firstReadFile,
123                                   final File secondReadFile,
124                                   final RandomGenerator random,
125                                   final RealDistribution length,
126                                   final RealDistribution insertSize,
127                                   final QualityStrategy quality,
128                                   final CoverageStrategy coverage,
129                                   final double mutationRate,
130                                   final MutationStrategy mutation) {
131
132        checkNotNull(firstReadFile);
133        checkNotNull(secondReadFile);
134        this.referenceFile = referenceFile;
135        this.firstReadFile = firstReadFile;
136        this.secondReadFile = secondReadFile;
137        this.random = random;
138        this.length = length;
139        this.insertSize = insertSize;
140        this.quality = quality;
141        this.coverage = coverage;
142        this.mutationRate = mutationRate;
143        this.mutation = mutation;
144    }
145
146
147    @Override
148    public void run() {
149        BufferedReader reader = null;
150        PrintWriter firstWriter = null;
151        PrintWriter secondWriter = null;
152        try {
153            reader = reader(referenceFile);
154
155            // since writes append to each output file, they must be deleted first
156            firstReadFile.delete();
157            secondReadFile.delete();
158
159            SequenceIterator sequences = SeqIOTools.readFastaDNA(reader);
160            while (sequences.hasNext()) {
161                Sequence sequence = sequences.nextSequence();
162                try {
163                    firstWriter = writer(firstReadFile, true);
164                    secondWriter = writer(secondReadFile, true);
165
166                    new org.nmdp.ngs.reads.GeneratePairedEndReads(sequence, FastqVariant.FASTQ_SANGER, random, length, insertSize, quality, coverage, mutationRate, mutation, firstWriter, secondWriter, new SangerFastqWriter()).run();
167                }
168                catch (IOException e) {
169                    e.printStackTrace();
170                    System.exit(-1);
171                }
172                finally {
173                    try {
174                        firstWriter.flush();
175                    }
176                    catch (Exception e) {
177                        // ignore
178                    }
179                    try {
180                        secondWriter.flush();
181                    }
182                    catch (Exception e) {
183                        // ignore
184                    }
185                }
186            }
187        }
188        catch (IOException e) {
189            e.printStackTrace();
190            System.exit(-1);
191        }
192        catch (BioException e) {
193            e.printStackTrace();
194            System.exit(-1);
195        }
196        finally {
197            try {
198                reader.close();
199            }
200            catch (Exception e) {
201                // ignore
202            }
203            try {
204                firstWriter.close();
205            }
206            catch (Exception e) {
207                // ignore
208            }
209            try {
210                secondWriter.close();
211            }
212            catch (Exception e) {
213                // ignore
214            }
215        }
216    }
217
218
219    /**
220     * Main.
221     *
222     * @param args command line arguments
223     */
224    public static void main(final String[] args) {
225        Switch help = new Switch("h", "help", "display help message");
226        FileArgument referenceFile = new FileArgument("r", "reference", "reference input file, in fasta format, default stdin", false);
227        FileArgument firstReadFile = new FileArgument("1", "first-read", "first read output file, in fastq format", true);
228        FileArgument secondReadFile = new FileArgument("2", "second-read", "second read output file, in fastq format", true);
229        DoubleArgument meanLength = new DoubleArgument("l", "mean-length", "mean length, default " + DEFAULT_MEAN_LENGTH, false);
230        DoubleArgument lengthVariation = new DoubleArgument("v", "length-variation", "length variation, default " + DEFAULT_LENGTH_VARIATION, false);
231        DoubleArgument meanInsertSize = new DoubleArgument("j", "mean-insert-size", "mean insert size, default " + DEFAULT_MEAN_INSERT_SIZE, false);
232        DoubleArgument insertSizeVariation = new DoubleArgument("k", "insert-size-variation", "insert size variation, default " + DEFAULT_INSERT_SIZE_VARIATION, false);
233        IntegerArgument minimumCoverage = new IntegerArgument("c", "minimum-coverage", "minimum coverage, default " + DEFAULT_MINIMUM_COVERAGE, false);
234        IntegerArgument meanCoverage = new IntegerArgument("g", "mean-coverage", "mean coverage", false);
235        StringArgument qualityType = new StringArgument("u", "quality", "quality strategy type {illumina, normal}, default normal", false);
236        DoubleArgument meanQualityWeight = new DoubleArgument("w", "mean-quality-weight", "mean quality weight, default " + DEFAULT_MEAN_QUALITY_WEIGHT, false);
237        DoubleArgument qualityWeightVariation = new DoubleArgument("t", "quality-weight-variation", "quality weight variation, default " + DEFAULT_QUALITY_WEIGHT_VARIATION, false);
238        DoubleArgument meanQuality = new DoubleArgument("q", "mean-quality", "mean quality, default " + DEFAULT_MEAN_QUALITY, false);
239        DoubleArgument qualityVariation = new DoubleArgument("a", "quality-variation", "quality variation, default " + DEFAULT_QUALITY_VARIATION, false);
240        StringArgument mutationType = new StringArgument("m", "mutation", "mutation strategy type {substitution, insertion, deletion, ambiguous, indel, composite}, default identity", false);
241        DoubleArgument extendInsertionRate = new DoubleArgument("x", "extend-insertion-rate", "extend insertion rate, default " + DEFAULT_EXTEND_INSERTION_RATE, false);
242        IntegerArgument maximumInsertionLength = new IntegerArgument("e", "maximum-insertion-length", "maximum insertion length, default " + DEFAULT_MAXIMUM_INSERTION_LENGTH, false);
243        DoubleArgument insertionRate = new DoubleArgument("i", "insertion-rate", "insertion rate, default " + DEFAULT_INSERTION_RATE, false);
244        DoubleArgument deletionRate = new DoubleArgument("d", "deletion-rate", "deletion rate, default " + DEFAULT_DELETION_RATE, false);
245        DoubleArgument substitutionRate = new DoubleArgument("s", "substitution-rate", "substitution rate, default " + DEFAULT_SUBSTITUTION_RATE, false);
246        DoubleArgument indelRate = new DoubleArgument("y", "indel-rate", "indel rate, default " + DEFAULT_INDEL_RATE, false);
247        DoubleArgument ambiguousRate = new DoubleArgument("b", "ambiguous-rate", "ambiguous substitution rate, default " + DEFAULT_AMBIGUOUS_RATE, false);
248        DoubleArgument mutationRate = new DoubleArgument("n", "mutation-rate", "mutation rate, default " + DEFAULT_MUTATION_RATE, false);
249        IntegerArgument seed = new IntegerArgument("z", "seed", "random number seed, default relates to current time", false);
250
251        ArgumentList arguments = new ArgumentList(help, referenceFile, firstReadFile, secondReadFile, meanLength, lengthVariation,
252                                                  meanInsertSize, insertSizeVariation, minimumCoverage, meanCoverage,
253                                                  qualityType, meanQualityWeight, qualityWeightVariation, meanQuality, qualityVariation,
254                                                  mutationType, extendInsertionRate, maximumInsertionLength, insertionRate, deletionRate,
255                                                  substitutionRate, indelRate, ambiguousRate, mutationRate, seed);
256
257        CommandLine commandLine = new CommandLine(args);
258        try {
259            CommandLineParser.parse(commandLine, arguments);
260            if (help.wasFound()) {
261                Usage.usage(USAGE, null, commandLine, arguments, System.out);
262                System.exit(-2);
263            }
264
265            RandomGenerator random = seed.wasFound() ? new MersenneTwister(seed.getValue()) : new MersenneTwister();
266
267            double lv = Math.max(NO_VARIATION, lengthVariation.getValue(DEFAULT_LENGTH_VARIATION));
268            RealDistribution length = new NormalDistribution(random, meanLength.getValue(DEFAULT_MEAN_LENGTH), lv, NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
269
270            double isv = Math.max(NO_VARIATION, insertSizeVariation.getValue(DEFAULT_INSERT_SIZE_VARIATION));
271            RealDistribution insertSize = new NormalDistribution(random, meanInsertSize.getValue(DEFAULT_MEAN_INSERT_SIZE), isv, NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
272
273            CoverageStrategy coverage = DEFAULT_COVERAGE;
274            if (minimumCoverage.wasFound()) {
275                coverage = new MinimumCoverageStrategy(minimumCoverage.getValue());
276            }
277            else if (meanCoverage.wasFound()) {
278                coverage = new MeanCoverageStrategy(meanCoverage.getValue());
279            }
280
281            QualityStrategy quality = null;
282            if ("illumina".equals(qualityType.getValue())) {
283                RealDistribution realDistribution = new NormalDistribution(random, meanQualityWeight.getValue(DEFAULT_MEAN_QUALITY_WEIGHT), qualityWeightVariation.getValue(DEFAULT_QUALITY_WEIGHT_VARIATION), NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
284                quality = new ScoreFunctionQualityStrategy(realDistribution, ScoreFunctions.illumina());
285            }
286            else {
287                RealDistribution realDistribution = new NormalDistribution(random, meanQuality.getValue(DEFAULT_MEAN_QUALITY), qualityVariation.getValue(DEFAULT_QUALITY_VARIATION), NormalDistribution.DEFAULT_INVERSE_ABSOLUTE_ACCURACY);
288                quality = new RealDistributionQualityStrategy(realDistribution);
289            }
290
291            MutationStrategy mutation = DEFAULT_MUTATION;
292            if (mutationType.wasFound()) {
293                if ("substitution".equals(mutationType.getValue())) {
294                    mutation = new SubstitutionMutationStrategy(random);
295                }
296                else if ("ambiguous".equals(mutationType.getValue())) {
297                    mutation = new AmbiguousSubstitutionMutationStrategy();
298                }
299                else if ("insertion".equals(mutationType.getValue())) {
300                    mutation = new InsertionMutationStrategy(random, extendInsertionRate.getValue(DEFAULT_EXTEND_INSERTION_RATE), maximumInsertionLength.getValue(DEFAULT_MAXIMUM_INSERTION_LENGTH));
301                }
302                else if ("deletion".equals(mutationType.getValue())) {
303                    mutation = new DeletionMutationStrategy();
304                }
305                else if ("indel".equals(mutationType.getValue())) {
306                    InsertionMutationStrategy insertion = new InsertionMutationStrategy(random, insertionRate.getValue(DEFAULT_INSERTION_RATE), maximumInsertionLength.getValue(DEFAULT_MAXIMUM_INSERTION_LENGTH));
307                    DeletionMutationStrategy deletion = new DeletionMutationStrategy();
308                    mutation = new IndelMutationStrategy(random, insertion, insertionRate.getValue(DEFAULT_INSERTION_RATE), deletion, deletionRate.getValue(DEFAULT_DELETION_RATE));
309                }
310                else if ("composite".equals(mutationType.getValue())) {
311                    SubstitutionMutationStrategy substitution = new SubstitutionMutationStrategy(random);
312                    InsertionMutationStrategy insertion = new InsertionMutationStrategy(random, insertionRate.getValue(DEFAULT_INSERTION_RATE), maximumInsertionLength.getValue(DEFAULT_MAXIMUM_INSERTION_LENGTH));
313                    DeletionMutationStrategy deletion = new DeletionMutationStrategy();
314                    IndelMutationStrategy indel = new IndelMutationStrategy(random, insertion, insertionRate.getValue(DEFAULT_INSERTION_RATE), deletion, deletionRate.getValue(DEFAULT_DELETION_RATE));
315                    AmbiguousSubstitutionMutationStrategy ambiguous = new AmbiguousSubstitutionMutationStrategy();
316                    mutation = new CompositeMutationStrategy(random, substitution, substitutionRate.getValue(DEFAULT_SUBSTITUTION_RATE), indel, indelRate.getValue(DEFAULT_INDEL_RATE), ambiguous, ambiguousRate.getValue(DEFAULT_AMBIGUOUS_RATE));
317                }
318            }
319
320            new GeneratePairedEndReads(referenceFile.getValue(), firstReadFile.getValue(), secondReadFile.getValue(), random, length, insertSize, quality, coverage, mutationRate.getValue(DEFAULT_MUTATION_RATE), mutation).run();
321        }
322        catch (CommandLineParseException e) {
323            Usage.usage(USAGE, e, commandLine, arguments, System.err);
324            System.exit(-1);
325        }
326    }
