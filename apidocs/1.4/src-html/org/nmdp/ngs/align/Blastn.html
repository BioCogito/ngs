023package org.nmdp.ngs.align;
024
025import static com.google.common.base.Preconditions.checkNotNull;
026
027import static org.dishevelled.compress.Sources.charSource;
028
029import java.io.BufferedReader;
030import java.io.File;
031import java.io.FileReader;
032import java.io.IOException;
033
034import java.nio.charset.Charset;
035
036import java.util.List;
037
038import com.google.common.collect.ImmutableList;
039import com.google.common.collect.Lists;
040
041import com.google.common.io.Files;
042
043/**
044 * Utility to run blastn via the command line.
045 */
046public final class Blastn {
047
048    /**
049     * Private no-arg constructor.
050     */
051    private Blastn() {
052        // empty
053    }
054
055    /**
056     * Return the high-scoring segment pairs (HSPs) from blastn of the source and target sequence files in FASTA format.
057     *
058     * @param sourceFile source sequence file in FASTA format, must not be null
059     * @param targetFile target sequence file in FASTA format, must not be null
060     * @return zero or more high-scoring segment pairs (HSPs) from blastn of the source and target sequence files
061     *    in FASTA format
062     * @throws IOException if an I/O error occurs
063     */
064    public static Iterable<HighScoringPair> blastn(final File sourceFile, final File targetFile) throws IOException {
065        checkNotNull(sourceFile);
066        checkNotNull(targetFile);
067
068        File blastResult = File.createTempFile("blastn", ".txt");
069
070        // blastn can't handle compressed files, so copy source and target to temp files, decompressing if necessary
071        File sourceFileCopy = File.createTempFile("sourceFile", ".fa");
072        charSource(sourceFile).copyTo(Files.asCharSink(sourceFileCopy, Charset.forName("UTF-8")));
073
074        File targetFileCopy = File.createTempFile("targetFile", ".fa");
075        charSource(targetFile).copyTo(Files.asCharSink(targetFileCopy, Charset.forName("UTF-8")));
076
077        ProcessBuilder blastn = new ProcessBuilder("blastn",
078                                                   "-subject", sourceFileCopy.getPath(),
079                                                   "-query", targetFileCopy.getPath(),
080                                                   "-outfmt", "6",
081                                                   "-out", blastResult.getPath());
082        Process blastnProcess = blastn.start();
083        try {
084            blastnProcess.waitFor();
085        }
086        catch (InterruptedException e) {
087            // ignore
088        }
089
090        BufferedReader reader = null;
091        List<HighScoringPair> hsps = Lists.newLinkedList();
092        try {
093            reader = new BufferedReader(new FileReader(blastResult));
094            while (reader.ready()) {
095                String line = reader.readLine();
096                if (line == null) {
097                    break;
098                }
099                String[] tokens = line.split("\t");
100                if (tokens.length == 12) {
101                    String source = tokens[0].trim();
102                    String target = tokens[1].trim();
103                    double percentIdentity = Double.parseDouble(tokens[2].trim());
104                    long alignmentLength = Long.parseLong(tokens[3].trim());
105                    int mismatches = Integer.parseInt(tokens[4].trim());
106                    int gapOpens = Integer.parseInt(tokens[5].trim());
107                    long sourceStart = Long.parseLong(tokens[6].trim());
108                    long sourceEnd = Long.parseLong(tokens[7].trim());
109                    long targetStart = Long.parseLong(tokens[8].trim());
110                    long targetEnd = Long.parseLong(tokens[9].trim());
111                    double evalue = Double.parseDouble(tokens[10].trim());
112                    double bitScore = Double.parseDouble(tokens[11].trim());
113
114                    hsps.add(new HighScoringPair(source, target, percentIdentity, alignmentLength, mismatches, gapOpens, sourceStart, sourceEnd, targetStart, targetEnd, evalue, bitScore));
115                }
116            }
117        }
118        finally {
119            try {
120                reader.close();
121            }
122            catch (Exception e) {
123                // empty
124            }
125            sourceFileCopy.delete();
126            targetFileCopy.delete();
127        }
128        return ImmutableList.copyOf(hsps);
129    }
130}
